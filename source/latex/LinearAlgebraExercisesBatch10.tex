\begin{snippetexercise}{linear-algebra-batch-10-ex-5}{}
    A \emph{Vandermonde matrix} is a matrix of the form
    \[V_n = \begin{pmatrix} 1 & x_1 & \dots & x_1^{n-1} \\ 1 & x_2 & \dots & x_2^{n-1} \\ \vdots & & \ddots & \vdots \\ 1 & x_n & \dots & x_n^{n-1} \end{pmatrix} \in \matrices_{n \times n}(\mathbb{F}),\]
    where \(x_1, \dots, x_n \in \mathbb{F}\), \(n \geq 2\).
    Prove that \(\det V_n = \prod_{1 \leq i < j \leq n} (x_j - x_i)\).
\end{snippetexercise}
