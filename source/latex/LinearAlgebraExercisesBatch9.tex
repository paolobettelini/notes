\begin{snippetexercise}{linear-algebra-batch-9-ex-1}{}
    Let \(V\) and \(W\) be \(\mathbb{F}\)-vector spaces and let \(f \in \mathcal{L}(V,W)\).
    We define a function \(f'\colon W' \fromto V'\) by \[f'(\psi) \triangleq \psi \circ f\] for every \(\psi \in W'\). In other words, \((f'(\psi))(v) = \psi(f(v))\) for every \(v \in V\).
    \begin{enumerate}[label=\Alph*]
        \item Prove that \(f' \in \mathcal{L}(W', V')\). \(f'\) is the \emph{dual map} or \emph{transpose} of \(f\).
        \item Let \(\mathcal{B}\) and \(\mathcal{C}\) be bases of \(V\) and \(W\) respectively, and let \(\mathcal{B}'\) and \(\mathcal{C}'\) be the bases of \(V'\) and \(W'\), dual to \(\mathcal{B}\) and \(\mathcal{C}\). Prove that \[\mathcal{M}(f', \mathcal{C}', \mathcal{B}') = \mathcal{M}(f, \mathcal{B}, \mathcal{C})^t.\]
    \end{enumerate}
\end{snippetexercise}

\begin{snippetexercise}{linear-algebra-batch-9-ex-2}{}
    Let \(g\in \text{End}(\realnumbers^2)\) such that
    \(g(1,5) = (3,7) \land g(2,8) = (-1,1)\). Find the \matrix of \(g\)
    in the canonical \basis of \(\realnumbers^2\).
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-9-ex-2-sol}{}
    The wanted \matrix \(M\) has form
    \[
        M = \begin{pmatrix}
            a & b \\
            c & d
        \end{pmatrix}
    \]
    We know that
    \[
        \begin{pmatrix}
            a & b \\
            c & d
        \end{pmatrix} \begin{pmatrix}
            1 \\
            5
        \end{pmatrix} = \begin{pmatrix}
            3 \\
            7
        \end{pmatrix}
        \quad \text{and} \quad
        \begin{pmatrix}
            a & b \\
            c & d
        \end{pmatrix} \begin{pmatrix}
            2 \\
            8
        \end{pmatrix} = \begin{pmatrix}
            -1 \\
            1
        \end{pmatrix}
    \]
    This gives the system
    \[
        \begin{cases}
            a + 5b = 3 \\
            c + 5d = 7 \\
            2a + 8b = -1 \\
            2c + 8d = 1
        \end{cases}
    \]
    We apply gaussian elimination
    \[
        \begin{pmatrix}
            1 & 5 & 0 & 0 & 3 \\
            0 & 0 & 1 & 5 & 7 \\
            2 & 8 & 0 & 0 & -1 \\
            0 & 0 & 2 & 8 & 1
        \end{pmatrix}
        \xrightarrow{\text{RREF}}
        \begin{pmatrix}
            1 & 5 & 0 & 0 & 3 \\
            0 & -2 & 0 & 0 & -7 \\
            0 & 0 & 1 & 5 & 7 \\
            0 & 0 & 0 & -2 & -13
        \end{pmatrix}
    \]
    which gives the matrix
    \[
        M = \begin{pmatrix}
            -\frac{29}{2} & \frac{7}{2} \\
            -\frac{51}{2} & \frac{13}{2}
        \end{pmatrix}
    \]
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-9-ex-3}{}
    Consider the \matrix
    \[
        A = \begin{pmatrix}
            1 & 0 \\ 1 & 0
        \end{pmatrix} \in \matrices_{2\times 2}(\realnumbers)
    \]
    and the endomorphism \(f_A \colon \matrices_{2\times 2}(\realnumbers) \fromto \matrices_{2\times 2}(\realnumbers)\) given by
    \(f_A(x) \triangleq AX-XA\).
    \begin{enumerate}[label=\Alph*]
        \item Determine the matrix associated to \(f_A\) with respect to the basis \(\{E_{1,1}, E_{1,2}, E_{2,1}, E_{2,2}\}\) of \(\matrices_{2\times 2}(\realnumbers)\).
        \item Find a basis for \(\text{ker} f_A\).
        \item Compute the rank of \(f_A\).
        \item Verify that \(\matrices_{2\times 2}(\realnumbers) = \text{ker} f_A \oplus \image f_A\).
    \end{enumerate}
\end{snippetexercise}
