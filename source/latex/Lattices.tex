\section{Lattices}

\begin{snippetdefinition}{lattice-definition}{Lattice}
    A \poset \((L, \leq)\) is said to be a \emph{lattice}
    if \(\forall x,y \in L\),
    \[
        \exists \sup\{x,y\}
        \land
        \exists \inf\{x,y\}
    \]
    \emph{Syntax:} we define the \emph{meet}
    and \emph{join} operations as
    \[
        x \origland y \triangleq \inf\{x,y\},
        \quad
        x \origlor y \triangleq \sup\{x,y\}
    \]
    respectively.
\end{snippetdefinition}
