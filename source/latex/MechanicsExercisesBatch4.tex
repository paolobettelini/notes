\begin{snippetexercise}{mechanics-ex-4.1}{\underline{4.1}}
    Two bodies, \(A\) and \(B\), with respective masses \(M_A\) and \(M_B\),
    where \(M_B>M_A\), slide down an inclined plane (with an inclination angle \(\alpha\)).
    They are in contact with each other,
    with \(B\) positioned higher than \(A\).
    Calculate the acceleration of the system consisting of the two bodies, if the coefficients of friction
    are \(\mu_A\) and \(\mu_B\) respectively. With what force does body \(B\) push body \(A\)?
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.1-sol}{\underline{4.1}}
    We have the equation of motion
    \begin{align*}
        (M_A + M_B)\ddot{x} &= (M_A + M_B)g\sin\alpha - \mu_A M_A g\cos\alpha - \mu_B M_B g \cos \alpha \\
        a &= g\left[
            \sin \alpha - \frac{\mu_A M_A + \mu_BM_B}{M_A + M_B} \cos\alpha
        \right]
    \end{align*}
    Let us write the Newton equation for \(M_A\)
    \begin{align*}
        M_A \vec{a} &= \vec{f} + M_A \vec{g} + \vec{f}_A \\
        M_A a &= f + M_Ag \sin\alpha - \mu_A M_A g \cos\alpha \\
        f &= M_A a - M_A g \sin\alpha + \mu_A M_A g \cos\alpha \\
         &= M_A g \cos\alpha \left[
            \mu_A  - \frac{\mu_A M_A + \mu_B M_B}{M_A + M_B}
        \right] \\
        &= \frac{M_A M_B}{M_A + M_B} g \cos\alpha(\mu_A - \mu_B)
    \end{align*}
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-4.2}{\underline{4.2}}
    A rope passing over a frictionless pulley has two masses, \(M\) and \(m\), attached at its ends,
    with \(M>m\). Determine the acceleration of the system and the tension in the rope.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.2-sol}{\underline{4.2}}
    We have
    \begin{align*}
        M\ddot{x} &= T-Mg \\
        m\ddot{x} &= mg - T
    \end{align*}
    By solving the system we get
    \begin{align*}
        a &= \frac{M-m}{M+m}g \\
        T = \frac{2Mm}{M+m}
    \end{align*}
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-4.3}{\underline{4.3}}
    Assuming all surfaces are frictionless and the inertia of the rope and pulley is negligible,
    find the value of the horizontal force \(F\) such that there is no relative motion between
    the masses \(m_1\), \(m_2\), and \(M\)\\
    \begin{center}
        \begin{tikzpicture}
            % Draw ground line
            \draw[dashed] (-1,0) -- (4,0);
        
            % Draw the large block M
            \draw[thick] (0,0) rectangle (3,2);
            \node at (1.5,1) {$M$};
        
            % Draw the force vector F
            \draw[thick, ->] (-0.5,1) -- (0,1) node[midway, above] {$F$};
        
            % Draw the small block m1 on top of M
            \draw[thick] (1,2) rectangle (2,2.8);
            \node at (1.5,2.4) {$m_1$};
        
            % Draw the small block m2 on the right of M
            \draw[thick] (3,0.5) rectangle (3.8,1.3);
            \node at (3.4,0.9) {$m_2$};
        
            % Draw the pulley system
            \draw[thick] (2,2.4) -- (3.4,2.4) -- (3.4,1.3);
            \draw[thick] (3,2) -- (3.35,2.35);
            \draw[thick, fill=white] (3.35,2.35) circle(0.1);
        \end{tikzpicture}
    \end{center}
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.3-sol}{\underline{4.3}}
    On the vertical axis we have
    \[
        m_2\ddot{x} = T-m_2g
    \]
    On the horizontal axis we have
    \[
        \begin{cases}
            m_1 \ddot{x} &= T \\
            F &= \ddot{x} (m_1 + m_2 + M)
        \end{cases}
    \]
    We want \(m_2\ddot{x} = 0\) meaning \(T = m_2g\).
    Finally,
    \[
        \ddot{x} = \frac{m_2}{m_1}g
    \]
    and
    \[
        F = \frac{m_2}{m_1}(M + m_1 + m_2)g
    \]
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-4.4}{\underline{4.4}}
    A particle of mass \(m\) is constrained to move without friction inside a conical
    surface with an angle \(\alpha\).
    Find the initial conditions such that the particle moves
    in uniform circular motion relative to the vertical axis of the cone.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.4-sol}{\underline{4.4}}
    Let \(\alpha\) be the semi-vertical angle of the cone, and use cylindrical
    coordinates \((r,\varphi,z)\), with the vertical axis as the \(z\)-axis.
    The constraint is
    \[
        z = r \cot\alpha .
    \]
    For uniform circular motion around the vertical axis, we need
    \[
        r(t) = R, \qquad \dot r(t) = 0, \qquad \dot\varphi(t) = \omega ,
    \]
    with \(R\) constant.
    The forces are gravity and the normal reaction \(N\). Since the motion has no
    vertical acceleration, the vertical components must balance:
    \[
        N\sin\alpha = mg .
    \]
    The horizontal component of the normal force provides the centripetal force:
    \[
        N\cos\alpha = mR\omega^2 .
    \]
    Dividing the two equations gives
    \[
        R\omega^2 = g\cot\alpha .
    \]
    Hence
    \[
        \omega = \pm \sqrt{\frac{g\cot\alpha}{R}} .
    \]
    Therefore the required initial conditions are
    \[
        r(0)=R, \qquad z(0)=R\cot\alpha, \qquad \varphi(0)=\varphi_0,
    \]
    \[
        \dot r(0)=0, \qquad \dot z(0)=0, \qquad
        \dot\varphi(0)=\pm \sqrt{\frac{g\cot\alpha}{R}} .
    \]
    Equivalently, the initial velocity must be purely azimuthal, with magnitude
    \[
        v_0 = R|\omega| = \sqrt{gR\cot\alpha}.
    \]
    The sign of \(\dot\varphi(0)\) determines the direction of rotation.
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-4.5}{\underline{4.5}}
    A block of mass \(m_1\) is placed on top of a block of mass \(m_2\), which is at rest on a smooth floor.
    If the coefficient of friction between the two blocks is \(\mu\), find the maximum value of the horizontal
    force \(F\) that can be applied to \(m_2\) so that \(m_1\) does not slip.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.5-sol}{\underline{4.5}}
    The force is given by
    \[ F = a(m_2 + m_1) \]
    The upper block only through the friction force
    \[
        \mu m_1 g = m_1 a
    \]
    meaning that \(a = \mu g\)
    and \(F = \mu g (m_2 + m_2)\).
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-4.6}{\underline{4.6}}
    A body of mass \(m\), placed on a rough horizontal plane (friction coefficient \(\mu\)),
    is pulled by a force \(F\) forming an angle \(\alpha\) with the horizontal.
    The body moves at a constant velocity.
    Determine the angle \(\alpha\) for which the force intensity is minimal;
    also, calculate the value of this minimal force.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.6-sol}{\underline{4.6}}
    We have the equations
    \begin{align*}
        m\ddot{x} &= F\cos\alpha - \mu R \\
        m\ddot{y} &= R - mg + F\sin\alpha
    \end{align*}
    From this we get
    \[ R = mg - F\sin\alpha \]
    and by substituting it into the first equation
    \[
        F = \frac{\mu m g}{\cos\alpha + \mu\sin\alpha}
    \]
    In order to minimize the force we compute
    \begin{align*}
        \frac{d}{d\alpha}\left(\cos\alpha + \mu\sin\alpha\right) &= 0
        \mu &= \tan\alpha
    \end{align*}
    By plugging \(\alpha = \arctan(\mu)\) we get the minimum force
    \[
        F_\text{min} = \frac{\mu mg}{\sqrt{1 + \mu^2}}
    \]
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-4.7}{\underline{4.7}}
    Two blocks, \(A\) and \(B\), with respective masses \(m_A\) and \(m_B\),
    are connected by a massless inextensible rope.
    Block \(A\) rests on an inclined plane with an angle \(\alpha\) relative to the horizontal
    and is also attached to a spring with elastic constant \(k\), whose other end is fixed to a support
    at the base of the inclined plane. Block \(B\) hangs through a pulley parallel to the vertical
    leg of the triangle formed by the inclined plane. Neglecting friction, determine the oscillation period of the
    two bodies around the equilibrium position.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-4.7-sol}{\underline{4.7}}
    Let \(x\) be the displacement of block \(A\) along the inclined plane, measured
    upward from the base of the plane. Since the rope is massless and inextensible,
    the displacement of block \(B\) has the same magnitude. Therefore the two blocks
    have the same speed:
    \[
        v_A = v_B = \dot x .
    \]
    Hence the kinetic energy is
    \[
        T = \frac{1}{2}m_A\dot x^2 + \frac{1}{2}m_B\dot x^2
          = \frac{1}{2}(m_A+m_B)\dot x^2 .
    \]

    The potential energy is the sum of the elastic energy of the spring and the
    gravitational energies. Up to irrelevant additive constants, it has the form
    \[
        V(x)
        =
        \frac{1}{2}kx^2
        + m_Agx\sin\alpha
        - m_Bgx .
    \]
    Thus the Lagrangian is
    \[
        L
        =
        \frac{1}{2}(m_A+m_B)\dot x^2
        -
        \left(
            \frac{1}{2}kx^2
            + m_Agx\sin\alpha
            - m_Bgx
        \right).
    \]
    The equation of motion is
    \[
        (m_A+m_B)\ddot x + kx + m_Ag\sin\alpha - m_Bg = 0 .
    \]
    The equilibrium position \(x_{\mathrm{eq}}\) satisfies
    \[
        kx_{\mathrm{eq}} + m_Ag\sin\alpha - m_Bg = 0,
    \]
    so
    \[
        x_{\mathrm{eq}}
        =
        \frac{g(m_B-m_A\sin\alpha)}{k}.
    \]

    Now write
    \[
        x = x_{\mathrm{eq}}+\eta .
    \]
    Substituting into the equation of motion gives
    \[
        (m_A+m_B)\ddot\eta + k\eta = 0 .
    \]
    Therefore the angular frequency of small oscillations around equilibrium is
    \[
        \omega = \sqrt{\frac{k}{m_A+m_B}} .
    \]
    Hence the oscillation period is
    \[
        T = 2\pi\sqrt{\frac{m_A+m_B}{k}} .
    \]
    The angle \(\alpha\) and gravity only determine the equilibrium position, not
    the period of oscillation.
\end{snippetsolution}
