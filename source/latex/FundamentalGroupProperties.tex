\section{Properties of the fundamental group}

\begin{snippetproposition}{basepoint-change-isomorphism}{Change of basepoint}
    Let \(X\) be an arc-connected \topologicalspace and \(a, b \in X\).
    Let \(\gamma \in \Omega(X, a, b)\).
    Then the map \(\gamma_\# \colon \pi_1(X, a) \fromto \pi_1(X, b)\)
    defined by
    \[
        \gamma_\#([\alpha]) = [i(\gamma) * \alpha * \gamma]
    \]
    is a group isomorphism.
\end{snippetproposition}

\begin{snippetproof}{basepoint-change-isomorphism-proof}{basepoint-change-isomorphism}{Change of basepoint}
    \textbf{Homomorphism:}
    \begin{align*}
        \gamma_\#([\alpha] \cdot [\beta])
        &= \gamma_\#([\alpha * \beta]) \\
        &= [i(\gamma) * \alpha * \beta * \gamma] \\
        &= [i(\gamma) * \alpha * \gamma * i(\gamma) * \beta * \gamma] \\
        &= \gamma_\#([\alpha]) \cdot \gamma_\#([\beta])
    \end{align*}
    
    \textbf{Isomorphism:}
    The inverse is \((i(\gamma))_\#\), since
    \((i(\gamma))_\# \circ \gamma_\# = \identity\).
\end{snippetproof}



\begin{snippetproposition}{fundamental-group-product}{Product spaces}
    Let \(X, Y\) be \topologicalspace[topological spaces] and \(a \in X\), \(b \in Y\).
    Then
    \[
        \pi_1(X \cartesianprod Y, (a, b)) \cong \pi_1(X, a) \cartesianprod \pi_1(Y, b)
    \]
\end{snippetproposition}

\begin{snippetproposition}{fundamental-group-retract}{Retracts and π₁}
    Let \(A \subseteq X\) be a \snippetref[retract-definition][retract] with retraction \(r \colon X \fromto A\).
    Then \(\pi_1(r) \colon \pi_1(X, a) \fromto \pi_1(A, a)\)
    is surjective for \(a \in A\).
    
    If \(A\) is a \snippetref[deformation-retract-definition][deformation retract],
    then \(\pi_1(X, a) \cong \pi_1(A, a)\).
\end{snippetproposition}

\begin{snippetcorollary}{contractible-trivial-fundamental-group}{Contractible spaces}
    If \(X\) is \snippetref[contractible-space-definition][contractible],
    then \(\pi_1(X, a) = \{e\}\) for any \(a \in X\).
\end{snippetcorollary}

\begin{snippetproposition}{homotopy-equivalent-same-pi1}{Homotopy equivalence and π₁}
    If \(f \colon X \fromto Y\) is a \snippetref[homotopy-equivalence-definition][homotopy equivalence],
    then \(\pi_1(f) \colon \pi_1(X, a) \fromto \pi_1(Y, f(a))\) is an isomorphism.
\end{snippetproposition}
