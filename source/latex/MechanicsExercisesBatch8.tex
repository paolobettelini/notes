\section{Exercises - Batch 8}

\begin{snippetexercise}{mechanics-ex-8.1}{\underline{8.1}}
    A projectile of mass \(m\) is launched from the ground at time \(t=0\)
    with an initial velocity \(v_0\) in a direction forming an angle of \(\alpha\)
    with the horizontal.
    During its flight, the projective explodes into two fragments of masses
    \(\frac{2}{3}m\) and \(\frac{1}{3}m\). The two fragments
    land on the ground simultaneously, and the distance of the lighter fragment from the launch point is
    \(x_2\). Determine at what distance from the launch point the heavier fragment lands (neglecting air resistance).
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.1-sol}{\underline{8.1}}
    Let \(x_1\) be the distance from the launch point at which the heavier fragment
    lands. Since the explosion is an internal process, the motion of the center of
    mass is the same as the motion of the original projectile.

    The range of the original projectile, if it did not explode, would be
    \[
        R = \frac{v_0^2\sin(2\alpha)}{g}
    \]
    Since the two fragments land simultaneously, at that instant both fragments are
    on the ground. Therefore the center of mass is also on the ground, so its
    horizontal coordinate is exactly \(R\).

    The horizontal coordinate of the center of mass at landing is
    \[
        x_{\mathrm{CM}}
        =
        \frac{
            \frac{2}{3}m x_1 + \frac{1}{3}m x_2
        }{m}
    \]
    Hence
    \[
        \frac{2}{3}x_1 + \frac{1}{3}x_2 = R
    \]
    Substituting the expression for \(R\), we obtain
    \[
        \frac{2}{3}x_1 + \frac{1}{3}x_2
        =
        \frac{v_0^2\sin(2\alpha)}{g}
    \]
    Solving for \(x_1\) gives
    \[
        x_1
        =
        \frac{3}{2}\frac{v_0^2\sin(2\alpha)}{g}
        -
        \frac{x_2}{2}
    \]
    Therefore the heavier fragment lands at a distance
    \[
        x_1
        =
        \frac{3v_0^2\sin(2\alpha)}{2g}
        -
        \frac{x_2}{2}
    \]
    from the launch point.
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-8.2}{\underline{8.2}}
    A body of mass \(m\) is placed on a perch at a height \(h\) above the ground.
    At a certain instant, the body explodes and splits into two fragments of masses \(m_1\) and \(m_2 = m_1 / 2\),
    which reach the ground respectively to the left and to the right of the perch.
    Given that the initial velocities of the fragments are directed horizontally and that th
    distance between the points where they hit the ground is \(d\), calculate the initial velocities of
    the two fragments.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.2-sol}{\underline{8.2}}
    Let \(v_1\) be the speed of the fragment of mass \(m_1\), which moves to the
    left, and let \(v_2\) be the speed of the fragment of mass \(m_2\), which moves
    to the right.

    Since the body is initially at rest and the explosion is internal, horizontal
    momentum is conserved:
    \[
        -m_1v_1 + m_2v_2 = 0
    \]
    Since
    \[
        m_2 = \frac{m_1}{2}
    \]
    we obtain
    \[
        m_1v_1 = \frac{m_1}{2}v_2
    \]
    and therefore
    \[
        v_2 = 2v_1
    \]

    The initial velocities are horizontal, so both fragments have the same vertical
    motion. The time needed to fall from height \(h\) is determined by
    \[
        h = \frac{1}{2}gt^2
    \]
    hence
    \[
        t = \sqrt{\frac{2h}{g}}
    \]

    During this time, the horizontal distances travelled by the two fragments are
    \(v_1t\) and \(v_2t\). Since they land on opposite sides of the perch, the
    distance between the impact points is
    \[
        d = v_1t + v_2t
    \]
    Using \(v_2 = 2v_1\), this becomes
    \[
        d = 3v_1t
    \]
    Hence
    \[
        v_1 = \frac{d}{3t}
    \]
    Substituting the expression for \(t\), we get
    \[
        v_1 = \frac{d}{3}\sqrt{\frac{g}{2h}}
    \]
    and therefore
    \[
        v_2 = \frac{2d}{3}\sqrt{\frac{g}{2h}}
    \]

    Thus, taking the positive \(x\)-direction to the right, the initial velocities
    of the two fragments are
    \[
        \vec v_1 = -\frac{d}{3}\sqrt{\frac{g}{2h}}\,\hat x
    \]
    \[
        \vec v_2 = \frac{2d}{3}\sqrt{\frac{g}{2h}}\,\hat x
    \]
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-8.3}{\underline{8.3}}
    Two bodies of masses \(m_1\) and \(m_2 = 2m_1\) are on a smooth horizontal plane,
    connected by a spring of negligible mass, elastic constant\(k\), and natural length \(L_0\).
    Initially, the spring is compressed by a length \(\Delta L\) and held in this state by a thin thread
    placed between the masses. Then, the thread is cut.
    Calculate the maximum velocities reached by the masses in their motion.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.3-sol}{\underline{8.3}}
    Since the plane is smooth, there are no external horizontal forces. Therefore
    the horizontal momentum of the system is conserved. The system is initially at
    rest, so at every instant
    \[
        m_1v_1 + m_2v_2 = 0
    \]
    Taking \(v_1\) and \(v_2\) as the magnitudes of the velocities, the two masses
    move in opposite directions and
    \[
        m_1v_1 = m_2v_2
    \]
    Since
    \[
        m_2 = 2m_1
    \]
    we get
    \[
        v_1 = 2v_2
    \]

    The maximum speeds are reached when the spring passes through its natural
    length, because at that instant all the initial elastic potential energy has
    been converted into kinetic energy. Hence
    \[
        \frac{1}{2}k(\Delta L)^2
        =
        \frac{1}{2}m_1v_1^2
        +
        \frac{1}{2}m_2v_2^2
    \]
    Substituting \(m_2=2m_1\) and \(v_1=2v_2\), we obtain
    \[
        \frac{1}{2}k(\Delta L)^2
        =
        \frac{1}{2}m_1(2v_2)^2
        +
        \frac{1}{2}(2m_1)v_2^2
    \]
    Therefore
    \[
        \frac{1}{2}k(\Delta L)^2
        =
        3m_1v_2^2
    \]
    and
    \[
        v_{2,\max}
        =
        \Delta L\sqrt{\frac{k}{6m_1}}
    \]
    Since \(v_1=2v_2\), the maximum speed of the first mass is
    \[
        v_{1,\max}
        =
        \Delta L\sqrt{\frac{2k}{3m_1}}
    \]

    Thus the lighter mass reaches the greater speed:
    \[
        v_{1,\max}
        =
        \Delta L\sqrt{\frac{2k}{3m_1}}
    \]
    \[
        v_{2,\max}
        =
        \Delta L\sqrt{\frac{k}{6m_1}}
    \]
    The two velocities have opposite directions.
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-8.4}{\underline{8.4}}
    A body of mass \(M\) in the shape of a wedge rests on a horizontal plane.
    A block of mass \(m\) is placed on top of the wedge at a height \(h\) relative to the horizontal plane.
    All surfaces are frictionless.
    The system is initially at rest, so the block slides along the wedge.
    Calculate the velocity of the wedge when the block reaches the horizontal plane
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.4-sol}{\underline{8.4}}
    Let \(\alpha\) be the angle of the inclined face of the wedge with the
    horizontal. Let \(X\) be the horizontal coordinate of the wedge and let \(s\)
    be the displacement of the block along the inclined face, measured downward.

    The velocity of the block is
    \[
        \vec v_m
        =
        (\dot X+\dot s\cos\alpha)\hat x
        -
        \dot s\sin\alpha \hat y
    \]
    Since there are no external horizontal forces, horizontal momentum is conserved.
    The system is initially at rest, so
    \[
        M\dot X + m(\dot X+\dot s\cos\alpha)=0
    \]
    Therefore
    \[
        \dot X
        =
        -\frac{m\dot s\cos\alpha}{M+m}
    \]

    When the block reaches the horizontal plane, it has lost gravitational potential
    energy equal to \(mgh\). Conservation of mechanical energy gives
    \[
        mgh
        =
        \frac{1}{2}M\dot X^2
        +
        \frac{1}{2}m
        \left[
            (\dot X+\dot s\cos\alpha)^2
            +
            \dot s^2\sin^2\alpha
        \right]
    \]
    Substituting
    \[
        \dot X
        =
        -\frac{m\dot s\cos\alpha}{M+m}
    \]
    into the energy equation, we obtain
    \[
        mgh
        =
        \frac{1}{2}
        \frac{m(M+m\sin^2\alpha)}{M+m}
        \dot s^2
    \]
    Hence
    \[
        \dot s^2
        =
        \frac{2gh(M+m)}{M+m\sin^2\alpha}
    \]

    Therefore the speed of the wedge is
    \[
        V
        =
        |\dot X|
        =
        \frac{m\cos\alpha}{M+m}
        \sqrt{
            \frac{2gh(M+m)}{M+m\sin^2\alpha}
        }
    \]
    Equivalently,
    \[
        V
        =
        m\cos\alpha
        \sqrt{
            \frac{2gh}
            {(M+m)(M+m\sin^2\alpha)}
        }
    \]
    The wedge moves horizontally in the direction opposite to the horizontal motion
    of the block.
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-8.5}{\underline{8.5}}
    A marble of mass \(m_1\) moves with velocity \(v_0\) on a smooth horizontal plane and undergoes
    a head-on elastic collision with another marble of mass \(m_2 = 3m_1\), initially at rest.
    Subsequently, the second marble falls down a step of height \(h\).
    Determine the velocity \(v\) with which the marble reaches the ground and the distance \(d\)
    from the edge of the step where it lands.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.5-sol}{\underline{8.5}}
    First consider the head-on elastic collision. The first marble has initial
    velocity \(v_0\), while the second one is initially at rest. For a one-dimensional
    elastic collision, the velocity of the second marble after the collision is
    \[
        v_2
        =
        \frac{2m_1}{m_1+m_2}v_0
    \]
    Since
    \[
        m_2 = 3m_1
    \]
    we get
    \[
        v_2
        =
        \frac{2m_1}{m_1+3m_1}v_0
        =
        \frac{v_0}{2}
    \]

    After the collision, the second marble leaves the edge of the step with
    horizontal velocity
    \[
        v_x = \frac{v_0}{2}
    \]
    and zero vertical velocity. During the fall, its vertical velocity when it
    reaches the ground satisfies
    \[
        v_y^2 = 2gh
    \]
    hence
    \[
        v_y = \sqrt{2gh}
    \]

    Therefore the magnitude of the velocity with which the marble reaches the
    ground is
    \[
        v
        =
        \sqrt{v_x^2+v_y^2}
        =
        \sqrt{\frac{v_0^2}{4}+2gh}
    \]

    The time of fall is determined by
    \[
        h = \frac{1}{2}gt^2
    \]
    so
    \[
        t = \sqrt{\frac{2h}{g}}
    \]
    Hence the horizontal distance from the edge of the step is
    \[
        d
        =
        v_x t
        =
        \frac{v_0}{2}\sqrt{\frac{2h}{g}}
    \]
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-8.6}{\underline{8.6}}
    A ball of mass \(m_1\) is attached to a spring with a natural length \(L_0\),
    with the other end fixed at a point \(O\).
    Initially, the ball compresses the spring by a length \(x\); it is then released and elastically
    collides with another ball of mass \(m_2 = 3m_1\) placed at a distance \(d\) from \(O\).
    Calculate the maximum compression of the spring in the motion of mass \(m_1\) following the collision.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.6-sol}{\underline{8.6}}
    Let \(k\) be the elastic constant of the spring. The second ball is initially
    at rest, at distance \(d\) from \(O\). Therefore, when the collision occurs,
    the length of the spring is \(d\), so its deformation is
    \[
        d-L_0
    \]
    The initial elastic energy is
    \[
        \frac{1}{2}kx^2
    \]
    Just before the collision, conservation of mechanical energy gives
    \[
        \frac{1}{2}kx^2
        =
        \frac{1}{2}m_1v^2
        +
        \frac{1}{2}k(d-L_0)^2
    \]
    Hence
    \[
        v^2
        =
        \frac{k}{m_1}
        \left[
            x^2-(d-L_0)^2
        \right]
    \]

    The collision is elastic and one-dimensional. Since the second ball is initially
    at rest, the velocity of \(m_1\) immediately after the collision is
    \[
        v_1'
        =
        \frac{m_1-m_2}{m_1+m_2}v
    \]
    Using
    \[
        m_2 = 3m_1
    \]
    we obtain
    \[
        v_1'
        =
        -\frac{1}{2}v
    \]
    Thus \(m_1\) reverses its direction and moves back toward \(O\).

    After the collision, the mechanical energy of \(m_1\) and the spring is
    conserved. If \(x_{\max}\) is the maximum compression reached after the
    collision, then at that point the velocity of \(m_1\) is zero, so
    \[
        \frac{1}{2}kx_{\max}^2
        =
        \frac{1}{2}m_1\left(\frac{v}{2}\right)^2
        +
        \frac{1}{2}k(d-L_0)^2
    \]
    Therefore
    \[
        x_{\max}^2
        =
        \frac{m_1v^2}{4k}
        +
        (d-L_0)^2
    \]
    Substituting the expression for \(v^2\), we get
    \[
        x_{\max}^2
        =
        \frac{1}{4}
        \left[
            x^2-(d-L_0)^2
        \right]
        +
        (d-L_0)^2
    \]
    Hence
    \[
        x_{\max}^2
        =
        \frac{x^2+3(d-L_0)^2}{4}
    \]
    and the maximum compression is
    \[
        x_{\max}
        =
        \frac{1}{2}
        \sqrt{x^2+3(d-L_0)^2}
    \]
    This result assumes that the collision actually occurs, namely
    \[
        x^2 \geq (d-L_0)^2
    \]
\end{snippetsolution}

\begin{snippetexercise}{mechanics-ex-8.7}{\underline{8.7}}
    A billiard player, aiming to pocket a ball of mass \(m\), strikes it with their cue ball,
    also of mass \(m\), so that it deflects at an angle \(\beta\) relative to the line connecting the
    two balls. Given the initial velocity \(v_0\) of the player's ball and assuming an elastic collision,
    determine the velocities of the two balls after the impact.
\end{snippetexercise}

\begin{snippetsolution}{mechanics-ex-8.7-sol}{\underline{8.7}}
    Let \(\hat n\) be the unit vector along the line connecting the centers of the
    two balls at impact, and let \(\hat t\) be the perpendicular unit vector. The
    initial velocity of the cue ball can be decomposed as
    \[
        \vec v_0 = v_0\cos\beta\,\hat n + v_0\sin\beta\,\hat t
    \]
    The second ball is initially at rest.

    In a smooth elastic collision, the impulse acts only along the line of centers.
    Therefore the tangential component of the velocity of the cue ball is unchanged,
    while the normal component is exchanged between the two equal masses.

    Hence, after the collision, the cue ball has velocity
    \[
        \vec v_1' = v_0\sin\beta\,\hat t
    \]
    while the ball initially at rest has velocity
    \[
        \vec v_2' = v_0\cos\beta\,\hat n
    \]
    Therefore the speeds of the two balls after impact are
    \[
        v_1' = v_0\sin\beta
    \]
    \[
        v_2' = v_0\cos\beta
    \]
    The two final velocities are perpendicular to each other, since one is along
    the tangent direction and the other is along the line of centers.
\end{snippetsolution}
