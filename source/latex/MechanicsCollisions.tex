\section{Collisions}

\begin{snippetdefinition}{elastic-collision-definition}{Elastic collision}
    An \emph{elastic collision} is a collision in which the total kinetic energy of the system
    is conserved. If the external impulse during the collision is negligible, the total linear
    momentum is also conserved. Thus, for two bodies,
    \[
        \vec{p}_1(t_0) + \vec{p}_2(t_0)
        =
        \vec{p}_1(t_1) + \vec{p}_2(t_1)
    \]
    and
    \[
        E_{c1}(t_0) + E_{c2}(t_0)
        =
        E_{c1}(t_1) + E_{c2}(t_1),
    \]
    where \(t_0\) and \(t_1\) denote, respectively, an instant immediately before and immediately
    after the collision.
\end{snippetdefinition}

\begin{snippet}{elastic-collisions-expl}
    Consider a collision. The total momentum satisfies
    \[
        \frac{d\vec{p}}{dt}
        =
        m_1\vec{a}_1 + m_2\vec{a}_2
        =
        \vec{f}_1 + \vec{f}_2,
    \]
    where \(\vec{f}_1\) and \(\vec{f}_2\) are the external forces acting on the two bodies.
    Integrating over the time interval of the collision, we obtain
    \[
        \integral[t_0][t_1][\frac{d\vec{p}}{dt}][t]
        =
        \vec{p}(t_1) - \vec{p}(t_0)
        =
        \integral[t_0][t_1][\vec{f}_1 + \vec{f}_2][t]
        \approx 0.
    \]
    Hence the total momentum is approximately conserved if the duration of the collision is
    considered instantaneous, so that the external impulse is negligible.

    Suppose that there exists a potential energy \(U(\vec{r}_1, \vec{r}_2)\) such that
    \[
        \begin{cases}
            \vec{F}_{1,2} = -\nabla_1 U, \\
            \vec{F}_{2,1} = -\nabla_2 U.
        \end{cases}
    \]
    Then the total mechanical energy is
    \[
        E = E_{c1} + E_{c2} + U.
    \]
    Its time derivative is
    \begin{align*}
        \frac{dE}{dt}
        &=
        \vec{v}_1 \cdot \left(\vec{F}_{1,2} + \vec{f}_1\right)
        +
        \vec{v}_2 \cdot \left(\vec{F}_{2,1} + \vec{f}_2\right)
        +
        \nabla_1 U \cdot \vec{v}_1
        +
        \nabla_2 U \cdot \vec{v}_2 \\
        &=
        \vec{v}_1 \cdot \vec{f}_1
        +
        \vec{v}_2 \cdot \vec{f}_2.
    \end{align*}
    Integrating this quantity gives
    \[
        \integral[t_0][t_1][\frac{dE}{dt}][t]
        =
        E(t_1) - E(t_0)
        =
        \integral[t_0][t_1][
            \vec{v}_1 \cdot \vec{f}_1
            +
            \vec{v}_2 \cdot \vec{f}_2
        ][t]
        \approx 0.
    \]
    Therefore,
    \[
        E(t_1) = E(t_0),
    \]
    that is, the total mechanical energy is conserved between an instant immediately before and
    an instant immediately after the collision.

    Since the positions of the bodies are practically the same immediately before and immediately
    after the collision, the potential energy terms cancel:
    \begin{align*}
        E_{c1}(t_0) + E_{c2}(t_0) + U(t_0)
        &=
        E_{c1}(t_1) + E_{c2}(t_1) + U(t_1), \\
        E_{c1}(t_0) + E_{c2}(t_0)
        &=
        E_{c1}(t_1) + E_{c2}(t_1).
    \end{align*}
    Such a collision is called an \emph{elastic collision}.
\end{snippet}
