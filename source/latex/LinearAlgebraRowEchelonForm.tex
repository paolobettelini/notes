\section{Row echelon forms}

\begin{snippetdefinition}{row-echelon-form-definition}{Row echelon form}
    A \snippetref[matrix-definition][matrix] is said to be in \textit{row echelon form}
    if it satisfies the following properties:
    \begin{enumerate}
        \item all rows having only zero entries are at the bottom;
        \item the leading entry (that is, the left-most nonzero entry) of every nonzero row,
        called the pivot, is on the right of the leading entry of every row above.
    \end{enumerate}
\end{snippetdefinition}

\begin{snippetproposition}{row-echelon-form-non-uniqueness}{Non uniqueness of row echelon form}
    \snippetref[row-echelon-form-definition][Row echelon form] is not unique.
\end{snippetproposition}

\begin{snippetdefinition}{reduced-row-echelon-form-definition}{Reduced row echelon form}
    A \snippetref[matrix-definition][matrix] is said to be in \textit{reduced row echelon form}
    if it is in \snippetref[row-echelon-form-definition][row echelon form],
    every \matrixpivot is \(1\), and each pivot is the only nonzero entry in
    its column.
\end{snippetdefinition}

\begin{snippetproposition}{reduced-row-echelon-form-uniqueness}{Uniqueness of reduced row echelon form}
    \snippetref[reduced-row-echelon-form-definition][Reduced row echelon form] is unique.
\end{snippetproposition}

\section{Gauss-Jordan algorithm}

\begin{snippettheorem}{gauss-jordan-algorithm-theorem}{Gauss-Jordan algorithm}
    Let \(\mathbb{F}\) be a \field and let \(A\in\matrices_n(\mathbb{F})\).
    Then \(A\) is invertible \ifandonlyif there exists a finite sequence of
    elementary row operations that transforms
    \[
        [A\mid I_n]
    \]
    into
    \[
        [I_n\mid A^{-1}].
    \]
\end{snippettheorem}

\begin{snippetproof}{gauss-jordan-algorithm-theorem-proof}{gauss-jordan-algorithm-theorem}{Gauss-Jordan algorithm}
    Suppose first that \(A\) is invertible. Gaussian elimination produces
    pivots in every column: otherwise the homogeneous system \(Ax=0\) would
    have a nonzero solution, contradicting invertibility. After possible row
    swaps and row additions, there is a product \(E\) of elementary \matrix[matrices]
    such that
    \[
        EA=U=
        \begin{bmatrix}
            p_1 & *   & *   & \cdots & * \\
            0   & p_2 & *   & \cdots & * \\
            0   & 0   & p_3 & \cdots & * \\
            \vdots & \vdots & \vdots & \ddots & \vdots \\
            0   & 0   & 0   & \cdots & p_n
        \end{bmatrix},
    \]
    with every \(p_i\neq0\). Multiplying by the diagonal \matrix
    \(D=\operatorname{diag}(p_1^{-1},\ldots,p_n^{-1})\) makes all pivots
    equal to \(1\). Further elementary row operations eliminate the entries
    above the pivots, so there is another product \(\widehat{E}\) of
    elementary \matrix[matrices] such that
    \[
        \widehat{E}DEA=I_n.
    \]
    Thus \(\widehat{E}DE=A^{-1}\). Applying the same row operations to the
    augmented \matrix gives
    \[
        [A\mid I_n]\longmapsto[I_n\mid A^{-1}].
    \]

    Conversely, suppose that elementary row operations transform
    \([A\mid I_n]\) into \([I_n\mid B]\). If \(M\) is the product of the
    corresponding elementary \matrix[matrices], then
    \[
        MA=I_n
        \qquad\text{and}\qquad
        MI_n=B.
    \]
    Hence \(B=M\) and \(BA=I_n\). Since \(M\) is a product of invertible
    elementary \matrix[matrices], it is invertible; from \(MA=I_n\) we get
    \(A=M^{-1}\), so \(A\) is invertible and \(B=A^{-1}\).
\end{snippetproof}
