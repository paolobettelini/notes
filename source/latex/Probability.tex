\section{Probability spaces}

\begin{snippetdefinition}{probability-space-definition}{Probability space}
    A \emph{probability space} is a triple
    \[
        (\Omega,\mathcal F,\mathbb P)
    \]
    where:
    \begin{enumerate}
        \item \(\Omega\neq\emptyset\) is the \set of all possible
            \outcome[outcomes], called the \emph{sample space};
        \item \(\mathcal F\) is a \sigmaalgebra on \(\Omega\), whose elements
            are called \emph{events};
        \item \(\mathbb P\colon \mathcal F\fromto[0,1]\) is a \measure such that
            \(\mathbb P(\Omega)=1\), called a \emph{probability measure}.
    \end{enumerate}
\end{snippetdefinition}

\begin{snippetdefinition}{finite-probability-space-definition}{Finite probability space}
    Let \(\Omega\neq\emptyset\) be a finite \set.
    A \emph{finite probability measure} on \(\Omega\) is a \function
    \[
        \mathbb P\colon \powerset(\Omega)\fromto[0,1]
    \]
    such that:
    \begin{enumerate}
        \item \(\mathbb P(\Omega)=1\);
        \item for every \(A,B\in\powerset(\Omega)\) such that
            \(A\intersection B=\emptyset\),
            \[
                \mathbb P(A\union B)=\mathbb P(A)+\mathbb P(B).
            \]
    \end{enumerate}
    The triple \((\Omega,\powerset(\Omega),\mathbb P)\) is called a
    \emph{finite probability space}.
\end{snippetdefinition}

\begin{snippetdefinition}{classical-finite-probability-definition}{Classical finite probability}
    Let \(\Omega\neq\emptyset\) be a finite \set.
    The \emph{classical finite probability measure} on \(\Omega\) is the
    \probabilitymeasure
    \[
        \mathbb P\colon\powerset(\Omega)\fromto[0,1],
        \qquad
        \mathbb P(A)=\frac{\cardinality{A}}{\cardinality{\Omega}}.
    \]
    Equivalently, every \outcome \(\omega\in\Omega\) has the same probability
    \[
        \mathbb P(\{\omega\})=\frac{1}{\cardinality{\Omega}}.
    \]
\end{snippetdefinition}

\begin{snippetdefinition}{event-operations-probability-logic}{Event operations}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace.
    For \(A,B\in\mathcal F\), the logical operations on \event[events]
    correspond to \set operations:
    \begin{enumerate}
        \item the negation of \(A\) corresponds to the complement
            \(\Omega\difference A\);
        \item the logical sum of \(A\) and \(B\), meaning that at least one of
            the two \event[events] occurs, corresponds to \(A\union B\);
        \item the logical product of \(A\) and \(B\), meaning that both
            \event[events] occur, corresponds to \(A\intersection B\);
        \item the logical difference of \(A\) and \(B\), meaning that \(A\)
            occurs and \(B\) does not, corresponds to \(A\difference B\);
        \item the symmetric difference of \(A\) and \(B\), meaning that exactly
            one of the two \event[events] occurs, corresponds to
            \[
                (A\intersection(\Omega\difference B))
                \union
                ((\Omega\difference A)\intersection B);
            \]
        \item the implication \(A\Rightarrow B\) corresponds to the inclusion
            \(A\subseteq B\);
        \item the equivalence of \(A\) and \(B\) corresponds to the equality
            \(A=B\);
        \item the \event[events] \(A\) and \(B\) are incompatible when
            \(A\intersection B=\emptyset\).
    \end{enumerate}
    \conditionalprobability is a genuinely probabilistic operation and is not
    represented by a \set operation alone.
\end{snippetdefinition}

\section{Basic rules}

\begin{snippettheorem}{measure-of-inverse-theorem}{Complement rule}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace and let
    \(A\in\mathcal F\). Then
    \[
        \mathbb P(\Omega\difference A)=1-\mathbb P(A).
    \]
\end{snippettheorem}

\begin{snippetproof}{measure-of-inverse-proof}{measure-of-inverse-theorem}{Complement rule}
    The \event[events] \(A\) and \(\Omega\difference A\) are disjoint, and
    \(A\union(\Omega\difference A)=\Omega\). Therefore
    \[
        1=\mathbb P(\Omega)
        =\mathbb P(A)+\mathbb P(\Omega\difference A),
    \]
    which gives the formula.
\end{snippetproof}

\begin{snippettheorem}{measure-of-subset-theorem}{Monotonicity of probability}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace and let
    \(A,B\in\mathcal F\). If \(A\subseteq B\), then
    \[
        \mathbb P(A)\leq\mathbb P(B).
    \]
\end{snippettheorem}

\begin{snippetproof}{measure-of-subset-proof}{measure-of-subset-theorem}{Monotonicity of probability}
    Since \(A\subseteq B\), the \event[events] \(A\) and \(B\difference A\)
    are disjoint and
    \[
        B=A\union(B\difference A).
    \]
    Hence
    \[
        \mathbb P(B)=\mathbb P(A)+\mathbb P(B\difference A)\geq\mathbb P(A).
    \]
\end{snippetproof}

\begin{snippettheorem}{probability-difference-rule}{Difference rule}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace and let
    \(A,B\in\mathcal F\). If \(B\subseteq A\), then
    \[
        \mathbb P(A\difference B)=\mathbb P(A)-\mathbb P(B).
    \]
\end{snippettheorem}

\begin{snippetproof}{probability-difference-rule-proof}{probability-difference-rule}{Difference rule}
    Since \(B\subseteq A\), the \event[events] \(B\) and \(A\difference B\)
    are disjoint and
    \[
        A=B\union(A\difference B).
    \]
    Thus
    \[
        \mathbb P(A)=\mathbb P(B)+\mathbb P(A\difference B),
    \]
    which gives the formula.
\end{snippetproof}

\begin{snippettheorem}{probability-two-event-union-rule}{Union of two events}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace and let
    \(A,B\in\mathcal F\). Then
    \[
        \mathbb P(A\union B)
        =
        \mathbb P(A)+\mathbb P(B)-\mathbb P(A\intersection B).
    \]
\end{snippettheorem}

\begin{snippetproof}{probability-two-event-union-rule-proof}{probability-two-event-union-rule}{Union of two events}
    Decompose the union into the disjoint union
    \[
        A\union B
        =
        (A\difference B)\union(B\difference A)\union(A\intersection B).
    \]
    By additivity on disjoint \event[events],
    \[
        \mathbb P(A\union B)
        =
        \mathbb P(A\difference B)
        +
        \mathbb P(B\difference A)
        +
        \mathbb P(A\intersection B).
    \]
    Applying the difference rule to \(A\intersection B\subseteq A\) and
    \(A\intersection B\subseteq B\), we get
    \[
        \mathbb P(A\union B)
        =
        \mathbb P(A)+\mathbb P(B)-\mathbb P(A\intersection B).
    \]
\end{snippetproof}

\begin{snippetproposition}{finite-additivity-probability}{Finite additivity}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace.
    Let \(n\in\naturalnumbers\) and let
    \(A_1,\ldots,A_n\in\mathcal F\) be pairwise disjoint \event[events].
    Then
    \[
        \mathbb P\left(\bigcup_{k=1}^n A_k\right)
        =
        \sum_{k=1}^n\mathbb P(A_k).
    \]
\end{snippetproposition}

\begin{snippetproof}{finite-additivity-probability-proof}{finite-additivity-probability}{Finite additivity}
    This follows by induction from the additivity of \(\mathbb P\) on two
    disjoint \event[events].
    The case \(n=1\) is trivial. If the statement holds for \(n\), then
    \(\bigcup_{k=1}^n A_k\) is disjoint from \(A_{n+1}\), and so
    \[
        \mathbb P\left(\bigcup_{k=1}^{n+1} A_k\right)
        =
        \mathbb P\left(\bigcup_{k=1}^n A_k\right)
        +
        \mathbb P(A_{n+1})
        =
        \sum_{k=1}^{n+1}\mathbb P(A_k).
    \]
\end{snippetproof}

\begin{snippetproposition}{finite-subadditivity-probability}{Finite subadditivity}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace.
    Let \(n\in\naturalnumbers\) and let
    \(A_1,\ldots,A_n\in\mathcal F\). Then
    \[
        \mathbb P\left(\bigcup_{k=1}^n A_k\right)
        \leq
        \sum_{k=1}^n\mathbb P(A_k).
    \]
\end{snippetproposition}

\begin{snippetproof}{finite-subadditivity-probability-proof}{finite-subadditivity-probability}{Finite subadditivity}
    Define
    \[
        B_1=A_1,
        \qquad
        B_k=A_k\difference\bigcup_{j=1}^{k-1}A_j
        \quad\text{for }2\leq k\leq n.
    \]
    Then \(B_1,\ldots,B_n\) are pairwise disjoint,
    \(B_k\subseteq A_k\) for every \(k\), and
    \[
        \bigcup_{k=1}^n B_k=\bigcup_{k=1}^n A_k.
    \]
    By finite additivity and monotonicity,
    \[
        \mathbb P\left(\bigcup_{k=1}^n A_k\right)
        =
        \sum_{k=1}^n\mathbb P(B_k)
        \leq
        \sum_{k=1}^n\mathbb P(A_k).
    \]
\end{snippetproof}

\begin{snippettheorem}{inclusion-exclusion-principle}{Inclusion-exclusion principle}
    Let \((\Omega,\mathcal F,\mathbb P)\) be a \probabilityspace.
    Let \(n\in\naturalnumbers\) and let
    \(A_1,\ldots,A_n\in\mathcal F\). Then
    \[
        \mathbb P\left(\bigcup_{i=1}^n A_i\right)
        =
        \sum_{\emptyset\neq I\subseteq\{1,\ldots,n\}}
        (-1)^{\cardinality{I}+1}
        \mathbb P\left(\bigcap_{i\in I}A_i\right).
    \]
\end{snippettheorem}

\begin{snippetproof}{inclusion-exclusion-principle-proof}{inclusion-exclusion-principle}{Inclusion-exclusion principle}
    We prove the formula by induction on \(n\). For \(n=1\) it is immediate.
    Assume it holds for \(n\), and put
    \[
        U_n=\bigcup_{i=1}^n A_i.
    \]
    By the formula for the union of two \event[events],
    \[
        \mathbb P(U_n\union A_{n+1})
        =
        \mathbb P(U_n)+\mathbb P(A_{n+1})
        -
        \mathbb P(U_n\intersection A_{n+1}).
    \]
    The first term is handled by the induction hypothesis applied to
    \(A_1,\ldots,A_n\). For the last term, note that
    \[
        U_n\intersection A_{n+1}
        =
        \bigcup_{i=1}^n (A_i\intersection A_{n+1}),
    \]
    so the induction hypothesis applied to
    \(A_1\intersection A_{n+1},\ldots,A_n\intersection A_{n+1}\)
    gives exactly the terms in the formula whose index \set contains
    \(n+1\), with the opposite sign. Combining these terms yields the formula
    for \(n+1\).
\end{snippetproof}

\begin{snippetproposition}{finite-probability-simplex-correspondence}{Finite probability measures as a simplex}
    Let \(\Omega=\{\omega_1,\ldots,\omega_N\}\) with
    \(N\in\naturalnumbers\). There is a bijection between:
    \begin{enumerate}
        \item finite \probabilitymeasure[probability measures] on
            \((\Omega,\powerset(\Omega))\);
        \item vectors
            \[
                (p_1,\ldots,p_N)\in[0,1]^N
                \suchthat
                \sum_{i=1}^N p_i=1.
            \]
    \end{enumerate}
    Under this bijection, \(p_i=\mathbb P(\{\omega_i\})\).
\end{snippetproposition}

\begin{snippetproof}{finite-probability-simplex-correspondence-proof}{finite-probability-simplex-correspondence}{Finite probability measures as a simplex}
    Let \(\mathbb P\) be a finite \probabilitymeasure on
    \((\Omega,\powerset(\Omega))\). Define
    \[
        p_i=\mathbb P(\{\omega_i\})
        \qquad
        1\leq i\leq N.
    \]
    Then \(p_i\in[0,1]\) and, by finite additivity,
    \[
        \sum_{i=1}^N p_i
        =
        \sum_{i=1}^N\mathbb P(\{\omega_i\})
        =
        \mathbb P(\Omega)=1.
    \]
    Conversely, let \((p_1,\ldots,p_N)\in[0,1]^N\) with
    \(\sum_{i=1}^Np_i=1\). Define
    \[
        \mathbb P(A)=
        \sum_{\omega_i\in A}p_i
        \qquad
        A\in\powerset(\Omega).
    \]
    This is a finite \probabilitymeasure, and it is the only one with
    \(\mathbb P(\{\omega_i\})=p_i\) for every \(i\), because every
    \(A\in\powerset(\Omega)\) is the disjoint union of the singletons it
    contains.
\end{snippetproof}
