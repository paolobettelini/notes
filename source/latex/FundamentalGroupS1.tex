\section{Fundamental group of the circle}

\begin{snippettheorem}{fundamental-group-circle}{Fundamental group of \(S^1\)}
    \[
        \pi_1(S^1, 1) \cong \integers
    \]
    The isomorphism is given by the degree map \(\deg \colon \pi_1(S^1, 1) \fromto \integers\).
\end{snippettheorem}

\begin{snippetproof}{fundamental-group-circle-proof}{fundamental-group-circle}{\(S^1\)}
    \begin{itemize}
        \item \emph{homomorphism:}
            For loops \(\alpha, \beta\), we have \(\deg(\alpha * \beta) = \deg(\alpha) + \deg(\beta)\)
            since the lift of \(\alpha * \beta\) starting at \(0\) ends at \(\deg(\alpha) + \deg(\beta)\).
        \item \emph{surjective:}
            For any \(n \in \integers\), the loop \(\alpha_n(t) = e^{2\pi i n t}\)
            has degree \(n\).
        \item \emph{injective:}
            If \(\deg(\alpha) = 0\), then \(\tilde{\alpha}(1) = 0 = \tilde{\alpha}(0)\).
            The path \(\tilde{\alpha}\) is a loop in \(\realnumbers\).
            Since \(\realnumbers\) is contractible, \(\tilde{\alpha} \simeq 0\),
            hence \(\alpha = e \circ \tilde{\alpha} \simeq e \circ 0 = 1_1\).
    \end{itemize}
\end{snippetproof}

\begin{snippetcorollary}{fundamental-group-torus}{Fundamental group of torus}
    \[
        \pi_1(T^2) \cong \integers \cartesianprod \integers
    \]
    since \(T^2 = S^1 \cartesianprod S^1\).
\end{snippetcorollary}
