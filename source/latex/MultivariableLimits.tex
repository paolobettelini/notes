\section{Multivariable limits}

\begin{snippetdefinition}{multivariable-limit-definition}{Multivariable limit}
    Let \(f \colon \Omega \subseteq \realnumbers^n \fromto \realnumbers^m\)
    and let \(a\) be an \accumulationpoint for \(\Omega\).
    Then
    \[
        \lim_{x \to a} f(x) = b \in \realnumbers^m
    \]
    if for all \(\varepsilon > 0\), there exists \(\delta > 0\)
    such that for all \(x \in \Omega\),
    \[
        0 < \|x - a\|_1 < \delta \implies \|f(x) - b\|_2 < \varepsilon
    \]
    where the \norm[norms] are generic.
\end{snippetdefinition}

\plain{The concept of limit at infinity means exiting compact sets.}

\begin{snippetdefinition}{multivariable-limit-at-infinity-definition}{Limit at infinity}
    Let \(f \colon \Omega \subseteq \realnumbers^n \fromto \realnumbers^m\)
    with \(\Omega\) unbounded.
    Then
    \[
        \lim_{x \to \infty} f(x) = b \in \realnumbers^m
    \]
    if for all \(\varepsilon > 0\), there exists \(M > 0\)
    such that for all \(x \in \Omega\),
    \[
        \|x\| > M \implies \|f(x) - b\| < \varepsilon
    \]
\end{snippetdefinition}

\begin{snippetdefinition}{multivariable-infinite-limit-definition}{Infinite limit}
    Let \(f \colon \Omega \subseteq \realnumbers^n \fromto \realnumbers^m\)
    and let \(a\) be an \accumulationpoint.
    Then
    \[
        \lim_{x \to a} f(x) = \infty
    \]
    if for all \(K > 0\), there exists \(\delta > 0\)
    such that for all \(x \in \Omega\),
    \[
        0 < \|x - a\| < \delta \implies \|f(x)\| > K
    \]
\end{snippetdefinition}

\begin{snippettheorem}{limit-path-characterization-theorem}{Limit characterization via paths}
    Let \(f \colon \Omega \subseteq \realnumbers^n \fromto \realnumbers^m\)
    and let \(a\) be an \accumulationpoint for \(\Omega\). Then
    \[
        \lim_{x \to a} f(x) = b
        \iff
        \lim_{t \to 0^+} f(\varphi(t)) = b
    \]
    for all \(\varphi \in \varphi_\Omega(a)\),
    where \(\varphi_\Omega(a)\) the \set of
    paths with support in \(\Omega\)
    such that \(\varphi(t) \to a\)
    as \(t \to 0^+\)
    and \(\varphi(t) \neq a\) for all \(t\).
\end{snippettheorem}
