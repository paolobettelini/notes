\section{Examples}

\begin{snippetproposition}{field-is-vector-space-with-itself}{A field as a vector space over itself}
    Let \(\mathbb{F}\) be a \field. Then \(\mathbb{F}\), with its \field
    addition as vector addition and its \field multiplication as scalar
    multiplication, is a \vectorspace over \(\mathbb{F}\).
\end{snippetproposition}

\begin{snippetproposition}{field-cartesian-prod-vector-space}{The fundamental model \(F^n\)}
    Let \(\mathbb{F}\) be a \field and let \(n\in\naturalnumbers\). The
    \set
    \[
        \mathbb{F}^n
        =
        \{(x_1,\ldots,x_n)\suchthat x_i\in\mathbb{F}\}
    \]
    is a \vectorspace over \(\mathbb{F}\) with componentwise addition
    \[
        (x_1,\ldots,x_n)+(y_1,\ldots,y_n)
        =
        (x_1+y_1,\ldots,x_n+y_n)
    \]
    and componentwise scalar multiplication
    \[
        \lambda(x_1,\ldots,x_n)
        =
        (\lambda x_1,\ldots,\lambda x_n).
    \]
\end{snippetproposition}

\begin{snippetproof}{field-cartesian-prod-vector-space-proof}{field-cartesian-prod-vector-space}{The fundamental model \(F^n\)}
    The \vectorspace axioms follow componentwise from the corresponding \field
    identities in \(\mathbb{F}\). For example,
    \[
        (x_1,\ldots,x_n)+(y_1,\ldots,y_n)
        =
        (y_1,\ldots,y_n)+(x_1,\ldots,x_n)
    \]
    follows from commutativity of addition in \(\mathbb{F}\). The zero \vector
    is
    \[
        0=(0,\ldots,0),
    \]
    and the additive inverse of \(x=(x_1,\ldots,x_n)\) is
    \[
        -x=(-x_1,\ldots,-x_n).
    \]
\end{snippetproof}

\begin{snippetexample}{empty-set-not-vector-space-example}{The empty \set}
    The empty \set \(\emptyset\) is not a \vectorspace over any \field,
    because every \vectorspace must contain an additive identity.
\end{snippetexample}

\begin{snippetexample}{zero-vector-space-example}{The zero vector space}
    Let \(\mathbb{F}\) be a \field. The \set
    \[
        V=\{0\}
    \]
    is a \vectorspace over \(\mathbb{F}\) with operations
    \[
        0+0=0,
        \qquad
        \lambda0=0
        \quad
        \text{for every } \lambda\in\mathbb{F}.
    \]
    It is called the \emph{zero vector space}.
\end{snippetexample}

\begin{snippetexample}{sequence-vector-space-example}{Sequence space}
    Let \(\mathbb{F}\) be a \field. The \set
    \[
        \mathbb{F}^{\naturalnumbers}
        =
        \left\{
            \{x_n\}_{n\in\naturalnumbers}
            \suchthat
            x_n\in\mathbb{F}
        \right\}
    \]
    of all sequences with values in \(\mathbb{F}\) is a \vectorspace over
    \(\mathbb{F}\) with operations
    \[
        \{x_n\}+\{y_n\}=\{x_n+y_n\},
        \qquad
        \lambda\{x_n\}=\{\lambda x_n\}.
    \]
    The zero \vector is the constantly zero sequence.
\end{snippetexample}

\begin{snippetexample}{function-vector-space-example}{Function space}
    Let \(A\neq\emptyset\) be a \set and let \(\mathbb{F}\) be a \field. The
    \set
    \[
        \mathbb{F}^A
        =
        \{f\colon A\fromto\mathbb{F}\}
    \]
    of all \function[functions] from \(A\) to \(\mathbb{F}\) is a \vectorspace over
    \(\mathbb{F}\) with pointwise operations
    \[
        (f+g)(a)=f(a)+g(a),
        \qquad
        (\lambda f)(a)=\lambda f(a).
    \]
\end{snippetexample}

\begin{snippetexample}{polynomial-vector-space-example}{Polynomial space}
    Let \(\mathbb{F}\) be a \field. The \set \(\mathbb{F}[x]\) of
    \polynomial[polynomials] with coefficients in \(\mathbb{F}\) is a \vectorspace over
    \(\mathbb{F}\). Addition is polynomial addition, and scalar
    multiplication is multiplication of every coefficient by the \vsscalar.
    The zero \vector is the zero \polynomial.
\end{snippetexample}

\begin{snippetexample}{matrix-vector-space-example}{Matrix space}
    Let \(\mathbb{F}\) be a \field and let \(m,n\in\naturalnumbers\). The
    \set
    \[
        \matrices_{m\times n}(\mathbb{F})
    \]
    of \(m\times n\) \matrix[matrices] with entries in \(\mathbb{F}\) is a \vectorspace
    over \(\mathbb{F}\), with componentwise addition and scalar
    multiplication:
    \[
        (A+B)_{ij}=A_{ij}+B_{ij},
        \qquad
        (\lambda A)_{ij}=\lambda A_{ij}.
    \]
\end{snippetexample}

\begin{snippet}{complex-vector-spaces-as-real-vector-spaces}
    Since \(\realnumbers\subseteq\complexnumbers\), every complex \vectorspace
    can also be viewed as a real \vectorspace by restricting scalar
    multiplication from complex scalars to real scalars. For example,
    \(\complexnumbers\) is generated by \(1\) as a complex \vectorspace, but
    it needs \(1\) and \(\mathrm{i}\) as generators when viewed as a real
    \vectorspace.
\end{snippet}
