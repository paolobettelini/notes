\section{Basic definitions}

\begin{snippetdefinition}{graph-definition}{Graph}
    A \emph{graph} \(G\) consists of:
    \begin{itemize}
        \item a \set \(V(G)\), whose elements are called \emph{vertices};
        \item a \set \(E(G)\), whose elements are called \emph{edges};
        \item an endpoint map assigning to each edge \(e\in E(G)\) an unordered
        pair of vertices, called the \emph{endpoints} of \(e\).
    \end{itemize}
    The two endpoints of an edge are allowed to coincide.
\end{snippetdefinition}

\begin{snippetdefinition}{graph-loop-definition}{Loop}
    Let \(G\) be a \graph and let \(e\in E(G)\).
    The edge \(e\) is called a \emph{loop} if its two endpoints coincide.
\end{snippetdefinition}

\begin{snippet}{graph-loop-illustration}
    A loop is drawn as an edge that starts and ends at the same vertex:
    \[
        \begin{tikzpicture}[scale=1]
            \coordinate (v) at (0,0);
            \fill (v) circle (1.8pt);
            \draw (v) .. controls +(0.6,0.6) and +(-0.6,0.6) .. (v);
            \node[below] at (v) {\(v\)};
        \end{tikzpicture}
    \]
\end{snippet}

\begin{snippetdefinition}{simple-graph-definition}{Simple graph}
    A \graph \(G\) is \emph{simple} if it has no loops and no multiple edges,
    that is, if no two distinct edges have the same endpoints.
\end{snippetdefinition}

\begin{snippetdefinition}{finite-graph-definition}{Finite graph}
    A \graph \(G\) is \emph{finite} if both \(V(G)\) and \(E(G)\) are finite
    \set[sets].
\end{snippetdefinition}

\begin{snippet}{graph-terminology-warning}
    Some authors use the word \emph{graph} for what is here called a
    \emph{simple graph}, and use \emph{multigraph} when loops or multiple edges
    are allowed.
\end{snippet}

\begin{snippetdefinition}{graph-adjacent-vertices-definition}{Adjacent vertices}
    Let \(G\) be a \graph.
    Two vertices \(u,v\in V(G)\) are \emph{adjacent} if there exists an edge
    \(e\in E(G)\) whose endpoints are \(u\) and \(v\).
\end{snippetdefinition}

\begin{snippetdefinition}{graph-adjacent-edges-definition}{Adjacent edges}
    Let \(G\) be a \graph.
    Two distinct edges \(e_1,e_2\in E(G)\) are \emph{adjacent} if they have a
    common endpoint.
\end{snippetdefinition}

\begin{snippetdefinition}{graph-order-definition}{Order of a graph}
    Let \(G\) be a finite \graph.
    The \emph{order} of \(G\) is the number \(\cardinality{V(G)}\) of vertices
    of \(G\).
\end{snippetdefinition}

\begin{snippetdefinition}{graph-size-definition}{Size of a graph}
    Let \(G\) be a finite \graph.
    The \emph{size} of \(G\) is the number \(\cardinality{E(G)}\) of edges of
    \(G\).
\end{snippetdefinition}

\begin{snippetdefinition}{graph-vertex-degree-definition}{Degree of a vertex}
    Let \(G\) be a \graph and let \(v\in V(G)\).
    The \emph{degree} of \(v\), denoted by \(\graphdeg(v)\), is the number of
    edges having \(v\) as an endpoint, where loops are counted twice.
\end{snippetdefinition}

\begin{snippetexample}{graph-degree-example}{Computing degrees}
    Consider the following graph:
    \[
        \begin{tikzpicture}[scale=1]
            \coordinate (A) at (-2,0.8);
            \coordinate (B) at (-2,-0.9);
            \coordinate (F) at (0,0.6);
            \coordinate (C) at (0,-0.8);
            \coordinate (D) at (1.35,-0.35);
            \coordinate (E) at (1.35,0.85);

            \node[left] at (A) {\(A\)};
            \node[left] at (B) {\(B\)};
            \node[below right] at (F) {\(F\)};
            \node[below] at (C) {\(C\)};
            \node[right] at (D) {\(D\)};
            \node[right] at (E) {\(E\)};

            \draw (A) .. controls (-1.25,1.15) and (-0.6,1.05) .. (F);
            \draw (A) .. controls (-1.25,0.35) and (-0.6,0.25) .. (F);
            \draw (F)--(C);
            \draw (C)--(D);
            \draw (D)--(E);
            \draw (F) .. controls +(0.35,0.85) and +(0.85,0.15) .. (F);

            \foreach \p in {A,B,C,D,E,F}{
                \fill (\p) circle (1.8pt);
            }
        \end{tikzpicture}
    \]
    The degrees are:
    \[
        \begin{array}{c|cccccc}
            v & A & B & C & D & E & F \\
            \hline
            \graphdeg(v) & 2 & 0 & 2 & 2 & 1 & 5
        \end{array}
    \]
    The vertex \(B\) is isolated, so it has degree \(0\). The vertex \(F\)
    has degree \(5\): the two parallel edges to \(A\) contribute \(2\), the
    edge to \(C\) contributes \(1\), and the loop contributes \(2\).
\end{snippetexample}

\begin{snippettheorem}{handshaking-lemma}{Handshaking lemma}
    Let \(G\) be a finite \graph. Then,
    \[
        \sum_{v\in V(G)}\graphdeg(v)=2\cardinality{E(G)}.
    \]
    In particular, the number of vertices of odd degree is even.
\end{snippettheorem}

\begin{snippetproof}{handshaking-lemma-proof}{handshaking-lemma}{Handshaking lemma}
    List all edges of \(G\), and for each edge list its endpoints. Each edge
    contributes exactly \(2\) to the sum of the degrees: an ordinary edge
    contributes \(1\) to each of its endpoints, and a loop contributes \(2\) to
    its unique endpoint. Hence,
    \[
        \sum_{v\in V(G)}\graphdeg(v)=2\cardinality{E(G)}.
    \]
    The right-hand side is even. In the sum on the left, vertices of even
    degree contribute even summands, while vertices of odd degree contribute
    odd summands. A sum of integers is even if and only if it contains an even
    number of odd summands. Therefore the number of vertices of odd degree is
    even.
\end{snippetproof}

\begin{snippetexercise}{infinite-graph-handshaking-exercise}{Infinite graphs and odd degrees}
    Let \(G\) be an infinite \graph in which every vertex has finite degree.
    Suppose that the number of vertices of odd degree is finite. Can we
    conclude that this number is even?
\end{snippetexercise}

\begin{snippetsolution}{infinite-graph-handshaking-solution}{Infinite graphs and odd degrees}
    No. The conclusion of the handshaking lemma may fail for infinite graphs.
    A counterexample is the infinite ray:
    \[
        \begin{tikzpicture}[scale=1]
            \foreach \x in {0,1,2,3,4,5}{
                \fill (\x,0) circle (1.6pt);
            }
            \foreach \x in {0,1,2,3,4}{
                \draw (\x,0)--(\x+1,0);
            }
            \node at (5.65,0) {\(\cdots\)};
        \end{tikzpicture}
    \]
    The first vertex has degree \(1\), and every other vertex has degree \(2\).
    Hence there is exactly one vertex of odd degree.
\end{snippetsolution}

\section{Representations}

\begin{snippetdefinition}{adjacency-matrix-definition}{Adjacency matrix}
    Let \(G\) be a finite simple \graph with vertices
    \(v_1,\ldots,v_n\). The \emph{adjacency matrix} of \(G\) is the
    \(n\times n\) matrix \(A=(a_{ij})\) defined by
    \[
        a_{ij}=
        \begin{cases}
            1 & \text{if } v_i \text{ and } v_j \text{ are adjacent},\\
            0 & \text{otherwise}.
        \end{cases}
    \]
\end{snippetdefinition}

\begin{snippetexample}{adjacency-matrix-example}{Adjacency matrix}
    For the following graph, with vertices ordered as \(a,b,c,d,e\), the
    adjacency matrix is:
    \[
        \begin{array}{c@{\qquad}c}
            \begin{tikzpicture}[main/.style={draw,circle}]
                \foreach \x [count=\xi] in {a,b,c,d,e} {
                    \node[main] (\xi) at
                        ({1.1*cos(360 / 5 * \xi)}, {1.1*sin(360 / 5 * \xi)})
                        {\(\x\)};
                }
                \draw (1)--(2);
                \draw (1)--(3);
                \draw (1)--(5);
                \draw (2)--(4);
                \draw (5)--(3);
            \end{tikzpicture}
            &
            A=
            \begin{blockarray}{cccccc}
                & a & b & c & d & e \\
                \begin{block}{c(ccccc)}
                a & 0 & 1 & 1 & 0 & 1 \\
                b & 1 & 0 & 0 & 1 & 0 \\
                c & 1 & 0 & 0 & 0 & 1 \\
                d & 0 & 1 & 0 & 0 & 0 \\
                e & 1 & 0 & 1 & 0 & 0 \\
                \end{block}
            \end{blockarray}
        \end{array}
    \]
\end{snippetexample}

\plain{For finite simple graphs, the adjacency matrix has zero diagonal and is symmetric.}

\section{Types of graphs}

\begin{snippetdefinition}{null-graph-definition}{Null graph}
    The \emph{null graph} is the \graph with no vertices and no edges.
\end{snippetdefinition}

\begin{snippetdefinition}{empty-graph-definition}{Empty graph}
    Let \(V\) be a \set. The \emph{empty graph} on \(V\) is the \graph
    \(G\) such that \(V(G)=V\) and \(E(G)=\emptyset\).
\end{snippetdefinition}

\begin{snippetdefinition}{trivial-graph-definition}{Trivial graph}
    A \graph \(G\) is \emph{trivial} if it has exactly one vertex and no edges.
\end{snippetdefinition}

\section{Subgraphs}

\begin{snippetdefinition}{subgraph-definition}{Subgraph}
    Let \(G\) and \(H\) be \graph[graphs]. Then \(H\) is a \emph{subgraph}
    of \(G\), denoted \(H\subseteq G\), if
    \[
        V(H)\subseteq V(G)
        \quad\text{and}\quad
        E(H)\subseteq E(G),
    \]
    and the endpoint map of \(H\) is the restriction of the endpoint map of
    \(G\).
\end{snippetdefinition}

\begin{snippetdefinition}{proper-subgraph-definition}{Proper subgraph}
    Let \(G\) and \(H\) be \graph[graphs]. Then \(H\) is a
    \emph{proper subgraph} of \(G\), denoted \(H\subset G\), if
    \(H\subseteq G\) and \(H\neq G\).
\end{snippetdefinition}

\begin{snippetdefinition}{spanning-subgraph-definition}{Spanning subgraph}
    Let \(G\) and \(H\) be \graph[graphs]. Then \(H\) is a
    \emph{spanning subgraph} of \(G\) if \(H\subseteq G\) and
    \(V(H)=V(G)\).
\end{snippetdefinition}

\begin{snippetdefinition}{vertex-induced-subgraph-definition}{Vertex-induced subgraph}
    Let \(G\) be a simple \graph and let \(S\subseteq V(G)\).
    The \emph{vertex-induced subgraph} of \(G\) by \(S\), denoted \(G[S]\),
    is the subgraph with vertex set \(S\) and with all edges of \(G\) whose
    endpoints both lie in \(S\).
\end{snippetdefinition}

\begin{snippetdefinition}{edge-induced-subgraph-definition}{Edge-induced subgraph}
    Let \(G\) be a \graph and let \(S\subseteq E(G)\).
    The \emph{edge-induced subgraph} of \(G\) by \(S\) is the subgraph with
    edge set \(S\) and with all vertices incident with at least one edge of
    \(S\).
\end{snippetdefinition}
