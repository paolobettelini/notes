\begin{snippetexercise}{linear-algebra-batch-11-ex-1}{}
    In \(\mathbb{F}^5\) consider the vectors
    \begin{align*}
        v_1 &= (1, 0, 1, 1, 0) \\
        v_2 &= (1, 1, 0, 1, 1) \\
        v_3 &= (0, 1, 0, 1, 0) \\
        v_4 &= (1, 0, 1, 0, 1) \\
        v_5 &= (1, 1, 1, 1, k)
    \end{align*}
    where \(k \in \mathbb{F}\) is a parameter.
    Determine whether \(\{v_1, \dots, v_5\}\) is a basis of \(\mathbb{F}^5\).
\end{snippetexercise}
