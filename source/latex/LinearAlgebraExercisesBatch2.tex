\begin{snippetexercise}{linear-algebra-batch-2-ex-1}{}
    Solve the following linear systems:
    \[
        \begin{cases}
            x_1 - 3x_2 - x_3 + x_4 = 0 \\
            2x_2 - 4x_3 + x_4 = 1 \\
            x_1 - x_2 + 4x_3 - 2x_4 = -1 \\
            2x_1 - 2x_2 - x_3 + 2x_4 = 0
        \end{cases}
        \quad
        \begin{cases}
            x + y + z + w = 1 \\
            8x + 4y + 2z + w = 5 \\
            27x + 9y + 3z + w = 14 \\
            64x + 16y + 4z + w = 30
        \end{cases}
    \]
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-2-ex-1-sol}{}
    For the first one we have
    \[
        \begin{bmatrix}
            1 & -3 & -1 & 1 \\
            0 & 2 & -4 & 1 \\
            1 & -1 & 4 & -2 \\
            2 & -2 & -1 & 2
        \end{bmatrix}
        \begin{pmatrix}
            x_1 \\
            x_2 \\
            x_3 \\
            x_4
        \end{pmatrix}
        = \begin{pmatrix}
            0 \\
            1 \\
            -1 \\
            0
        \end{pmatrix}
    \]
    using Gaussian elimination we get
    \[
        \begin{bmatrix}
            1 & -3 & -1 & 1 \\
            0 & 2 & -4 & 1 \\
            0 & 0 & 9 & -4 \\
            0 & 0 & 0 & 2
        \end{bmatrix}
        \begin{pmatrix}
            x_1 \\
            x_2 \\
            x_3 \\
            x_4
        \end{pmatrix}
        = \begin{pmatrix}
            0 \\
            1 \\
            -2 \\
            0
        \end{pmatrix}
    \]
    fron which we get
    \[
        (x_1, x_2, x_3, x_4) = \left(
            -\frac{1}{18}, \frac{1}{18}, -\frac{2}{9}, 0
        \right)
    \]
    For the second one we have
    \[
        \begin{bmatrix}
            1 & 1 & 1 & 1 \\
            8 & 4 & 2 & 1 \\
            27 & 9 & 3 & 1 \\
            64 & 16 & 4 & 1
        \end{bmatrix}
        \begin{pmatrix}
            x \\
            y \\
            z \\
            w
        \end{pmatrix}
        = \begin{pmatrix}
            1 \\
            5 \\
            14 \\
            30
        \end{pmatrix}
    \]
    using Gaussian elimination we get
    \[
        \begin{bmatrix}
            1 & 1 & 1 & 1 \\
            0 & -4 & -6 & -7 \\
            0 & 0 & 3 & \frac{11}{2} \\
            0 & 0 & 0 & -1
        \end{bmatrix}
        \begin{pmatrix}
            x_1 \\
            x_2 \\
            x_3 \\
            x_4
        \end{pmatrix}
        = \begin{pmatrix}
            1 \\
            -3 \\
            \frac{1}{2} \\
            0
        \end{pmatrix}
    \]
    fron which we get
    \[
        (x,y,z,w) = \left(
            \frac{1}{3}, \frac{1}{2}, \frac{1}{6}, 0
        \right)
    \]
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-2-ex-2}{}
    Compute every possible product between the following matrices:
    \[
        A = \begin{pmatrix}
            1 & 2 & 4
        \end{pmatrix},
        \quad
        B = \begin{pmatrix}
            3 \\ 2 \\ 1
        \end{pmatrix},
        \quad
        C = \begin{pmatrix}
            1 & 2 & 4 \\
            2 & 0 & -1
        \end{pmatrix},
        \quad
        D = \begin{pmatrix}
            1 & 2 \\
            0 & 1 \\
            -1 & 4
        \end{pmatrix}
    \]
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-2-ex-2-sol}{}
    The possible products are:
    \begin{align*}
        AB &= 11 \\
        AD &= \begin{pmatrix}-3 & 20\end{pmatrix} \\
        CB &= \begin{pmatrix}11 \\ 5\end{pmatrix} \\
        CD &= \begin{pmatrix}-3 & 20 \\ 3 & 0\end{pmatrix} \\
        BA &= \begin{pmatrix}3 & 6 & 12 \\ 2 & 4 & 8 \\ 1 & 2 & 4\end{pmatrix} \\
        DC &= \begin{pmatrix}5 & 2 & 2 \\ 2 & 0 & -1 \\ 7 & -2 & -8\end{pmatrix}
    \end{align*}
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-2-ex-3}{}
    Solve the linear system given by \(Ax = b\) where
    \[
        A = \begin{pmatrix}
            10 & -7 & 0 \\ -3 & 2 & 6 \\ 5 & -1 & 5
        \end{pmatrix}, \quad
        b = \begin{pmatrix}
            1 \\ 1 \\ 1
        \end{pmatrix}
    \]
    \begin{enumerate}
        \item Compute the \(A = LU\) factorization.
        \item Use the \(A = LU\) factorization to solve the linear system \(Ax = b\)
        by separately solving \(Ly = b\) and \(Ux = y\).
    \end{enumerate}
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-2-ex-3-sol}{}
    \begin{enumerate}
        \item We compute the \(A = LU\) factorization by performing Gaussian elimination. We get
        \[
            L = \begin{pmatrix}
                1 & 0 & 0 \\ -\frac{3}{10} & 1 & 0 \\ \frac{1}{2} & -25 & 1
            \end{pmatrix}, \quad
            U = \begin{pmatrix}
                10 & -7 & 0 \\ 0 & -\frac{1}{10} & 6 \\ 0 & 0 & 155
            \end{pmatrix}
        \]
        \item We solve \(Ly = b\):
        \[
            \begin{pmatrix}
                1 & 0 & 0 \\ -\frac{3}{10} & 1 & 0 \\ \frac{1}{2} & -25 & 1
            \end{pmatrix} \begin{pmatrix}
                y_1 \\ y_2 \\ y_3
            \end{pmatrix} = \begin{pmatrix}
                1 \\ 1 \\ 1
            \end{pmatrix} \implies
            \begin{cases}
                y_1 = 1 \\
                -\frac{3}{10}y_1 + y_2 = 1 \\
                \frac{1}{2}y_1 - 25y_2 + y_3 = 1
            \end{cases} \implies
            y = \begin{pmatrix}
                1 \\ \frac{13}{10} \\ 33
            \end{pmatrix}
        \]
        Then, we solve \(Ux = y\):
        \[
            \begin{pmatrix}
                10 & -7 & 0 \\ 0 & -\frac{1}{10} & 6 \\ 0 & 0 & 155
            \end{pmatrix} \begin{pmatrix}
                x_1 \\ x_2 \\ x_3
            \end{pmatrix} = \begin{pmatrix}
                1 \\ \frac{13}{10} \\ 33
            \end{pmatrix} \implies
            \begin{cases}
                10x_1 - 7x_2 = 1 \\
                -\frac{1}{10}x_2 + 6x_3 = \frac{13}{10} \\
                155x_3 = 33
            \end{cases} \implies
            x = \begin{pmatrix}
                -\frac{9}{155} \\ -\frac{7}{31} \\ \frac{33}{155}
            \end{pmatrix}
        \]
    \end{enumerate}
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-2-ex-4}{}
    Use the Gauss-Jordan method to invert the following matrices:
    \[
        \begin{pmatrix}
            1 & 0 & 0 \\ 1 & 1 & 1 \\ 0 & 0 & 1
        \end{pmatrix},
        \quad
        \begin{pmatrix}
            2 & -1 & 0 \\ -1 & 2 & -1 \\ 0 & -1 & 2
        \end{pmatrix},
        \quad
        \begin{pmatrix}
            0 & 0 & 1 \\ 0 & 1 & 1 \\ 1 & 1 & 1
        \end{pmatrix}.
    \]
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-2-ex-4-sol}{}
    By applying Gaussian elimination to the augmented matrix \([A | I]\), we get the inverse matrices:
    \begin{align*}
        \begin{pmatrix}
            1 & 0 & 0 \\ 1 & 1 & 1 \\ 0 & 0 & 1
        \end{pmatrix}^{-1}
        &= \begin{pmatrix}
            1 & 0 & 0 \\ -1 & 1 & -1 \\ 0 & 0 & 1
        \end{pmatrix} \\
        \begin{pmatrix}
            2 & -1 & 0 \\ -1 & 2 & -1 \\ 0 & -1 & 2
        \end{pmatrix}^{-1}
        &= \frac{1}{4}\begin{pmatrix}
            3 & 2 & 1 \\ 2 & 4 & 2 \\ 1 & 2 & 3
        \end{pmatrix} \\
        \begin{pmatrix}
            0 & 0 & 1 \\ 0 & 1 & 1 \\ 1 & 1 & 1
        \end{pmatrix}^{-1}
        &= \begin{pmatrix}
            0 & -1 & 1 \\ -1 & 1 & 0 \\ 1 & 0 & 0
        \end{pmatrix}
    \end{align*}
\end{snippetsolution}
