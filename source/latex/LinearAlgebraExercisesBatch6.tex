\begin{snippetexercise}{linear-algebra-batch-6-ex-1}{}
    Determine whether the following collections of points in \(\mathbb{A}^3(\realnumbers)\) are affinely independent:\\
    \begin{align*}
        &\text{a) } \{(2, 1, 4), (4, 4, -1), (6, 7, -6)\}; &&\text{c) } \{(1, 2, 3), (5, -4, 7), (-1, -1, 6), (9, 2, 1)\};\\
        &\text{b) } \{(1, 2, 3), (-4, 2, 1), (1, 1, 2)\}; &&\text{d) } \{(3, 4, 5), (4, 5, 6), (1, 6, 6), (-1, 5, 8)\}.
    \end{align*}
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-6-ex-1-sol}{}
    A set of points \(\{P_0, \dots, P_k\}\) is affinely independent if and only if the vectors \(\vec{P_0 P_1}, \dots, \vec{P_0 P_k}\) are linearly independent.
    \begin{enumerate}[label=\alph*)]
        \item Let \(P_0 = (2, 1, 4)\), \(P_1 = (4, 4, -1)\), \(P_2 = (6, 7, -6)\). The vectors are \(v_1 = P_1 - P_0 = (2, 3, -5)\) and \(v_2 = P_2 - P_0 = (4, 6, -10)\). Since \(v_2 = 2v_1\), the vectors are linearly dependent, so the points are affinely dependent.
        \item Let \(P_0 = (1, 2, 3)\), \(P_1 = (-4, 2, 1)\), \(P_2 = (1, 1, 2)\). The vectors are \(v_1 = P_1 - P_0 = (-5, 0, -2)\) and \(v_2 = P_2 - P_0 = (0, -1, -1)\). These two vectors are linearly independent, so the points are affinely independent.
        \item Let \(P_0 = (1, 2, 3)\). The associated vectors \(P_1 - P_0, P_2 - P_0, P_3 - P_0\) are \(v_1 = (4,-6,4)\), \(v_2 = (-2,-3,3)\), \(v_3 = (8,0,-2)\). The determinant of the matrix with these vectors as columns is zero, meaning they are linearly dependent. Thus, the points are affinely dependent.
        \item Let \(P_0 = (3, 4, 5)\). The associated vectors are \(v_1 = (1,1,1)\), \(v_2 = (-2,2,1)\), \(v_3 = (-4,1,3)\). The determinant of the matrix with these vectors as columns is \(13 \neq 0\), meaning they are linearly independent. Thus, the points are affinely independent.
    \end{enumerate}
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-6-ex-2}{}
    In the affine space \(\mathbb{A}^3(\realnumbers)\) consider the points \[P = (1, 1, 1), \quad Q = (3, 3, 5), \quad R = (-1, 0, 2) \quad \text{and} \quad S = (-3, 1, -2).\]
    \begin{enumerate}[label=\Alph*]
        \item Verify that \(\{P, Q, R, S\}\) is an affine frame of reference in \(\mathbb{A}^3(\realnumbers)\).
        \item Calculate the coordinates of \(O = (0, 0, 0)\) in this affine frame of reference.
    \end{enumerate}
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-6-ex-2-sol}{}
    \begin{enumerate}[label=\Alph*]
        \item To verify that \(\{P, Q, R, S\}\) is an affine frame of reference, we check if the vectors \(\vec{PQ}, \vec{PR}, \vec{PS}\) form a basis of \(\realnumbers^3\). We have \(\vec{PQ} = (2, 2, 4)\), \(\vec{PR} = (-2, -1, 1)\), \(\vec{PS} = (-4, 0, -3)\).
        The determinant of the matrix formed by these vectors is
        \[
            \begin{vmatrix}
                2 & -2 & -4 \\
                2 & -1 & 0 \\
                4 & 1 & -3
            \end{vmatrix} = -30 \neq 0
        \]
        Thus, they are linearly independent and form a \basis, meaning it is an affine frame of reference.
        \item The coordinates \((x,y,z)\) of \(O=(0,0,0)\) in this frame satisfy \(\vec{PO} = x\vec{PQ} + y\vec{PR} + z\vec{PS}\).
        Since \(\vec{PO} = (-1, -1, -1)\), we need to solve the linear system:
        \[
            \begin{cases}
                2x - 2y - 4z = -1 \\
                2x - y = -1 \\
                4x + y - 3z = -1
            \end{cases}
        \]
        Solving the system yields \(x = -\frac{11}{30}\), \(y = \frac{4}{15}\), and \(z = -\frac{1}{15}\).
        The coordinates are \((-\frac{11}{30}, \frac{4}{15}, -\frac{1}{15})\).
    \end{enumerate}
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-6-ex-3}{}
    In each of the following cases determine, if it exists, the value of the parameter \(m \in \realnumbers\) for which a triplet of \emph{collinear} points in \(\mathbb{A}^3(\realnumbers)\) is obtained, that is, points lying on a line:\\
    \begin{align*}
        &\text{a) } \{(2, -1, 2), (1, 1, 1), (2, 1 - m, 4)\}; &&\text{c) } \{(1, -m, 0), (m, 1, 1), (1, -1, 3)\};\\
        &\text{b) } \{(3, 0, 0), (0, 1, 1), (m, m, m)\}; &&\text{d) } \{(1, m, 0), (2, \sqrt{2}, 1), (2, 110, 1)\}.
    \end{align*}
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-6-ex-3-sol}{}
    Three points \(A, B, C\) are collinear if the vectors \(\vec{AB}\) and \(\vec{AC}\) are linearly dependent (proportional).
    \begin{enumerate}[label=\alph*)]
        \item \(\vec{AB} = (-1, 2, -1)\) and \(\vec{AC} = (0, 2-m, 2)\). Since the first component of \(\vec{AC}\) is 0 while for \(\vec{AB}\) is \(-1\), they cannot be proportional for any \(m \in \realnumbers\). No such \(m\) exists.
        \item \(\vec{AB} = (-3, 1, 1)\) and \(\vec{AC} = (m-3, m, m)\). They are proportional if \((m-3, m, m) = k(-3, 1, 1)\). This implies \(k = m\) from the second component, so \(m-3 = -3m\), which gives \(m = \frac{3}{4}\).
        \item \(\vec{CA} = (0, 1-m, -3)\) and \(\vec{CB} = (m-1, 2, -2)\). They are proportional if one is a scalar multiple of the other, requiring \(m-1 = 0\) due to the first component, meaning \(m=1\). For \(m=1\), \(\vec{CA} = (0, 0, -3)\) and \(\vec{CB} = (0, 2, -2)\), which are not proportional. Thus, no such \(m\) exists.
        \item \(\vec{BA} = (-1, m-\sqrt{2}, -1)\) and \(\vec{BC} = (0, 110-\sqrt{2}, 0)\). Since the first component of \(\vec{BC}\) is 0 and for \(\vec{BA}\) is \(-1\), they cannot be proportional. Thus, no such \(m\) exists.
    \end{enumerate}
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-6-ex-4}{}
    In \(\mathbb{A}^4(\realnumbers)\) we fix the affine frame of reference \(\{0, e_1, e_2, e_3, e_4\}\) and consider the affine subspaces \[\pi = \{(x, y, z, t) \suchthat 2x + y - z + 3t = 2\} \quad \text{and} \quad \pi' = (1, 0, 0, 0) + \text{span}((3, 0, 1, 3), (2, -1, 0, -4)).\]
    Find parametric and Cartesian equations for \(\pi \cap \pi'\).
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-6-ex-4-sol}{}
    The parametric equations for \(\pi'\) can be written with parameters \(\alpha, \beta\) as:
    \[
        \begin{cases}
            x = 1 + 3\alpha + 2\beta \\
            y = -\beta \\
            z = \alpha \\
            t = 3\alpha - 4\beta
        \end{cases}
    \]
    By evaluating the Cartesian equation of \(\pi\) at these coordinates, we get the intersection:
    \[
        2(1 + 3\alpha + 2\beta) + (-\beta) - (\alpha) + 3(3\alpha - 4\beta) = 2 \implies 14\alpha - 9\beta = 0
    \]
    Setting \(\alpha = 9s\) and \(\beta = 14s\), we obtain the parametric equations for the intersection line \(\pi \cap \pi'\):
    \[
        \begin{cases}
            x = 1 + 55s \\
            y = -14s \\
            z = 9s \\
            t = -29s
        \end{cases}
    \]
    To find Cartesian equations for \(\pi \cap \pi'\), we first find the Cartesian equations of \(\pi'\).
    Eliminating parameters \(\alpha\) and \(\beta\) yields \(x + 2y - 3z = 1\) and \(4y + 3z - t = 0\).
    Thus, the Cartesian equations for \(\pi \cap \pi'\) are given by the system:
    \[
        \begin{cases}
            x + 2y - 3z = 1 \\
            4y + 3z - t = 0 \\
            2x + y - z + 3t = 2
        \end{cases}
    \]
\end{snippetsolution}

\begin{snippetexercise}{linear-algebra-batch-6-ex-5}{}
    In \(\mathbb A^3(\realnumbers)\) consider the lines \(l, l'\) with parametric equations
    \[
        \begin{cases}
            x = 3 + 2t \\
            y = -2 + t \\
            z = 4 + 5t
        \end{cases}, \quad \begin{cases}
            x = -3 + 2u \\
            y = 2-u \\
            z = 7-5u
        \end{cases}
    \]
    respectively.
    Determine whether the lines are coincident, parallel, intersecting or skew.
\end{snippetexercise}

\begin{snippetsolution}{linear-algebra-batch-6-ex-5-sol}{}
    The direction vectors of \(l\) and \(l'\) are \(v = (2, 1, 5)\) and \(v' = (2, -1, -5)\). Since they are not scalar multiples of each other, the lines are not parallel, nor do they coincide.
    To check if they intersect, we set their parametric equations equal (using different parameters \(t\) and \(u\)):
    \[
        \begin{cases}
            3 + 2t = -3 + 2u \\
            -2 + t = 2 - u \\
            4 + 5t = 7 - 5u
        \end{cases}
    \]
    The second equation implies \(t + u = 4\).
    The third equation implies \(5(t + u) = 3 \implies t + u = \frac{3}{5}\).
    This is a contradiction (\(4 \neq \frac{3}{5}\)), meaning the system has no solution. Since they are neither parallel nor intersecting, the lines are skew.
\end{snippetsolution}
