\section{Category of topological spaces}

\begin{snippetdefinition}{top-category-definition}{\(\mathbf{Top}\)}
    The \emph{\(\mathbf{Top}\) category} is defined as the
    \category where:
    \begin{enumerate}
        \item \(\catob(\mathbf{Top})\) is the class of all \topologicalspace[topological spaces];
        \item \(\forall A,B \in \catob(\mathbf{Set}), \cathom_\mathbf{Set}(A,B)\)
        is the \set of all \topologycontinuous[continuous functions] \(f\colon X \fromto Y\);
        \item composition is the \function composition.
    \end{enumerate}
\end{snippetdefinition}

\begin{snippetdefinition}{open-sets-category-definition}{Category of open sets}
    Let \(X\) be a \topologicalspace.
    The \emph{\(\text{Open}(X)\) category} is defined as the
    \category where:
    \begin{enumerate}
        \item \(\catob(\text{Open}(X))\) is the \set of all \topologicalspace[open sets] of \(X\);
        \item \(\forall U,V \in \catob(\text{Open}(X))\)
        \[
            \cathom_{\text{Open}(X)}(U,V)
            = 
            \begin{cases}
                \{\inclusionmap_{U,V}\} & \text{if } U \subseteq V, \\
                \varnothing & \text{otherwise},
            \end{cases}
        \]
        where \(\inclusionmap_{U,V} \colon U \inclusion V\) is the inclusion map;
        \item composition is given by composition of inclusions.
    \end{enumerate}
\end{snippetdefinition}
