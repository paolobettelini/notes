\section{Successione di funzioni}

\begin{radioactive}
\begin{snippetdefinition}{Successione di funzioni}
    Una \emph{successione di funzioni} è una famiglia di funzioni \(\{f_n\}_{n\in\naturalnumbers}\)
    definite su un dominio comune \(f_n \colon D \to \realnumbers\).
\end{snippetdefinition}
\end{radioactive}

\begin{radioactive}
\begin{snippetdefinition}{Convergenza in un punto}
    Sia \(\{f_n\}_{n\in\naturalnumbers}\) una successione di funzioni.
    La successione converge in un punto \(x_0\) se
    \[
        \lim_{n\to\infty} f_n(x_0) < \infty
    \]
\end{snippetdefinition}
\end{radioactive}

\begin{radioactive}
\begin{snippetdefinition}{Convergenza puntuale}
    Sia \(\{f_n\}_{n\in\naturalnumbers}\) una successione di funzioni.
    La successione \emph{converge puntualmente} ad una funzione \(f\colon D \to \realnumbers\) se
    \[
        \forall x\in D, \lim_{n\to\infty} f_n(x) = f(x)
    \]
\end{snippetdefinition}
\end{radioactive}

\begin{radioactive}
Quindi la successione converge in ogni punto, ma la velocità di convergenza può dipenderere dal punto.
\end{radioactive}

\begin{radioactive}
\begin{snippetexample}{Convergenza puntuale e limitatezza}
    Limite puntuali di funzioni limitate non sono necessariamente limitate.
    Prendiamo \(S = \realnumbers\) come dominio e \(f_n(x) = e^x \chi_{[-n, n]}(x)\), che sono limitate.
    Il candidato per la convergenza puntuale è \(f(x) = e^x\), che tuttavia non è limitato.
    Fissiamo quindi \(x \in S\) e \(\varepsilon > 0\) e cerchiamo
    \(N(\varepsilon, x)\) naturale che verifica la definizione di convergenza puntuale.
    Possiamo prendere \(N(\varepsilon, x) = \lceil |x| \rceil\) allora
    \begin{align*}
        |f_n(x) - f(x)| = 0 < \varepsilon
    \end{align*}
    per \(n > N\).
\end{snippetexample}
\end{radioactive}

\begin{radioactive}
\begin{snippetexample}{Convergenza puntuale e continuità}
    Limite puntuali di funzioni continue non sono necessariamente continue.
    Prendiamo \(S = [-1,1]\) come dominio e
    \[
        f_n(x) = -\chi_{[-1, -\frac{1}{n})}(x) + nx \chi_{[-\frac{1}{n}, \frac{1}{n}]}(x) + \chi_{(\frac{1}{n}, 1]}(x)
        = \begin{cases}
            -1 & x \in \left[-1, -\frac{1}{n}\right) \\
            nx & x \in \left[-\frac{1}{n}, \frac{1}{n}\right] \\
            1 & x \in \left(\frac{1}{n}, 1\right]
        \end{cases}
    \]
    che sono continue.
    \begin{center}
        \begin{tikzpicture}[scale=2]
            \draw[->] (-1.5, 0) -- (1.5, 0) node[right] {$x$};
            \draw[->] (0, -1.5) -- (0, 1.5) node[above] {$y$};

            \draw[dashed, gray] (0.35, 0) node[below, text=black] {$\frac{1}{n}$} -- (0.35, 1);
            
            \draw[dashed, gray] (0, 1) node[left, text=black] {$1$} -- (1, 1);
            \draw[dashed, gray] (0, -1) node[right, text=black] {$-1$} -- (-1, -1);

    Il candidato limite è \[
        f(x) = -\chi_{[-1, 0]}(x) + \chi_{[0,1]}(x) = \begin{cases}
            -1 & x \in [-1, 0) \\
            0 & x = 0 \\
            1 & x \in (0, 1]
        \end{cases}
    \]
    Notiamo anche che questa funzione non converge in maniera uniforme
    perché punti più vicini all'origine sono più difficile da essere avvicinati dagli \(f_n\)
    rispetto a punti più distanti.
    Fissiamo \(x \in S\) e \(\varepsilon > 0\).
    Vogliamo trovare \(N(x, \varepsilon)\) che soddisfi la definizione.
    Possiamo scegliere una \(N\) tale che \(N > \frac{1}{|x|}\).
\end{snippetexample}
\end{radioactive}

\begin{radioactive}
\begin{snippetexample}{Convergenza puntuale delle derivate della successione diverso dalla derivata del limite}
    Prendiamo \(S = \realnumbers\) come dominio e
    \[
        f_n(x) = \frac{\sin(nx)}{\sqrt{n}}
    \]
    Notiamo che \[
        |f_n(x)| \leq \frac{1}{\sqrt{n}} \to 0
    \]
    quindi \(f_n(x)\) converge puntualmente a \(f(x) = 0\).
    Tuttavia, \[
        f_n'(x) = \frac{n\cos(nx)}{\sqrt{n}} = \sqrt{n} \cos(nx)
    \]
    mentre \(f'(x) = 0\). Il limite di \(f_n'(x)\)
    non esiste per ogni \(x \in S\) e quindi non converge puntualmente a \(f'(x)\).
\end{snippetexample}
\end{radioactive}

\begin{radioactive}
\begin{snippetexample}{Convergenza puntuale degli integrali diverso dall'integrale del limite}
    Prendiamo \(S = \realnumbers\) come dominio e
    \[
        f_n(x) = n \chi_{(0, \frac{1}{n})}(x)
    \]
    Il candidato limite è \(f(x) = 0\). 
    Fissiamo \(x \in S\) e \(\varepsilon > 0\) e cerchiamo un
    \(N(\varepsilon, x)\) naturale che verifichi la definizione di
    convergenza puntuale.
    Distinguamo due casi:
    \begin{itemize}
        \item Se \(x \le 0\), allora \(x \notin (0, \frac{1}{n})\) per qualsiasi \(n \ge 1\), dunque \(f_n(x) = 0\). Possiamo scegliere banalmente \(N = 1\).
        \item Se \(x > 0\), affinché la funzione si annulli abbiamo bisogno che \(x \ge \frac{1}{n}\), ovvero \(n \ge \frac{1}{x}\). Possiamo allora prendere \(N(\varepsilon, x) = \lceil \frac{1}{x} \rceil\).
    \end{itemize}
    In entrambi i casi, per \(n > N\), otteniamo:
    \begin{align*}
        |f_n(x) - f(x)| = 0 < \varepsilon
    \end{align*}
    il che dimostra che \(f_n\) converge puntualmente a \(f=0\).
    Sia \(f_n\) che \(f\) sono integrabili secondo Riemann.
    Tuttavia,
    \[
        \integral[0][1][f_n(x)][x] = \integral[0][\frac{1}{n}][n][x] = 1
    \]
    mentre invece
    \[
        \integral[0][1][f(x)][x] = 0
    \]
    che non coincidono. Quindi, la successione numerica
    \[
        \left\{\integral[0][1][f_n(x)][x]\right\}_{n \in \naturalnumbers}
    \]
    non converge a \(\integral[0][1][f(x)][x]\).
\end{snippetexample}
\end{radioactive}

\begin{radioactive}
Vogliamo che la successione di funzioni si avvicini in maniera uniforme
alla sua funzione limite, cioè vogliamo rafforzare la definizione
rendendo la convergenza indipendente dalla coordinata.
Questo ci permette di avere una convergrnza che si comporta meglio rispetto alla limitatezza,
continuità, derivata e integrale.
\end{radioactive}

\begin{radioactive}
\begin{snippetdefinition}{Convergenza uniforme}
    Sia \(\{f_n\}_{n\in\naturalnumbers}\) una successione di funzioni.
    La successione \emph{converge uniformemente} ad una funzione \(f\colon D \to \realnumbers\) se
    \[
        \sup_{x\in D} \left| f_n(x) - f(x) \right| \to 0
    \]
    per \(n\to\infty\).
\end{snippetdefinition}
\end{radioactive}

\begin{radioactive}
Oppure possiamo dire che la condizione è che
\[
    \forall \varepsilon > 0, \exists N \suchthat \forall n>N, ||f_n-f||_{\infty, E} < \varepsilon
\]

Dovremmo dire che la differenza
\[
    |f_n(x) - f| \leq \varepsilon
\]
ma siccome ciò deve valere per tutte le \(x\) possiamo utilizzare il supremum.

Quindi la velocità di convergenza è la stessa in ogni punto. 
Ogni cosa che converge uniformemente converge puntualmente.
\end{radioactive}

\begin{radioactive}

\begin{radioactive}
In generale per la convergenza uniforme, vogliamo che \(|f_n(x) - f(x)| < \varepsilon\),
cioè \(f(x) - \varepsilon < f_n(x) - f(x) + \varepsilon\).
In altre parole, vogliamo che i grafici delle \(f_n\) siano definitivamente contenuti
nella regione delimitata dalle curve \(f-\varepsilon\) e \(f + \varepsilon\).
\end{radioactive}

\begin{radioactive}
\begin{snippettheorem}{Criterio di Cauchy}
    Una successione di funzioni \(\{f_n\}\) converge uniformemente a \(f \colon S \fromto \realnumbers\)
    in \(S\) se e solo se per ogni \(\varepsilon > 0\) esiste \(N(\varepsilon)\) naturale tale che
    \[
        \forall m,n > N,
        |f_n(x) - f_m(x)| < \varepsilon, \forall x \in S
    \]
\end{snippettheorem}
\end{radioactive}

\begin{radioactive}
\begin{snippetproof}{}
    \begin{iffproof}
        \item Fissiamo \(\varepsilon > 0\).
        Dalla definizione di convergenza uniforme,
        esiste \(N\left(\frac{\varepsilon }{2}\right)\) tale che
        \[
            \left|f_n(x) - f(x)\right| < \frac{\varepsilon}{2}
        \]
        per ogni \(n > N\) e \(x \in S\).
        Quindi, \begin{align*}
            \left|f_n(x) - f(x)\right| &\leq \left|f_n(x) - f(x)\right| + \left|f_n(x) - f(x)\right| \\
            &\leq \frac{\varepsilon}{2} + \frac{\varepsilon}{2} = \varepsilon
        \end{align*}
        \item Mostriamo la convergenza uniforme. Fissiamo \(x \in S\).
        Allora, per la condizione soddisfatta, \(\{f_n(x)\}\) è una successione
        numerica che verifica il criterio di Cauchy. Di conseguenza, esiste
        \(f_x\) tale che
        \[
            \lim_{n \to \infty} f_n(x) = f_x
        \]
        per ogni \(x \in S\) siccome l'ipotesi vale per tutte le \(x\).
        Definiamo \(f \colon S \fromto \realnumbers\) data da
        \[
            f(x) \triangleq f_x
        \]
        Per iptoesi per ogni \(\varepsilon > 0\) esiste \(N(\varepsilon)\) naturale tale che
        \begin{align*}
            |f_n(x) - f_m(x)| < \varepsilon
        \end{align*}
        per tutte le \(m,n > N\) e \(x \in S\).
        Prendendo il limite con \(m \to \infty\), per definizione di \(f\), otteniamo
        \[
            |f_n(x) - f(x)| < \varepsilon
        \]
        per tutte le\(m,n > N\) e \(x \in S\).
    \end{iffproof}
\end{snippetproof}
\end{radioactive}

Vediamo ora come si comporta la convergenza uniforme rispetto
alle proprietà studiate precedentemente.

\begin{snippetdefinition}{Limitatezza successione di funzioni}
    Diciamo che la successione di funzioni \(\{f_n\}\)
    è \emph{limitata} in \(S\) se \(\forall n \in \naturalnumbers\), \(\exists r_n > 0\)
    tale che \(|f_n(x) \leq M_n\).
\end{snippetdefinition}

\begin{snippetdefinition}{Equilimitatezza successione di funzioni}
    Diciamo che la successione di funzioni \(\{f_n\}\)
    è \emph{equilimitata} in \(S\) se \(\exists M > 0\) tale che
    \(\forall n \in \naturalnumbers\), \(|f_n(x) \leq M\).
\end{snippetdefinition}

Equilimitato implica limitato.

\begin{snippetproposition}{}
    Sia \(\{f_n\}\) convergente uniformemente a \(f\) in \(S\)
    con \(f \colon S \fromto \realnumbers\).
    Se \(f_n\) è limitata per ogni \(n \in \naturalnumbers\),
    allora \(f\) è limitata in \(S\) e \(\{f_n\}\) è equilimitata.
\end{snippetproposition}

\begin{snippetproof}{}
    Per la definizione di convergenza uniforme, per ogni \(\varepsilon > 0\)
    esiste \(N(\varepsilon)\) naturale tale che
    \[
        |f_n(x) - f(x)| < \varepsilon, \quad \forall n > N, x \in S
    \]
    Fissiamo arbitrariamente \(\varepsilon = 1\) e prendiamo
    \(N_1 = N(1)\), che verifica la defiinzionoe di convergenza uniforme con \(\varepsilon = 1\).
    Allora,
    \begin{align*}
        |f(x)| &\leq |f(x) - f_{N_1}(x)| + |f_{N_1}(x)| \\
        &\leq 1 + |f_{N_1}(x)| \leq 1 + M_{N_1}
    \end{align*}
    dove \(M_{N_1} > 0\) tale che \(|f_{N_1}(x)| \leq M_{N_1}\) per ogni \(x \in S\), siccome è limitata.
    Siccome \(N_1\) è fissato, esiste \(M>0\) tale che \(|f(x)| \leq M\) per \(x \in S\).
    Proviamo ora la equilimitatezza di \(\{f_n\}\) in \(S\).
    Abbiamo visto che \begin{align*}
        |f_n(x)| \leq |f_n(x) - f(x)| + |f(x)| \leq 1 + M 
    \end{align*}
    Ma per ipotesi sappiamo che per ogni \(n \in \naturalnumbers\),
    esiste \(M_n\) tale che \(|f_n(x)| < M_n\) per ogni \(x \in S\).
    Allora definiamo \(M' = \max\{M_1, \cdots, M_{N_1}, 1 + M\}\), vale che
    \[
        |f_n(x)| \leq M', \quad \forall x \in S, \forall n \in \naturalnumbers
    \]
\end{snippetproof}

Non vale il viceversa.

\begin{snippetexample}{L'equilimitatezza non implica nemmeno la convergenza puntuale.}
    Consideriamo \(f_n(x) = \sin(nx)\), che ovviamente è equilimitata da \(M=1\).
    Tuttavia, non converge puntualmente in \(S = [0, 2\pi]\). Basta prendere \(x = \frac{\pi}{2}\)
    \begin{align*}
        \sin\left(n \frac{\pi}{2}\right) = \begin{cases}
            1 & n = 1 + 4k \\
            0 & n = 2k \\
            -1 & n = 3 + 4k
        \end{cases}
    \end{align*}
\end{snippetexample}

Non vale nemmeno un analogo di Bolzano Weierstrass.

\begin{snippetexample}{L'equilimitatezza non implica un analogo di Bolzano Weierstrass con convergenza puntuale.}
    Consideriamo \(f_n(x) = \sin(nx)\), che ovviamente è equilimitata da \(M=1\).
    Supponiamo che esista una sottosuccessione \(\{f_{n_k}\}\) che converge puntualmente
    con \(n_k \to \infty\) per \(k \to \infty\).
    Definisco
    \[
        g_k(x) = \left(f_{n_k}(x) - f_{n_{k+1}}(x)\right)^2
        = \left(\sin(n_k x) - \sin(n_{k+1}x)\right)^2
    \]
    è immediato verificare che \(g_k\) converge puntualmente a zero in \(S\),
    poiché \(\{f_{n_k}\}\) converge in \(S\).
    Inoltre, \(|g_k| \leq 4\).
    Per il teorema della convergenza dominata, che vedremo in futuro,
    vale \begin{align*}
        \lim_{k \to \infty} \integral[0][2\pi][g_k(x)][x]
        = \integral[0][2\pi][\lim_{k\to\infty} g_k(x)][x] = 0
    \end{align*}
    Tuttavia, abbiamo che \(\forall k \in \naturalnumbers\),
    \[
        \integral[0][2\pi][g_k(x)][x]
        = \integral[0][2\pi][\left(\sin(n_k x) - \sin(n_{k+1}x)\right)^2][x]
        = 2\pi
    \]
    che è assurdo \lightning.
\end{snippetexample}

\begin{snippetexercise}{}
    Dimostrare che per ogni \(i,j\) naturali vale \[
        \integral[0][2\pi][\sin(ix)\sin(jx)][x] = \begin{cases}
            0 & i \neq j \\
            \pi & i = j
        \end{cases}
    \]
\end{snippetexercise}

Studiamo ora la continuità e la convergenza uniforme.

\begin{snippettheorem}{Scambio del limite}
    Siano \(\{f_n\}\) una successione di funzioni uniformemente convergente ad \(f\)
    in \(S\) con \(f \colon S \fromto \realnumbers\).
    Sia \(x_0 \in S\) punto di accumulazione tale che
    \[
        \exists \lim_{x\to x_0} f_n(x) \in \realnumbers, \quad \forall n \in \naturalnumbers
    \]
    Allora, vale che
    \[
        \lim_{n \to \infty} \lim_{x \to \infty} f_n(x)
        = \lim_{x \to \infty} \lim_{n \to \infty} f_n(x)
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    Fissiamo \(x_0 \in S\) punto di accumulazione
    e definiamo \[ A_n \triangleq \lim_{x \to x_0} f_n(x) \] per \(n \in \naturalnumbers\).
    Per il criterio di Cauchy, siccome abbiamo convergenza uniforme,
    esiste \(N(\varepsilon)\) naturale tale che
    \[
        |f_n(x) - f_m(x)| < \varepsilon
    \]
    per tutte le \(m,n > N\) e \(x \in S\).
    La funzione \(x \to |x|\) è continua quindi se
    facciamo tendere \(x \to x_0\) nell'ultima espressione,
    otteniamo
    \[
        |A_m - A_n| < \varepsilon
    \]
    per tutte le \(m,n > N\) e \(x \in S\).
    Di conseguenza, \(\{A_n\}\) è una successione numerica che verifica il criterio di Cauchy.
    Quindi, converge e definiamo
    \[
        A \triangleq \lim_{n \to \infty} A_n
    \]
    Quindi la parte sinistra di 
        \[
        \lim_{n \to \infty} \lim_{x \to \infty} f_n(x)
        = \lim_{x \to \infty} \lim_{n \to \infty} f_n(x)
    \]
    ha un senso. Ci manca da dimostrare che
    \[
        \lim_{x\to x_0} f(x) = A
    \]
    Cioè, \(\forall \varepsilon > 0\), \(\exists \delta(\varepsilon, x)\)
    tale che \(|f(x) - A| < \epsilon\) per ogni \(x\in S \difference \{x_0\}\) tale che \(|x-x_0| < \delta\).
    Sappiamo che \(f_n\) converge uniformemente a \(f\) quindi esiste \(N\left(\frac{\varepsilon}{3}\right)\)
    naturale tale che
    \begin{align*}
        |f_n(x) - f(x)| < \frac{\varepsilon}{3}, \quad \forall n > N, x \in S
    \end{align*}
    Inoltre, sappiamo che \(A_n\) converge ad \(A\) per \(n \to +\infty\),
    e quindi esiste \(N_2\left(\frac{\varepsilon}{3}, x_0\right)\) naturale tale che 
    \[
        |A_m - A| < \frac{\varepsilon}{3}, \quad \forall n \geq N_2
    \]
    Per definizione, sappiamo che per ogni \(n \in \naturalnumbers\),
    \[
        A_n \triangleq \lim_{x\to x_0} f_n(x)
    \]
    quindi fissato \(M > \max\{N_1, N_2\}\), esiste
    \(\overline\delta(M, x_0, \varepsilon)\) tale che
    
    \[
        |f_{M}(x) - A_{M}| < \frac{\varepsilon}{3},
        \quad \forall x \in S \difference \{x_0\} \suchthat |x-x_0| < \overline\delta
    \]
    Combinando tutto, otteniamo che
    \begin{align*}
        |f(x) - A| &\leq |f(x) - f_{M}(x)| + |f_{M}(x) - A_{M}| + |A_{M} - A| \\
        &\leq \frac{\varepsilon}{3} + \frac{\varepsilon}{3} + \frac{\varepsilon}{3} = \varepsilon
    \end{align*}
    per tutte le \(x \in S \difference \{x_0\}\) tale che \(|x-x_0| < \overline\delta\).
\end{snippetproof}

Notiamo che se \(\{f_n\}\) converge a \(f\), \(x_0 \in S\)
punto di accumulazione tale che
\[
    \lim_{x\to x_0} f(x) = +\infty, \quad \forall n \in \naturalnumbers
\]
allora
\[
    \lim_{x \to x_0} f(x) = +\infty
\]
e analogamente per \(-\infty\).
Avremmo potuto unificare il tutto con gli intorni della retta reale estesa.


Se richiediamo che \(f_n(x)\) siano continue allora il limite per \(x \to x_0\)
è precisamente \(f_n(x_0)\), e quindi da questo teorema otteniamo immediatamente il corollario.

\begin{snippetcorollary}{}
    Siano \(\{f_n\}\) una successione di funzioni continue uniformemente convergente ad \(f\)
    in \(S\) con \(f \colon S \fromto \realnumbers\).
    Allora \(f\) è continua su \(S\) e \(\forall x_0 \in S\) punto di accumulazione vale
    \[
        \lim_{n \to \infty} \lim_{x \to x_0} f_n(x)
        = \lim_{x \to x_0} \lim_{n \to \infty} f_n(x) = f(x_0)
    \]
\end{snippetcorollary}

\begin{snippetproof}{}
    per esercizio dimostrarla usando la continuità
    per successioni. Ora la dimostriamo con la definizione classica.
    Dobbimo mostrare che per ogni \(x_0 \in S\),
    la funzione \(f\) è continua in \(x_0\),
    cioè \(\forall \varepsilon > 0\) esiste \(\delta(x_0, \varepsilon) > 0\)
    tale che \[
        |f(x) - f(x_0)| < \varepsilon, \quad \forall x \in S \suchthat |x-x_0| < \delta
    \]
    Fissiamo \(x_0 \in S\) e \(\varepsilon > 0\).
    Visto che \(f_n\) converge uniformemente ad \(f\), esiste \(N\left(\frac{\varepsilon}{2}\right)\)
    naturale tale che
    \[
        |f_n(x) - f(x)| < \frac{\varepsilon}{3}, \quad \forall n > N, \forall x \in S
    \]
    Fissiamo \(n_0 > N\).
    Per ipotesi, \(f_{n_0}\) è continua in \(x_0\),
    quindi esiste \(\overline \delta\left(x_0, n_0, \frac{\varepsilon}{3}\right)\) tale che
    \[
        |f_{n_0}(x) - f_{n_0}(x_0)| < \frac{\varepsilon}{3},
        \quad \forall x \in S \suchthat |x-x_0| \leq \overline\delta
    \]
    Combinando il tutto otteniamo che \(\forall x \in S\) tale che \(|x-x_0| < \overline\delta\) vale
    \begin{align*}
        |f(x) - f(x_0)| &\leq |f(x) - f_{n_0}(x)| + |f_{n_0}(x) - f(x_0)|
        + |f_{n_0}(x) - f_{n_0}(x_0)| \\
        &\leq \frac{\varepsilon}{3} + \frac{\varepsilon}{3} + \frac{\varepsilon}{3} = \varepsilon
    \end{align*}
    Quindi scegliendo \(\delta = \overline \delta\) viene verifica la definizione di continuità in \(x_0\)
    per \(f\).
\end{snippetproof}

\begin{snippetexample}{Convergenza puntuale non implica convergenza uniforme}
    Prendiamo \(S = [0,1]\) come dominio e \(f_n(x) = n^2 x (1-x^2)^n\)
    converge puntualmente a \(f(x) = 0\) ma non in maniera uniforme.
\end{snippetexample}

\begin{snippettheorem}{Teorema del dini (un altro)}
    Supponiamo di avere un compatto \(S=K\) compatto. Prendiamo
    \begin{enumerate}
        \item \(\{f_n\}\) successione di funzioni continue su \(K\);
        \item \(\{f_n\}\) converge puntualmente a \(f \colon K \fromto \realnumbers\) continua;
        \item \(\{f_n\}\) successione monotona non crescente, cioè
    \end{enumerate}
    Allora, \(\{f_n\}\) converge uniformemente in \(K\) a \(f\).
\end{snippettheorem}

\begin{snippetproof}{}
    Definiamo \(g_n \triangleq f_n - f\).
    Notiamo che, dalle ipotesi, valgono:
    \begin{enumerate}
        \item \(g_n \to g\) puntualmente con \(g=0\);
        \item \(g_n \colon K \fromto \realnumbers\) continua per tutte le \(n\in\naturalnumbers\);
        \item \(\{g_n\}\) monotona non crescente
    \end{enumerate}
    Inoltre, notiamo che se dimostriamo che \(g_n\) converge uniformemente a \(g\), allora abbiamo finito.
    Fissiamo \(\varepsilon>0\). Definiamo
    \[
        K_n \triangleq \left\{
            x \in K \suchthat |g_n(x)| \geq \varepsilon
        \right\}
    \]
    cioè dove non è verificata la definizione di convergenza uniforme.
    Siccome \(x \fromto |g_n(x)|\) è continua, abbiamo che \(K_n\) è chiuso e \(K_n \subseteq K\).
    Siccome è un sottoinsieme chiuso di un compatto, è compatto. Inoltre dalla monotonia di
    \(\{g_n\}\) sono annidati \(K_{n+1} \subseteq K_n\).
    Prendiamo \(x \in K\). Per ipotesi, \(g_n(x) \to 0\) per \(n \to +\infty\).
    Di conseguenza, esiste \(N(x) \in \naturalnumbers\) tale che \(x \notin K_n\)
    \(\forall n \geq N\).
    Quindi, \[ x \notin \bigcap_{k=1}^\infty K_n \]
    Possiamo ripetere il medesimo ragionamento per tutte le \(x \in K\) quindi otteniamo
    \[
        \bigcap_{n=1}^\infty K_n = \emptyset
    \]
    Sappiamo che l'intersezione numerabile di compatti annidati è non vuota (topologia citare il teorema).
    Quindi se l'intersezione è vuota, almeno uno deve essere vuota (contronominale).
    Per definizione di \(K_n\), esiste un vale che
    \begin{align*}
        |g_n(x)| \leq \varepsilon, \quad \forall x \in K \difference K_n = K \difference \emptyset = K
    \end{align*}
    per ogni \(n > N\).
\end{snippetproof}

\begin{snippetexample}{}
    Consideriamo \(S = (0, 1]\) e
    \[
        f_n(x) = \frac{1}{nx + 1}
    \]
    È facile verificare che \(f_n \to f\) puntualmente con \(f = 0\).
    Non possiamo applicare il teorema in quanto il dominio non è compatto.
    Infatti, preso \(\varepsilon = \frac12\), comunque io prenda \(N \in \naturalnumbers\)
    per soddisfare la definizione, troov sempre un \(x \in S\) tale che \(x \in K_n\)

Alternatively, we can say that the condition is
\[
    \forall \varepsilon > 0, \exists N \suchthat \forall n>N, ||f_n-f||_{\infty, E} < \varepsilon
\]
We should say that the difference
\[
    |f_n(x) - f| \leq \varepsilon
\]
but since this must hold for all \(x\), we can use the supremum.
Questo è come dire che il limite della successione
\[
    \lim_n \left(\sup_{x\in S} |f_n(x) - f(x)\right) = 0
\]

\begin{snippetproposition}{Criteri convergenza uniforme}
    Supponiamo che esista una successione \(\{g_n\}\) e \(C>0\) tale che
    \[
        \sup_{x\in S} |f_n(x) - f_m(x)| \leq C \cdot \sup_{x\in S}|g_n(x) - g_m(x)|
    \]
    Allora se \(\{g_n\}\) converge uniformemente, anche \(\{f_n\}\) converge uniformemente.
    Viceversa, se troviamo una successione di punti \(\{x_n\} \subseteq S\)
    tali che
    \begin{align*}
        \lim_n |f_n(x_n) - f(x_n)| > 0
    \end{align*}
    Allora \[
        \sup_{x\in S} |f_n(x) - f(x)| \geq |f_n(x_n) - f(x_n)|
    \]
    cioè il supremum è sicuramente maggiore dell'espressione valutata in un singolo punto.
    In questo caso \(\{f_n\}\) non converge uniformemente.
\end{snippetproposition}

\begin{snippetdefinition}{Spazio funzioni continue limitate}
    \[
        C_b(S) = \{
            f \colon S \fromto \realnumbers \suchthat
            f \text{ continua e limitata}    
        \}
    \]
\end{snippetdefinition}

Su questo spazio abbiamo la norma
\[
    ||f||_{\infty} \triangleq \sup_{x \in S} |f(x)|
\]
Notiamo che \(||f_n-f||_\infty\) non è altro che la distanza \(d_\infty\)
indotta dalla \(||\cdot||_\infty\) sullo spazio \(C_b(S)\),
dove \(d_\infty(g, f) = ||f-g||_\infty\).

Abbiamo dimostrato complessivamente che \(C_b(S)\) è uno spazio completo.

\begin{snippettheorem}{}
    \(C_b(S)\) è uno spazio di Banach rispetto alla norma infinito.
\end{snippettheorem}

\begin{snippetproof}{}
    Consideriamo una successione \(\{f_n\}\) nello spazio \(C_b(S)\) di Cauchy, cioè
    \[
        \lim_{m,n} d_\infty(f_n, f_m) = \lim_{m,n} ||f_n - f_m||_\infty = 0
    \]
    Vogliamo verificare che esiste quindi \(f \in C_b(S)\) tale che
    \[
        \lim_n d_\infty(f_n, f) = \lim_n ||f_n - f|| = 0
    \]
    Ma questo seguente dai risultati precedente e dalle definizioni equivalenti di convergenza uniforme.
    Infatti, il criterio di Cauchy implica
    che esista \(f \colon S \fromto \realnumbers\) tale che \(f_n\) converge uniformemente a \(f\) in \(S\).
    Inoltre, per ipotesi \(f_n\) continua e limitata.
    Quindi, \(f\) è pure continua e limitata su \(S\), cioè \(f \in C_b(S)\).
\end{snippetproof}

\begin{snippettheorem}{Stone-Weierstrass}
    Sia \(S=K\) compatto.
    Allora \(\forall f \in C_b(K)\), esiste \(\{P_n\}\) successione di polinomi
    \(\realnumbers[x]\) (anche complessi) tali che \(P_n \to f\) uniformemente in \(K\).
    In altre parole, se consideriamo \(P(K)\) l'insieme dei polinomi a coefficienti reali, abbiamo che
    \(P(K)\) è denso in \(C_b(K)\) rispetto alla topologia indotta
    dalla norma infinito.
\end{snippettheorem}

\subsection{Teorema di Ascoli-Arzelà}

Il seguente teorema è un analogo di Bolzano-Weierstrass, ma bisogna aggiungere una condizione.
Cerchiamo delle condizioni sulla nostra successione di funzioni che garantisca che
si possa estrarre una sottosuccessione convergente.
Oltre all'equilimitatezza ci serve l'equicontinuità.

\begin{snippetdefinition}{Equicontinuità}
    Diciamo che la successione \(\{f_n\}\) è \emph{equicontinua}
    su \(S\), se \(\forall \varepsilon > 0\) esiste \(\delta(\varepsilon) > 0\)
    tale che se \(|x-y| < \delta\) per \(x,y \in S\), allora
    \[
        |f_n(x) - f_n(y)| < \varepsilon
    \]
    per \(n \in \naturalnumbers\).
\end{snippetdefinition}

Quindi è una sorta di continuità uniforme 
L'equicontinuità implica che \(f_n\) sia uniformemente continua su \(S\), ma non il contrario,
cioè se tutti le \(f_n\) sono uniformemente continue il \(\delta\) dell'uniforme continuità delle \(f_n\)
potrebbe dipendere da \(n\).

\begin{snippetproposition}{}
    Se tutte le \(f_n\) sono lipschitziane su \(S\),
    un criterio di equicontinuità è che esista \(L > 0\) indipendente da \(n\)
    che verifica la definizione di lipschitzianità per ogni \(f_n\).
\end{snippetproposition}

\begin{snippetproof}{}
    Abbiamo \[
        |f_n(x) - f_n(y)| \leq L|x-y|
    \]
    e prendiamo \(\delta < \varepsilon / L\).
\end{snippetproof}

In particolare se \(f_n\) è \(\mathcal C^1\),
allora è sufficiente che \(\{f_n'\}\) sia equilimitata per garantire l'equilipschizianità,
come in analisi I.

\begin{snippetproposition}{}
    Sia \(S=K\) compatto e \(\{f_n\}\) successione di funzioni continue
    su \(K\) tali che \(f_n\) converge uniformemente a \(f\) in \(K\)
    con \(f \colon K \fromto \realnumbers\).
    Allora, \(\{f_n\}\) è equicontinua.
\end{snippetproposition}

\begin{snippetproof}{}
    Notiamo che, visto che \(K\) è compatto, questa proposizione si lega con il fatto
    che se \(f_n\) continua su un compatto, quindi limitata, allora \(\{f_n\}\)
    equilimitata poiché converge uniformemente. % link prop
    Vogliamo mostrare che \(\forall \varepsilon > 0\)
    esiste \(\delta(\varepsilon) > 0\) tale che
    per ogni \(x,y \in K\) quando \(|x-y|<\delta\) vale
    \[
        |f_n(x) - f_n(y)| < \epsilon 
    \]
    Per la definizione di convergenza uniforme esiste
    \(N\left(\frac{\varepsilon}{3}\right)\)
    naturale tale che
    per ogni \(n \geq N\) e \(x \in K\)
    vale \[
        |f_n(x) - f(y)| < \varepsilon
    \]
    inoltre, \(f\) è limite unifome di funzioni continue,
    quindi è continua.
    Siccome è continua su un compatto, è uniformemente continua.
    Di conseguenza esiste \(\delta_0\left(\frac{\varepsilon}{3}\right)\) tale che
    \[
        |f(x) - f(y)| < \frac{\varepsilon}{3}
    \]
    per tutte le \(x,y \in K\) ove \(|x-y| < \delta_0\).
    Combinando tutto otteniamo che \(\forall n > N\)
    \begin{align*}
        |f_n(x) - f_m(y)| &\leq |f_n(x) - f(x)| + |f(x) - f(y)| + |f_n(y) - f(y)| \\
        &\leq \frac{\varepsilon}{3} + \frac{\varepsilon}{3} + \frac{\varepsilon}{3}
        = \varepsilon
    \end{align*}
    Però per dimostrare l'equicontinuità
    dobbiamo trovare un \(\delta\) che vada bene anche per \(f_1, \cdots, f_{N}\).
    Anche loro sono continue su un compatto quindi uniformemente continue.
    Di conseguenza,
    esistono \(\delta_i(\varepsilon) > 0\) tale che
    \[
        |f_i(x) - f_i(y)| < \varepsilon
    \]
    per ogni \(x,y \in K\)
    dove \(|x-y| < \delta_i\).
    Possiamo quindi prendere \(\min\{\delta_0, \delta_1, \cdots, \delta_N\}\).
\end{snippetproof}

\begin{snippetlemma}{Lemma di Ascoli-Arzelà}
    Sia \(S=K\) compatto e \(\{f_n\}\) una successione di funzioni equicontinua
    su \(K\) che converge puntualmente ad \(f \colon S \fromto \realnumbers\) in \(K\).
    Allora, \(\{f_n\}\) converge uniformemente a \(f\).
\end{snippetlemma}

Quindi \(f\) è pure continua.

\begin{snippetproof}{}
    Per il criterio di Cauchy,
    ci basta dimostrare che per ogni \(\varepsilon > 0\) esiste \(N(\varepsilon)\) naturale tale che
    \[
        \forall m,n > N,
        |f_n(x) - f_m(x)| < \varepsilon, \forall x \in K
    \]
    Dall'equicontinuità, \(\exists \delta\left(\frac{\varepsilon}{3}\right) > 0\)
    tale che \(\forall x,y \in K\) e \(i \in \naturalnumbers\)
    \[
        |x-y| < \delta \implies
        |f_i(x) - f_i(y)| < \varepsilon
    \]
    Visto che \(K\) è compatto, esistono \(x_1, \cdots, x_N \in K\)
    tale che
    \[
        K \subseteq \bigcup_{j=1}^N B_\delta(x_j)
    \]
    Visto che \(\{f_n\}\) converge puntualmente, in \(K\)
    per il criterio di Cauchy per successioni numeriche,
    \(\forall J \in \{1, \cdots, N\}\) esiste \(N_J\left(\frac{\varepsilon}{3}, x_j\right)\)
    naturale tale che
    \[
        |f_n(x_j) - f_m(x_j)| < \frac{\varepsilon}{3}
    \]
    per tutte le \(m,n > N_J\).
    Prendiamo \(\overline N = \max \{N_1, \cdots, N_N\}\).
    Combinando tutto
    \begin{align*}
        |f_n(x) - f_m(x)| &\leq |f_n(x) - f_n(x_i)| + |f_m(x_i) - f_m(x)|
        + |f_n(x_i) - f_m(x_i)|
    \end{align*}
    dove \(x_i \in K\) tale che \(x \in B_\delta(x_i)\), che esiste sempre.
    Quindi, \(|x-x_i| < \delta \) e di conseguenza
    \[
        |f_n(x) - f_m(x)| < \frac{\varepsilon}{3} + \frac{\varepsilon}{3} + \frac{\varepsilon}{3} = \varepsilon
    \]
\end{snippetproof}

\begin{snippettheorem}{Ascoli-Arzelà}
    Prendiamo \(S=K\) compatto e consideriamo
    \(\{f_n\}\) successione di funzioni
    \begin{enumerate}
        \item equicontinua su \(K\);
        \item equilimitata su \(K\);
    \end{enumerate}
    Allora, esiste una sottosuccessione uniformemente convergente in \(K\).
\end{snippettheorem}

\begin{snippetproof}{Ascoli-Arzelà}
    Per il lemma ci basta mostrare che esiste una sottosuccessione
    che converge puntualmente.
    Prendimo \(r_1 \in K \intersection \rationalnumbers\).
    Consideriamo la successione \(\{f_n(r_1)\}\).
    Grazie all'equilimitatezza essa è limitata, e quindi per
    Bolzano-Weierstrass esiste una sottosuccessione \(\{f_{n_k}(r_1)\}\)
    convergente. Denotiamo per comodità \(f^1_n \triangleq f_{n_k}\).
    Ripetiamo il ragionamento prendendo \(r_2 \in K \intersection \rationalnumbers\)
    diverso da \(r_1\) e consideriamo la successione \(\{f^1_n(r_2)\}_{n \in J}\)
    con \(J \subseteq \naturalnumbers\) infinito.
    Per lo stesso ragionamento di prima estratto una sottosuccessione convergente
    \(\{f^2_{n_k}(r_2)\}_{k\in\naturalnumbers}\) e denotiamo \(f^2_n \triangleq f^1_{n_k}\).
    Ripetiamo il processo all'infinito e quindi otteniamo una famiglia di sottosuccessioni
    \[
        \{f_n\} \supseteq \{f_n^1\} \supseteq \{f_n^3\} \cdots
    \]
    Consideriamo la diagonale del seguente diagramma:

    è contenuta in \(\{f_n\}\)
    allora è equicontinua, quindi esiste per ogni \(\varepsilon > 0\) un \(\delta\left(\frac{\varepsilon}{3}\right)\)

    \]
    Combinando il tutto ottengo che
    \begin{align*}

    L'epsilon dipende da \(x\) ma noi vogliamo la convergenza puntuale.

\begin{snippetexample}{}
    Questo teorema ha ipotesi molto forti,
    ma il risultato è tuttavia banale-
    Anche in casi ``buoni'' non è scontato poter estrarre una sottosuccssione convergente.
    Prendiamo \(S = K = [0,1]\) come dominio e
    \[
        f_n(x) = \frac{x^2}{x^2 + (1-nx)^2}
    \]
    È facile notare che questa converge puntualmente ad \(f=1\).
    Inoltre \(|f_n(x)| \leq 1\)
    e \(f_n(1/n) = 1\).
    Se esistesse una sottosuccessione
    \(\{f_n\}\) dovrebbe convergere ad \(f\), tuttavia
    \[
        \sup_{x \in [0,1]} |f_{n_k} - f| = \sup_{x \in [0,1]}
        |f_{n_k}(x)| > f_{n_k}\left(\frac{1}{n_k}\right) > 1
    \]
    che è assurdo perché il sup è sempre maggiore o uguale della funzione valutata in un punto.
\end{snippetexample}

\begin{snippetexercise}{}
    Siano \(I\) un intervallo e \(f_n\colon I \fromto \realnumbers\) una successione di funzioni derivabili su \(I\). Provare che
    se \(\{f_n(x_0)\}\) converge per qualche \(x_0 \in I\) e se \(\{f'\}\) converge localmente uniformemente in \(I\),
    allora \(\{f_n\}\) converge localmente uniformemente in \(I\) ad una primitiva di \[\lim_{n\to \infty} f_n'\].
\end{snippetexercise}

Studiamo ora la convergenza uniforme e l'integrazione.

\begin{snippettheorem}{}
    Sia \(S = [a,b]\) intervallo reale e \(\{f_n\}_{n \in \naturalnumbers}\) una successione di funzioni Riemann integrabili
    su \(S\) e \(f_n \to f\) uniformemente con \(f \colon [a,b] \fromto \realnumbers\). Allora, \(f\) è Riemann integrabile
    in \([a,b]\) e vale
    \[
        \integral[a][b][f(x)][x] = \lim_{n} \integral[a][b][f_n(x)][x]
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    Come prima cosa dobbiamo dimostrare che \(f\) è Riemann integrabile su \(S\). Data la convergenza uniforme, 
    \(\forall \varepsilon > 0\) esiste \(N(\varepsilon)\) naturale tale che
    \[
        \sup_{x\in [a,b]} |f_n(x) - f(x)| < \varepsilon, \quad \forall n \geq N
    \]
    Quindi
    \[
        f_n(x) - \frac{\varepsilon}{b-a} \leq f(x) \leq f_n(x) + \frac{\varepsilon}{b-a}, \quad \forall x \in [a,b]
    \]
    Presa una partizione finita di \([a,b]\) \(a = t_0 < t_1 < \cdots < t_n = b\) definiamo
    \(I_k = [x_{k-1}, x_k]\) con \(k \in \{1,\cdots, n\}\). Dall'espressione sopra otteniamo che per tutti i \(k\),
    \[
        \inf_{x\in I_k} f_n(x) - \frac{\varepsilon}{b-a} \leq \inf_{x \in I_k} f(x) \leq \inf_{x\in I_k} f_n(x) + \frac{\varepsilon}{b-a}
    \]
    e
    \[
        \sup_{x\in I_k} f_n(x) - \frac{\varepsilon}{b-a} \leq \sup_{x \in I_k} f(x) \leq \sup_{x\in I_k} f_n(x) + \frac{\varepsilon}{b-a}
    \]
    Otteniamo quindi che
    \begin{align*}
        &\sum_{k=1}^n \left(\sup_{x\in I_k} f_n(x)\right)(t_k - t_{k-1}) - \frac{\varepsilon}{b-a} \sum_{k=1}^n (t_k - t_{k-1})
        \leq \sum_{k=1}^n \left(\sup_{x\in I_k} f(x)\right)(t_k - t_{k-1}) \\
        &\leq \sum_{k=1}^n \left(\sup_{x\in I_k} f_n(x)\right)(t_k - t_{k-1}) + \frac{\varepsilon}{b-a} \sum_{k=1}^n (t_k - t_{k-1})
    \end{align*}
    che sono quindi le somme \(S(f_n, P)\) e \(S(f, P)\).
    Allora
    \[
        S(f_n, P) - \varepsilon \leq S(f, P) \leq S(f_n, P) + \varepsilon
    \]
    dove \(P\) è la partizione data.
    Nello stesso modo, usando la versione con l'infimum, otteniamo
    \[
        s(f_n, P) - \varepsilon \leq s(f, P) \leq s(f_n, P) + \varepsilon
    \]
    dove
    \[
        s(f, P) = \sum_{k=1}^n \inf_{x\in I_k} f(x)(t_k-t_{k-1})
    \]
    Questo vale per ogni partizione \(P\) di \([a,b]\). Per defininizione di inegrale di Riemann \(f\) è integrabile se
    \[
        \sup_P S(f,P) = \sup_P s(f,P) < \infty
    \]
    dove \(P\) rappresenta tutte le possibili partizioni.
    Se questo è vero
    \[
        \integral[a][b][f(x)][x] = \sup_P S(f, P) = \sup_P s(f,P)
    \]
    applicando questa definizione di integrabilità e facendo il supremum sulle partizioni otteniamo 
    \[
        \integral[a][b][f_n(x) - \varepsilon][x] \leq \sup_P s(f,P) \leq \sup_P S(f, P) \leq \integral[a][b][f_n(x) + \varepsilon][x]
    \]
    che possiamo fare siccome \(f_n\) è integrabile.
    Quindi concludiamo che
    \[
        \left|
            \sup_P S(f, P) - \sup_P s(f,P)
        \right| \leq \varepsilon
    \]
    di conseguenza
    \[
        \integral[a][b][f(x)][x] = \sup_P S(f, P) = \sup_P s(f, P)
    \]
    cioè \(f\) è integrabile.
    Mostriamo ora la proposizione con il limite.
    Sappiamo che \(\forall \varepsilon > 0\) esiste \(N(\varepsilon)\) naturale tale che
    \[ |f_n(x) - f(x)| \leq \frac{\varepsilon}{b-a} \]
    per \(n \geq N\).
    Allora otteniamo
    \begin{align*}
        \left|
            \integral[a][b][f(x)][x] - \integral[a][b][f_n(x)][x]
        \right| &\leq \integral[a][b][|f_n(x) - f(x)|][x] < \integral[a][b][\frac{\varepsilon}{b-a}][x] = \varepsilon
    \end{align*}
\end{snippetproof}

Questo risultato non vale per integrali impropri.

\begin{snippetexample}{Controesempio - Limite di funzioni integrabili non integrabile su dominio illimitato}
    Sia \(S = [1, +\infty)\) e consideriamo su tale dominio
    \[
        f_n(x) \triangleq \frac{1}{x} \chi_{[1,n]} = \begin{cases}
            \frac1x & x \in [1,n] \\
            0 & x \in (n, \infty)
        \end{cases}
    \]
    È facile mostrare che le \(f_n\) sono integrabili su \(S\), infatti fissato \(n\) abbiamo che
    \begin{align*}
        \integral[1][\infty][f_n(x)][x] &= \lim_{t \to \infty} \integral[1][t][\frac{1}{x}\chi_{[1,n]}][x] \\
        &= \lim_{t\to\infty} \integral[1][\min\{t,n\}][\frac1x][x] = \integral[1][n][\frac1x][x] = \ln|n|
    \end{align*}
    Mostriamo ora la convergenza uniforme a \(f(x) = 1/x\) in \([1, +\infty)\). Abbiamo
    \begin{align*}
        \sup_{x\in [1, +\infty)} |f_n(x) - f(x)| &= \sup_{x\in [1, +\infty)} \left|
            \frac{1}{x} \chi_{[1,n]}(x) - \frac1x
        \right| \\
        &= \sup_{x\in [1, +\infty)} \frac{1}{x} \chi_{[1, n]}(x) = \frac{1}{n} \to 0
    \end{align*}
    per \(n\to\infty\).
    Quindi \(f_n\) converge uniformemente a \(f(x)\), che tuttavia non è Riemann integrabile su \([1, +\infty)\).
\end{snippetexample}

\begin{snippetexample}{Controesempio - Il limite degli integrali è diverso dall'integrale del limite}
    Sia \(S=\realnumbers\) e su tale dominio definiamo
    \[
        f_n(x) \triangleq \frac{1}{n} \chi_{[n, 2n]}(x) = \begin{cases}
            \frac1n & x \in [n,2n] \\
            0 & x \notin [n, 2n]
        \end{cases}
    \]
    Mostriamo l'integrabilità fissato \(n\),
    \begin{align*}
        \integral[-\infty][+\infty][f_n(x)][x] &= \lim_{\gamma \to -\infty} \lim_{\mu \to +\infty}
        \integral[\gamma][\mu][\frac{1}{n}\chi_{[n,2n]}(x)][x] \\
        &= \lim_{\mu \to +\infty} \integral[n][\min\{\mu, 2n\}][\frac{1}{n}][x] \\
        &= \integral[n][2n][\frac1n][x] = 1
    \end{align*}
    Mostriamo che \(\{f_n\}\) converge uniformemente a \(f(x) = 0\) su \(\realnumbers\). Abbiamo
    \begin{align*}
        \sup_{x\in \realnumbers} |f_n(x) - f(x)| &= \sup_{x\in \realnumbers} \left|
            \frac{1}{n} \chi_{[n,2n]}(x)
        \right| \\
        &= \frac{1}{n} \to 0
    \end{align*}
    per \(n\to\infty\). Tuttavia,
    \[
        \integral[-\infty][\infty][f(x)][x] = 0 \neq 1 = \lim_n \integral[-\infty][+\infty][f_n(x)][x]
    \]
\end{snippetexample}

Guardiamo ora la convergenza uniforme e la derivazione.

\begin{snippettheorem}{}
    Sia \(S = [a,b]\) intervallo reale e \(\{f_n\}\) successione di funzioni \(f_n \colon [a,b] \fromto \realnumbers\) derivabili
    tale che \(\exists x_0 \in [a,b]\) tale che \(\{f_n(x_0)\}\) è convergente.
    Se la successione \(\{f_n'\}\) converge uniformementein \(S\), allora \(\exists f \colon S \fromto \realnumbers\) tale che
    \(f_n \to f\) uniformemente e \(f_n' \to f'\) uniformemente in \(S\).
    In particolare,
    \[
        \lim_n f_n'(x) = f'(x), \quad x \in [a,b]
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    Cominciamo con un caso facile, cioè \(f_n \in \mathcal C^1([a,b])\) per tutte le \(n\).
    In questo caso posso usare il teorema fondamentale del calcolo integrale,
    \[
        f_n(x) = f_n(x_0) + \integral[x_0][x][x][f_n'(y)][y]
    \]
    Per ipotesi \(f_n(x_0) \to c \in \realnumbers\) per \(n \to \infty\),
    \(f_n' \to h\) uniformemente con \(h \colon S \fromto \realnumbers\) continua perché le \(f_n'\) sono continue.
    Inoltre, per il teorema d'integrazione precedente
    \[
        \lim_n \integral[x_0][x][f_n'(y)][y] =\integral[x_0][x][h(y)][y]
    \]
    Da questo deduciamo che \(f_n \to f\) puntualmente con \[
        f(x) = c + \integral[x_0][x][h(y)][y]
    \]
    Per il teorema fondamentale del calcolo integrale \(f'(x) = h(x)\) per tutte le \(x\in[a,b]\).
    Ci manca da dimostrare che \(f_n \to f\) uniformemente in \([a,b]\). Abbiamo
    \begin{align*}
        \sup_{x\in[a,b]} |f_n(x) - f(x)| &= \sup_{x\in [a,b]} \left|f_n(x_0) + \integral[x_0][x][f_n'(y)][y] - c - \integral[x_0][x][h(y)][y]\right| \\
        &\leq |f_n(x_0) - c| + \sup_{x\in[a,b]} \integral[x_0][x][|f_n'(y) - h(y)|][y] \\
        &\leq |f_n(x_0) - c| + \sup_{x\in[a,b]} \integral[a][b][|f_n'(y) - h(y)|][y] \\
        &\leq |f_n(x_0) - c| + (b-a) \sup_{x\in [a,b]} |f_n'(x) - h(x)| \to 0
    \end{align*}
    per \(n\to\infty\). Il primo termine tende a zero per la convergenza a \(c\) e il secondo per la convergenza uniforme ad \(h\). \\
    Questa dimostrazione ci fa interpretare questo teorema come una sorta di corollario del precedente, infatti abbiamo chiesto che \(f_n' \to h\)
    uniformemente e poi abbiamo dimostrato che \(f_n\) converge uniformemente ad una primitiva di \(h\). \\
    Passiamo ora al caso dove \(f_n\) non è necessariamente \(\mathcal C^1([a,b])\).
    Cominciamo utilizzato il criterio di Cauchy per dimostrare che \(\{f_n\}\) converge uniformemente a qualcosa, cioè
    \(\forall \varepsilon > 0\), \(\exists N(\varepsilon)\) naturale tale che
    \[
        |f_n(x) - f_m(x)| < \varepsilon, \quad \forall n,m>N, \forall x \in [a,b]
    \]
    Per comodità definiamo \(g_{m,n} = f_n - f_m\).
    Dalle ipotesi deduciamo che:
    \begin{enumerate}
        \item \(\exists x_0 \in [a,b]\) tale che \(\exists N_0\left(\frac{\varepsilon}{2}, x_0\right)\) naturale tale che
        \[
            |g_{n,m}(x_0)| = |f_n(x_0) - f_m(x_0)| < \frac{\varepsilon}{2(b-a)}, \quad \forall n,m>N_0, \forall x \in [a,b]
        \]
        \item \(\{f_n'\}\) converge uniformemente quindi \(\exists N_1\left(\frac{\varepsilon}{2}\right)\) naturale
        tale che 
        \[
            |g_{n,m}'| = |f_n'(x) - f_n'(x)| < \varepsilon,  \forall n,m>N_1, \forall x \in [a,b]
        \]
    \end{enumerate}
    Fissiamo \(\overline N = \max\{N_0, N_1\}\). Applichiamo il teorema del valor medio sulla funzione
    \(g_{n_m}\) con \(n,m > N\), quindi \(\forall x \in [a,b]\), esiste \(t_0 \in \left(\min\{x,x_0\}, \max\{x,x_0\}\right)\) tale che
    \[
        |g_{n,m}(x) - g_{n,m}(x_0)| \leq |x-x_0| \cdot |g_{n,m}'(x_0)| < (b-a) \cdot \frac{\varepsilon}{2(b-a)} = \frac{\varepsilon}{2}
    \]
    Combinando tutto otteniamo che
    \begin{align*}
        |f_n(x) - f_m(x)| &= |g_{n,m}(x)| \leq |g_{n,m}(x) - g_{n,m}(x_0)| + |g_{n,m}(x_0)| \\
        &< \frac{\varepsilon}{2} + \frac{\varepsilon}{2} = \varepsilon, \quad \forall n,m \geq N, \forall x \in [a,b]
    \end{align*}
    Abbiamo mostrato che esiste \(f\) continua tale che \(f_n \to f\) uniformemente.
    Ora dobbiamo mostrare che \(f\) è derivabile e \(f_n' \to f'\) uniformemente.
    Per ipotesi sappiamo già che \(\{f_n'\}\) converge uniformente
    a qualcosa, quindi per unicità del limite basta dimostrare che \(f_n' \to f\) puntualmente in \([a,b]\).
    Fissato \(x\in [a,b]\) e \(n\) naturale definiamo
    \[
        \Delta_n (t) \triangleq \frac{f_n(x) - f_n(t)}{x-t}, \quad \forall t \in [a,b] \difference \{x\}
    \]
    Notiamo che
    \begin{itemize}
        \item \[ \lim_{t\to x} \Delta_n(t) = f_n'(x), \quad \forall n \in \naturalnumbers \]
        \item \[ \lim_n \Delta_n(t) = \Delta(t) = \frac{f(x) - f(t)}{x-t}, \quad t \in [a,b] \difference \{x\} \]
    \end{itemize}
    Per concludere ci basta dimostrare che
    \begin{align*}
        \lim_{t\to x} \lim_n \Delta_n(t) &\overset{(1)}{=} \lim_n \lim_{t \to x} \Delta_n(t) \\
        &= \lim_{t\to x} \Delta(t) = \lim_{t\to x} \frac{f(x) - f(t)}{x-t} \\
        &\overset{(2)}{=} \lim_{n} f_n'(x)
    \end{align*}
    Se dimostriamo che vale la \((1)\) otteniamo \((2)\) abbiamo finito.
    Se \(\{\Delta_n\}\) converge uniformemente in \([a,b]\difference \{x\}\), visto
    che \(x\) è un punto di accumulazione, per il teorema di cambio limite otteniamo \((1)\).
    Per il teorema del valor medio applicato a \(f_n - f_m\), \(\exists t_0 \in \left(\min\{x, t\}, \max\{x,t\}\right)\) tale che
    \begin{align*}
        |\Delta_n(t) - \Delta_m(t)| &= \frac{
            |f_n(x) - f_n(t) - f_m(x) + f_m(t)
        }{
            |t-x|
        } \\
        &\leq \frac{|t-x|}{|t-x|} \cdot |f_n'(t_0) - f_m'(t_0)| \\
        &= |f_n'(t_0) - f_m'(t_0)|, \quad \forall t \in [a,b]\difference \{x\}
    \end{align*}
    Di conseguenza 
    \[
        \sup_{t\in [a,b]\difference \{x\}} |\Delta_n(t) - \Delta_m(t)| \leq
        \sup_{t\in [a,b] \difference \{x\}} |f_n'(t) - f_m'(t)|
    \]
    Ma per ipotesi \(\{f_n'\}\) converge uniformemente quindi anche \(\{\Delta_n\}\) converge uniformemente in \([a,b] \difference \{x\}\).
\end{snippetproof}

\begin{snippetexample}{Successione delle derivate che converge uniformemente ma non la successione}
    Sia \(S = [0,1]\) e su tale dominio \(f_n(x) = x + n\).
    Ovviamente \(f_n' \equiv\) 1 per tutte le \(n\), e quindi \(f_n' \to f\)
    uniformemente con \(f(x) = 1\).
    Tuttavia, non vi è convergenza nemmeno puntuale per \(\{f_n\}\).
    Il teorema non funziona in quanto manca un singolo punto di convergenza.
    Il punto è fondamentale in quanto ci dà la convergenza in quel punto e il resto è dato
    dalle pendenza della derivata.
\end{snippetexample}

\begin{snippetexample}{Limite uniforme di funzioni derivabili può non essere derivabile}
    Sia \(S = [-a, a]\) con \(a > 0\) e su tale dominio definiamo \(f_n(x) = \sqrt{\frac1n + x^2}\).
    Abbiamo che \(f_n \to f\) puntualmente con \(f(x) = |x|\) in \(S\).
    TODO Esercizio: mostrare che la convergenza è pure uniforme.
    Notiamo subito che \(f\) non è derivabile in \(x=0\) e quindi non è derivabile in \(S\).
    Ovviamente le \(f_n\) sono differenziabili su \(S\) e \[
        f_n' = \frac{x}{\sqrt{\frac1n + x^2}}
    \]
    Notiamo che \(\{f_n'\}\) converge puntualmente a \[
        h(x) = \begin{cases}
            -1 & x < 0 \\
            0 & x = 0 \\
            +1 & x > 0
        \end{cases}
    \]
    su \(S\).
    Tuttavia, la convergenza su \(S\) non è unifrome.
    Il punto critico è chiaramente \(x=0\).
    Notiamo subito che
    \begin{align*}
        \left|f_n'\left(\frac1n\right) - h\left(\frac1n\right)\right|
        &= \left|
            \frac{\frac1n}{\sqrt{\frac1n + \frac{1}{n^2}}} - 1
        \right| \\
        &= \left|
            \frac{1}{\sqrt{n+1}} - 1
        \right| \to 1
    \end{align*}
    per \(n \to +\infty\).
    Quindi,
    \[
        \sup_{x \in [-1,1]} |f_n(x) - h(x)| \geq \left|f_n\left(\frac1n\right) - h\left(\frac1n\right)\right| \to 1
    \]
    per \(n \to \infty\).
    Quindi \(f_n\) non converge uniformemente ad \(h\) in \(S\).
    Se \(S = [\varepsilon, a]\) con \(\varepsilon > 0\) allora convergerebbe.
\end{snippetexample}

Il teorema di derivazione che abbiamo visto funziona nel caso
\(S = [a,b]\) con \(a<b\).
Vediamo quanto è importante questa ipotesi.

\begin{snippetexample}{}
    Sia \(S = \realnumbers\) e su tale dominio \(f_n(x) = x/n\).
    È facile dimostrare che converge puntualmente a \(f=0\) su \(S\),
    e inoltre
    \(f_n'(x) = \frac1n\).
    Se potessimo applicare il teorema nel caso di \(S\) non limitato,
    allora avremmo che \(f_n' \to f'\) uniformemente con \(f' = 0\).
    Tuttavia, questo è falso in quanto
    \begin{align*}
        \sup_{x\in\realnumbers} |f_n(x) - f(x) > |f_n(x) - f(n)| = 1
    \end{align*}
    per \(n \to \infty\).
    Tuttavia, su \(S = [a,b]\) converge uniformemente e infatti
    \[
        \sup_{x\in [a,b]} |f_n(x) - f(x)| = \frac{\max\{|a|,|b|\}}{n} \to 0
    \]
    per \(n \to \infty\).
\end{snippetexample}

Questo esempio ci suggerisce di di introdurre
una nozione di convergenza uniforme locale.

\begin{snippetdefinition}{Convergenza locale}
    Diciamo che \(\{f_n\}\)
    \emph{converge localmente uniformemente} a \(f \colon S \fromto \realnumbers\)
    in \(S\) se \(\forall B \subseteq S\) limitati, \(f_n \to f\) uniformemente in \(B\).
\end{snippetdefinition}

Riformuliamo il teorema parlando quindi di convergenza uniforme locale

\begin{snippettheorem}{}{}
    Sia \(\{f_n\}\) un successione di funzioni derivabili su \(S\)
    tali che \(\exists x_0 \in S\) tale che \(\{f_n(x_0)\}\) converge.
    Se \(\{f_n'\}\) converge localmente uniformemente in \(S\),
    allora \(\{f_n\}\) converge localmente uniformemente in \(S\) ad una funzione \(f\colon S \fromto \realnumbers\)
    e vale
    \[
        \lim f_n'(x) = f'(x), \quad \forall x \in S
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    Dimostrazione per esercizio (foglio di esercizi).
\end{snippetproof}

\section{Serie di funzioni}

\begin{snippetdefinition}{Serie puntualmente convergente}
    Diciamo che la serie \(\sum f_n\) \emph{converge puntualmente}
    se la successione delle somma parziali \(\{S_n\}\) con
    \[
        S_n(x) \triangleq \sum_{k=1}^n f_k
    \]
    converge puntualmente ad una funzione \(s \colon S \fromto \realnumbers\).
\end{snippetdefinition}

Useremo molto il criterio di Cauchy
applicato alle somme parziali, quindi otteniamo il risultato seguente:

\begin{snippetproposition}{Criterio di Cauchy puntuale}
    La serie \(\sum f_n\) converge puntualmente in \(S\)
    se e solo se \(\forall x \in S, \forall \varepsilon > 0, \exists N(x, \varepsilon) \in \naturalnumbers\)
    tale che
    \[
        |S_n(x) - S_m(x)| = \left|
            \sum_{k=1}^n f_k(x) - \sum_{k=1}^m f_k(x)
        \right| = \left|
            \sum_{k=n+1}^m f_k(x)
        \right| < \varepsilon, \quad \forall m > n \geq N
    \]
\end{snippetproposition}

A volte puô essere comodo scegliere \(m=n+h\) con \(h \in \naturalnumbers\).

Analogamente a quano detto per le successioni di funzioni,
la convergenza puntuale non si comporta bene rispetto alla derivabilità e integrabilità.

\begin{snippetexample}{}
    Sia \(S = \realnumbers\)
    e su tale dominio definiamo
    \[
        f_n(x) = \frac{x^2}{{(1 + x^2)}^n}
    \]
    Studiamo la serie \(f(x) = \sum f_n\).
    Notiamo che \(f(x) = 0\). Studiamo cosa succede in \(\realnumbers \difference \{0\}\).
    Abbiamo
    \begin{align*}
        \sum_{k=1}^n f_n(x) &= \sum_{k=1}^n \frac{x^2}{{(1 + x^2)}^n} \\
        &= x^2 \sum_{k=0}^n \left(
            \frac{1}{1 + x^2}
        \right)^k - x^2
    \end{align*}
    Abbiamo tolto un termine per renderla geometrica.
    Abbiamo quindi una serie gometrica di ragione \(1 / (1 + x^2)\).
    Allora per \(n \to \infty\) la serie tende a
    \[
        x^2 \left(
            \frac{1}{1- \frac{1}{1 + x^2}} - 1
        \right) = 1
    \]
    Concludiamo quindi che
    \[
        \sum f_n = \begin{cases}
            0 & x = 0 \\
            1 & x \neq 0
        \end{cases}
    \]
\end{snippetexample}

\begin{snippetdefinition}{Serie uniformemente convergente}
    Diciamo che la serie \(\sum f_n\) \emph{converge uniformemente}
    in \(S\) se \(\forall \varepsilon > 0, \exists N(\varepsilon) \in \naturalnumbers\)
    tale che
    \[
        |S_m(x) - S_n(x)| = \left|
            \sum_{k=1}^n f_k(x) - S(x)
        \right| < \varepsilon, \quad \forall n > N
    \]
\end{snippetdefinition}

Alternativamente
\[
    \sup_{x \in S} |S_n(x) - S(x)| = \sup_{x\in S} \left|
        \sum_{k=1}^n f_k(x) - S(x)
    \right| < \varepsilon, \quad \forall n > N
\]

Anche qua possiamo applicare il criterio di Cauchy.

\begin{snippetproposition}{Criterio di Cauchy per series}
    La serie \(\sum f_n\) converge uniformemente in \(S\)
    se e solo se \(\forall \varepsilon > 0\), \(\exists n \in \naturalnumbers\) tale che
    \begin{align*}
        |S_n(x) - S_m(x)| &= \left|
            \sum_{k=1}^n f_k(x) - \sum_{k=1}^m f_k(x) -
        \right| \\
        &= \left|
            \sum_{k=n+1}^m f_k(x)
        \right| < \varepsilon, \quad \forall m > n \geq N, \forall x \in S
    \end{align*}
\end{snippetproposition}

Analogamente
\[
    \sup_{x \in S} \left|
        \sum_{k=n+1}^m f_k(x)
    \right| < \varepsilon
\]

\begin{snippetdefinition}{Convergenza totale}
    Diciamo che la serie \(\sum f_n\) \emph{converge totalmente}
    in \(S\) se la serie numerica
    \[
        \sum_{n=1}^\infty ||f_n||_\infty
    \]
    converge.
\end{snippetdefinition}

È equivalente a dire che esiste una successione numerica convergente che maggior termine-per-termine.

\begin{snippettheorem}{Teorema di Weierstrass}
    Sia \(\{f_n\}\) successione di fuzioni tale che \(\exists \{M_n\}\)
    successione di numeri positivi tali che
    \[
        ||f_n||_\infty < M_n, \quad \forall n \in \naturalnumbers
    \]
    e \[
        \sum_{m=1}^\infty M_n < \infty
    \]
    Allora \(\sum f_n\) converge uniformemente.
    In particolare, monvergenza totale di serie implica convergenza uniforme di serie.
\end{snippettheorem}

\begin{snippetproof}{Teorema di Weierstrass}
    Sfruttiamo il criterio di Cauchy.
    Dato \(\varepsilon > 0\) cerchiamo \(N(\varepsilon) \in \naturalnumbers\)
    che verifica il criterio.
    Siccome \(\sum M_n < \infty\), per il criterio di Cauchy
    per serie numeriche esiste \(N(\varepsilon)\) naturale tale che
    \begin{align*}
        \left|
            \sum_{k=n+1}^m M_k
        \right|
        = \sum_{k=n+1}^m M_k < \varepsilon, \quad \forall m > n \geq N
    \end{align*}
    Allora
    \begin{align*}
        \left|
            \sum_{k=n+1}^m f_k(x)
        \right| &\leq \sum_{k=n+1}^m |f_k(x)| \leq \sum_{k=n+1}^m ||f_k(x)||_\infty
        \leq \sum_{k=1}^m M_k < \infty
    \end{align*}
\end{snippetproof}

\begin{snippetexample}{Convergenza uniforme non implica convergenza totale}
    Consideriamo \(S = [1, +\infty)\)
    e su tale dominio
    \[
        f_n(x) = \frac1n \chi_{[n, n+1)}(x)
    \]
    Le somme parziali sono date da
    \[
        S_n(x) = \sum_{k=1}^n f_k(x) = \begin{cases}
            \frac1k & x \in [k, k+1) \land k \in \{1,\cdots, n\} \\
            0 & x \notin [1, n+1)
        \end{cases}
    \]
    Solo un termine della serie è diverso da zero.
    In questo caso \(k\) è il floor quindi
    \[
        f(x) = \frac{1}{\lfloor x \rfloor}, \quad x \in [1, \infty)
    \]
    Usiamo il criterio di Cauchy e quindi fissiamo \(\varepsilon > 0\),
    e per comodità prendo \(m=n+h\) con \(h\) naturale
    \begin{align*}
        \left| \sum_{k=n}^{n+1} f_k(x) \right|
        &= \sum_{k=n}^{n+1} f_k(x) = \sum_{k=n}^{n+1} \frac1n \chi_{[n, n+1)}(x) \\
        &\leq \frac{1}{n} \to 0
    \end{align*}
    per \(n \to \infty\) per ogni \(x \in S\).
    Abbiamo dimostrato che converge uniformemente.
    Vediamo subito che non converge totalmente, infatti
    \begin{align*}
        \sum_{k=m}^{n+h} ||f_k||_\infty = \sum_{k=m}^{n+h} \frac1n \chi_{[n, n+1)}(x)
        < \sum_{k=m}^{n+h} \frac{1}{k}
    \end{align*}
    che non converge.
\end{snippetexample}

\begin{snippetproposition}{}
    Se una serie converge uniformemente allora il termine generale tende uniformemente a zero.
\end{snippetproposition}

\begin{snippetexercise}{}
    Consideriamo \(f_i(x) = 1/i\). Questa funzione tende uniformemente a \(0\),
    ma la sua serie non converge uniformemente.
\end{snippetexercise}

\begin{snippetexercise}{}
    La serie \(\frac{1}{n^x}\) converge se e solo se \(x>1\), cioè sul dominio \(D = (1, \infty)\).
    Su \(D\) ha convergenza puntuale
    \[
        f_n(x) = \frac{1}{n^x} \leq \frac{1}{n} \to 0
    \]
    per \(n \to \infty\).
    Il termine generale tende uniformemente a \(0\).
    Supponiamo per assurdo che la serie converga uniformemente. Allora vale il teorema del doppio limite
    \[
        \lim_{x\to 1}
        \left(
            \lim_n \sum_{i=1}^n f_i(x)
        \right)
        \lim_n \left(
            \lim_{x\to 1}\lim_n \sum_{i=1}^n f_i(x)
        \right)
    \]
    However, \(l_n = \sum^n f_i(1) = \sum \frac1i\) does not converge, which is absurd \lightning.
\end{snippetexercise}

\begin{snippetproposition}{}
    Siano \(f_i\) continue su \([a,b)\) e sia \(\sum f_i(x)\) converge in \((a,b)\),
    ma \(\sum f_i(a)\) non convergente.
    Allora la serie \(\sum f_i\) non converge uniformemente in \((a,b)\).
\end{snippetproposition}

\begin{snippetproposition}{}
    Siano \(f_i \geq 0\), \(f_i \to 0\) in \(S\).
    Se \(\{f_i\}\) tende uniformemente a zero, allora
    \[
        \sum_{i=1}^\infty {(-1)}^i f_i(x)
    \]
    converge uniformemente in \(S\).
\end{snippetproposition}

\begin{snippetproof}{}
    Infatti
    \[
        \left|
            \sum_{i=1}^\infty {(-1)}^i f_i(x)-
            \sum_{i=1}^n {(-1)}^i f_i(x)
        \right| \leq f_{n+1}(x)
    \]
    è evidente che se \(f_n \to 0\) per \(n\to\infty\) la serie converge uniformemente.
\end{snippetproof}

\begin{snippetexample}{Convergenza uniforme non implica convergenza assoluta}
    \[
        \sum {(-1)}^n \frac1n
    \]
\end{snippetexample}

\begin{snippetexample}{}
    Sia \(S = \realnumbers^+\) e su tale dominio \(f_n(x) = \frac1n\chi_{[n-1, n)}(x)\). Allora la serie
    \[
        \sum_{n=1}^\infty f_n(x)
    \]
    è una funzione a scala. La convergenza è uniforme ed è assoluta.
    La convergenza non è tuttavia totale, infatti
    \[
        \sum_{n=1}^\infty \sup_{x \in \realnumbers^+} |f_n(x)| = \sum_{i=1}^\infty \frac1n
    \]
    che diverge. Infatti,
    \[
        \sup_{x \in \realnumbers^+} |f_n(x)| = \frac1n
    \]
\end{snippetexample}

\begin{snippetproposition}{}
    Siano \(f_i \colon S \fromto \realnumbers\). Se \(\sum f_i\) converge totalmente, allora converge uniformemente in \(S\).
\end{snippetproposition}

\begin{snippetproof}{}
    Per ipotesi \(|f_i(x)| \leq M_i\) per \(\sum M_i\) convergente.
    Mostriamo che questo implica la condizione di Cauchy uniforme.
    \[
        |f_{\tilde n}(x) + f_{{\tilde n} + 1}(x) + \cdots + f_{{\tilde n} + \varphi}| \leq
        \sum_{i=f_{\tilde n}}^{{\tilde n} + \varphi} |f_i(x)| \leq 
        \sum_{i=f_{\tilde n}}^{{\tilde n} + \varphi} M_i < \varepsilon
    \]
    pur di prendere \(\tilde n\) arbitrariamente grande, perché
    \(\sum M_i\) converge, allora soddisfa le condizioni di Cauchy.
    Dat oche la stima vale per tutte le \(x\in S\), abbiamo trovato che \(\sum f_i\) soddisfa la condizione di Cauchy
    uniforme e quindi converge uniformemente.
\end{snippetproof}

\begin{snippetexercise}{}
    Consideriamo \[
        f(x) = \sum_{n=1}^\infty \frac{
            \sqrt[3]{\sin(x)}
        }{
            n^2 + x^2
        }
    \]
    Dimostrare che \(f\) è continua a stabilire in quali punti è differenziabile.
    Stabilire se \(f\) è integrabile in \(\realnumbers\). 
\end{snippetexercise}

\begin{snippetsolution}{}
    Non possiamo dedurre la convergenza uniforme di \(f\) dalla convergenza uniforme della serie
    di \(\frac{1}{n^2 + x^2}\). Tuttavia, possiamo dedurrne la continuità dalla continuità della serie di \(\frac{1}{n^2 + x^2}\).
    Siccome \[
        \frac{1}{n^2 + x^2} \leq \frac{1}{n^2}
    \]
    allora serie converge totalmente, e quindi converge uniformemente, allora \(f\) è continua.
    Stabiliamo ora se
    \[
        \sum \frac{1}{n^2 + x^2}
    \]
    è differenziabile.
    Calcoliamo la serie delle derivata, ossia
    \[
        - \sum_{n=1}^\infty \frac{2x}{(n^2 + x^2)^2}
    \]
    Verifichiamo che converga localmente uniformemente: sia \(x\in(a,b)\), allora
    \[
        \left|
            \frac{2x}{(n^2 + x^2)^2}
        \right| \leq \frac{
            2\max\{|a|, |b|\}
        }{n^4}, \quad \forall x, \forall n
    \]
    Per il teorema di derivazione per serie poniamo derivate termine a termine
    \[
        \left(
            \sum_{n=1}^\infty \frac{1}{n^2 + x^2}
        \right)' = - \sum_{n=1}^\infty \frac{2x}{(n^2 + x^2)^2}, \quad \forall x \in \realnumbers
    \]
    Sicuramente \(f\) è derivabile per \(x\neq k\pi\) con \(k\in\integernumbers\).
    Se \(f = gh\) con \(g(x) \neq 0\) e tale che non esiste \(h'(\overline x)\) con
    \(h'(\overline x) = 0\), allora non esiste \((gh)'(\overline x)\).
    Infatti, se esistesse allora siccome \(f = gh\) avremmo che \(h = f/g\),
    ma allora sarebbe derivabile in \(\overline x\) perché quoziente di funzioni derivabili è derivabile.
    Siccome la serie di \(\frac{1}{n^2 + x^2}\) non si annulla mai, \(f\) non è derivabile
    per \(x = k\pi\) con \(k\) intero.
    Siccome \(f\) è dispari, se è integrabile sulla retta reale, allora l'integrale vale zero.
    Basta studiare \(f\) su \((0, \infty)\).
    La nostra funzione è il prodotto di una funzione periodica e media nulla per una funzione per una funzione che tende
    a zero momentaneamente, quindi \(f\) è integrabile.
\end{snippetsolution}

\section{Serie di potenze}

È una serie di funzioni della forma \(f_n(x) = a_n(x-x_n)^n\).
Di sicuro la serie converge almeno nel centro della serie di potenze.
A meno di traslazioni possiamo sempre considerare la forma
\[
    \sum_{n=0}^\infty a_n x^n
\]
In generale è più comodo considerare la serie di potenze complessa.
Se \(S\) è l'insieme di convergenza della serie di potenza,
si dimostra che \(S^\circ\) è un cerchio, anche degenere cioè vuoto o tutto il piano complesso.
L'insieme di convergenza può avere dei pezzi di frontiera.
Il fatto che \(S^\circ\)
sia un cerchio dipende dalla seguente proprietà:

\begin{snippetproposition}{}
    Se la serie di potenze \(\sum a_nx^n\) converge in qualche punto \(\tilde x\)
    e se \(x\) è un punto tale che \(|x| < |\tilde x|\),
    allora la serie converge assolutamente in \(x\).
\end{snippetproposition}

\begin{snippetproof}{}
    Per ipotesi la serie è convergente nel punto dato, quindi
    \(\{a_n \tilde x^n\}\) è limitata.
    Supponiamo che \(\tilde x \neq 0\), in cui caso il teorema è banale.
    Prendiamo un punto \(x\) complesso tale che \(|x| < |\tilde x|\).
    Abbiamo quindi
    \begin{align*}
        a_n x^n &= a_n x^n \cdot \frac{
            x^n
        }{
            \tilde x^n
        } \\
        |a_n x^n| &= \underbrace{|a_n \tilde x^n|}_{\text{bounded}} \cdot \left|\frac{
            x
        }{
            \tilde x
        }\right|^n
    \end{align*}
    Il primo termine è limitato, mentre il secondo è il termine generale di una serie geometrica di ragione
    \(<1\), quindi la serie converge assolutamente.
    Siccome la serie converge assolutamente in un qualche punto \(\overline x\),
    allora la serie converge totalmente in tutto il cerchio chiuso che ha \(\overline x\)
    sul bordo, per maggiorazione, e quindi convergenza uniforme in tutto il cerchio chiuso.
\end{snippetproof}

Quindi nell'interno del cerchio ogni serie di potenza converge \emph{localmente} uniformemente.

\begin{snippettheorem}{Criterio di Cauchy-Hadamard}
    Sia \(\sum a_n x^n\) una serie di potenza.
    Il raggio di convergenza è dato da
    \[
        R = \left(
            \limsup_n \sqrt[n]{|a_n|}
        \right)^{-1}
    \]
    con la convenzione che \(1/0 = \infty\) (la serie converge su \(\complexnumbers\))
    e \(1/\infty = 0\) (la serie converge nel centro).
\end{snippettheorem}

\begin{snippetproof}{Criterio di Cauchy-Hadamard}
    Sia \(l = \limsup_n \sqrt[n]{|a_n|}\).
    Abbimo 3 casi: \(l = 0\), \(l = +\infty\) e \(0 < l < +\infty\).
    Vogliamo mostrare che se \(|x| < 1/l\),
    la serie converge e che se \(|x| > 1/l\) la serie diverge.
    Sia \(|x| < 1/l\).
    Dalla definizione di \(l\) abbiamo che \(\forall \varepsilon > 0\),
    \[
        |a_n|^{\frac1n} < l+\varepsilon
    \]
    definitivamente cioè \(|a_n| < (k+\varepsilon)^n\).
    Ma \(|x| < 1/l\) implica che \(|x|l < 1\).
    Moltiplicando \(l\) con qualcosa di leggermente più grande, il prodotto resta
    \(< 1\). Cioè esiste \(\tilde \varepsilon > 0\)
    tale che \(|x|(l + \tilde \varepsilon) < 1\).
    Quindi \[
        |a_nx^n| < (|x|(l + \tilde \varepsilon))^n
    \]
    definitivamente. Abbiamo maggiorato il termine generale della serie di potenza
    con il termine generale di una serie geometrica convergente, e quindi la serie converge in \(x\).
    Mostrimao ora ch se \(|x| > 1/l\), la serie diverge.
    Dalla definizione di \(l\) sappiamo che \(\forall \varepsilon > 0\)
    esistono infiniti termini della successione tale che
    \[
        \sqrt[n]{|a_n|} > l-\varepsilon
    \]
    Cioè, esiste una sottosuccessione che tende ad \(l\).
    Abbiamo quindi
    \[
        |a_nx^n| \geq (|x|(l - \varepsilon))^n
    \]
    per infiniti valori di \(n\).
    Siccome \(|x|>1/l\), allora \(|x|l>1\), cioè \(\exists \tilde \varepsilon > 0\)
    tale che \(|x|(l-\tilde \varepsilon) > 1\).
    La serie ha quindi infiniti termini che sono maggiori di \(1\), quindi diverge.
\end{snippetproof}

\begin{snippetexample}{}
    Calcola il raddio di convergenza della serie di potenza
    \[
        \sum_{i=0}^\infty n^3 x^n
    \]
    Il raggio è \(R = 1\).
    Studiamo se vi è convergenza sul bordo di raggio \(R = 1\).
    In tal caso, il termine generale \(|n^3 x^n| = |n^3|\) che non tende a zero,
    quindi diverge.
    Quindi la serie converge solo sul cerchio aperto di raggio \(1\).
\end{snippetexample}

\begin{snippetexample}{}
    Calcola il raggio di convergenza della serie di potenza
    \[
        \sum_{i=0}^\infty \frac{x^n}{n^3}
    \]
    Il raggio è \(R = 1\).
    Sul bordo, per \(x=1\) la serie converge assolutamente,
    allora ce l'ho anche in tutto il cerchio chiuso.
    Quindi la serie converge su tutto il cerchio chiuso di raggio \(R=1\).
\end{snippetexample}

\begin{snippettheorem}{Criterio del rapporto}
    Sia \(\sum a_nx^n\)
    una serie di potenza tale che \(a_n \neq 0\) definitamente.
    Se \[
        \lim\left|
            \frac{a_{n+1}}{a_n}
        \right|
    \]
    esiste e converge a \(l\), allora il raggio di convergnza è \(R = 1/l\).
\end{snippettheorem}

\begin{snippetproof}{Per esercizio}
    Dimostrazione simile al criterio di Cauchy-Hadamard e
    il criterio del rapporto per le serie numeriche.
\end{snippetproof}

\begin{snippetexample}{}
    Se \(\sum a_n x^n\) dove \(a_n = 1\) se \(n = 2k+1\)
    e \(a_n = e^{-n}\) se \(n = 2k\).
    La classe limite è \(\{1, 0\}\).
    Per il criterio di Cauchy-Hadamard,
    la radice \(n\)-esima dei possibili valori sono \(1\) e \(e^{-1}\),
    quindi il limite superiore è \(1\).
    Allora, il raggio di convergenza è \(R=1\).
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo la serie \(\sum x^{2n}\).
    Non possiamo utilizzare il criterio del rapporto
    in quanto ci sono definitivamente termini nulli.
    Tuttavia,
    ponendo \(y = x^2\) vediamo che questa è la serie geometria e quindi il raggio di
    convergenza è \(R=1\).
\end{snippetexample}

\begin{snippetexample}{}
    Calcola il raggio di convergenza della serie di potenza
    \[
        \sum_{i=0}^\infty \frac{x^n}{n!}
    \]
    Il raggio è \(R=\infty\) per il criterio del rapporto.
    Questa convergenza non è uniforme.
\end{snippetexample}

\begin{snippetproposition}{}
    Da scrivere leggendo la dimostrazione
\end{snippetproposition}

\begin{snippetproof}{}
    Se il raggio di convergenza è \(\infty\),
    l'unico caso in cui la convergenza è uniforme su \(\complexnumbers\)
    è quello in cui i coefficienti sono definitivamente nullo, cioè
    la serie è di un polinomio.
    Se la serie convergesse uniformemente sul piano complesso,
    allora il termine dovrebbe convergere uniformemente a zero
    \[
        \sup_{x \in \complexnumbers} |a_nx^n| < \varepsilon
    \]
    definitivamente.
    Il termine \(a_n\) è indipendente e quindi può uscire fuori dal supremum,
    ma il supremum di \(|x^n|\) è infinito, quindi serve \(a_n = 0\) definitivamente.
\end{snippetproof}

\begin{snippetdefinition}{Funziona intera}
    Se \(\sum a_nx^n\) converge su tutto il piano complesso, essa si dice
    \emph{funzione intera}.
\end{snippetdefinition}

\begin{snippetexample}{}
    I polinomi, \(e^x\), \(\sin(x)\) sono funzioni intere.
\end{snippetexample}

Se \(\sum a_nx^n\) è intera, allora converge localmente totalmente e quindi
converge localmente uniformemente, cioè converge uniformemente su ogni compatto.

\begin{snippetexample}{}
    Calcola il raggio di convergenza della serie di potenza
    \[
        \sum_{i=0}^\infty n!x^n
    \]
    La serie ha raggio di convergenza \(R=0\).
\end{snippetexample}

\begin{snippetexample}{}
    Calcola il raggio di convergenza della serie di potenza
    \[
        \sum_{i=0}^\infty \left(
            \frac{x}{27}
        \right)^n
    \]
    La serie ha raggio di convergenza \(R=27\).
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo la serie
    \[
        \sum \frac{x^n}{x^2}
    \]
    ha raggio di convergenza \(R=1\),
    converge sul disco chiuso perché converge assolutamente, come per esempio in \(x=1\).
\end{snippetexample}

\begin{snippettheorem}{Teorema di Abel}
    Se la serie di potenza \(\sum a_nx^n\) converge
    in qualche punto \(\tilde x\) sul bordo del raggio di convergenza,
    ma non converge assolutamente in \(\tilde x\),
    allora \(\sum a_nx^n\) converge uniformemente
    sul raggio che collega \(\tilde x\) al centro.
    Inoltre, la convergenza uniforme la si può estendere ad ogni punto
    del diametro escluso il punto opposto a \(\tilde x\).
\end{snippettheorem}

Mi estende la convergenza uniforme fino al punto nel bordo dato.

\begin{snippetexample}{}
    Consideriamo la serie
    \[
        \sum \frac{x^n}{n}
    \]
    che ha raggio di convergenza \(R=1\).
    Sul bordo, in \(x=1\) la serie non converge mentre \(x=-1\) converge.
    Siccome è una serie di Dirichlet, converge in tuti i punti del borso tranne \(x=1\).
    La convergenza è uniforme in quanto totale in tutti i punti del cerchio discosti dal bordo.
    Vorrei sapere se la convergenza è uniforme
    anche sui punti del bordo. Possiamo usare il teorema di Abel.
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo la serie reale
    \[
        \sum {(-1)}^{n+1} \frac{x^n}{n}
    \]
    Il raggio di convergena è \(R=1\).
    Si dimostra che la serie converge a \(\log(1 + x)\)
    per \(x \in (-1, 1)\).
    La serie non converge assolutamente in nessun punto del bordo,
    cioè nè per \(x=1\) e \(x=-1\).
    Il teorema di Abel ci consente di dire che la somma della serie è \(\log 2\),
    cioè anche sul bordo.
    Questo è dato dal teorema del doppio limite,
    \[
        \lim_{x\to 1} \left(
            \underbrace{\sum_{n=1}^\infty {(-1)}^{n+1} \frac{x^n}{n}}_{\log(1+x)}
        \right) = \sum_{n=1}^\infty \frac{{(-1)}^{n+1}}{n}
    \]
    che possiamo fare siccome c'è convergenza uniforme, fino ad \(1\)
    grazie al teorema di Abel.
    Quindi la serie converge uniformemente in \(\realnumbers\)
    su tutti i sottoinsiemi del tipo \((\alpha, 1]\)
    con \(\alpha > -1\).
\end{snippetexample}

\begin{snippetexample}{A volte il lim sup è più facile da calcolare dal limite}
    Consideriamo la serie di potenze
    \[
        \sum_{n=0}^\infty (\sin n)^n x^n
    \]
    Usiamo il criterio di Cauchy-Hadamard. Abbiamo
    il limite
    \[
        \lim_n \sqrt[n]{
            |\sin n|^n
        } = \lim_n |\sin n|
    \]
    non esiste, mentre il lim sup
    \[
        \limsup_n \sqrt[n]{
            |\sin n|^n
        } = \limsup_n |\sin n| = 1
    \]
    in quanto la successione \(\{\sin n\}\) è densa in \([-1, 1]\).
\end{snippetexample}

\begin{snippetexample}{A volte il lim sup è più facile da calcolare dal limite}
    Consideriamo la serie di potenze
    \[
        \sum_{n=0}^\infty \sin n x^n
    \]
    Usiamo il criterio di Cauchy-Hadamard. Abbiamo
    \[
        \limsup \sqrt[n]{
            |\sin n|
        } = 1
    \]
    Mentre calcolare il limite non è banale.
\end{snippetexample}

\begin{snippetproposition}{}
    Consideriamo una serie di potenze reale
    \(\sum a_n x^n\) con raggio di convergenza \(0 < R \leq \infty\)
    e consideriamo \(x \in (-R, R)\).
    Sappiamo che converge in almeno un punto, e la serie delle derivate è
    \[
        \sum_{n=1}^\infty na_nx^{n-1}
    \]
    che è ancora una serie di potenze, con un suo insieme di convergenza.
    Il raggio di convergenza viene mantenuto.
    La convergenza può perdere dei punti del bordo, ma il raggio di convergenza è il medesimo.
\end{snippetproposition}

\begin{snippetproof}{}
    Consideriamo la serie delle derivate
    \[
        g(x) = \sum_{n=1}^\infty na_nx^{n-1}
    \]
    Consideriamo la funzione \(xg(x)\) per avere il termine \(na_nx^n\),
    tanto la convergenza rimane la medesima. Applicando il teorema di Cauchy-Hadamard
    vediamo che
    \begin{align*}
        \limsup_n \sqrt[n]{
            n|a_n|
        } = \limsup_n \sqrt[n]{|a_n|}
    \end{align*}
    e quindi il raggio di convergenza rimane il medesimo.
\end{snippetproof}

Ogni serie di potenze con raggio di convergenza positivo è una funzione derivabile,
che continua a convergere almeno sul cerchio aperto.
Il medesimo processo può essere iterato per derivate superiori.

\begin{snippetcorollary}{}
    Sia \[
        f(x) = \sum_{n=0}^\infty a_nx^n
    \]
    con raggio di convergenza \(R > 0\), allora \(f\in \mathcal C^\infty((-R, R))\) e \[
        f^{(k)}(x) = \sum_{n=k}^\infty n(n-1)\cdots(n-k+1)a_nx^{n-k}
    \]
\end{snippetcorollary}

\begin{snippetexample}{}
    Consideriamo la serie di potenze
    \[
        \sum_{n=1}^\infty \frac{x^n}{n}
    \]
    che ha raggio di convergenza \(R=1\).
    Essa converge in \([-1, 1]\). La serie delle derivate è la serie geometrica
    \[
        \sum_{n=1}^\infty x^n
    \]
    che ha sempre raggio di convergenza \(R=1\)
    ma converge solo su \((-1, 1)\).
\end{snippetexample}

\begin{snippetcorollary}{}
    Consideriamo la serie di potenze
    \[
        \sum_{n=1}^\infty \frac{x^n}{n^2}
    \]
    ha raggio di convergenza \(R=1\).
    Nel punto \(x=1\) converge assolutamente, e quindi converge totalmente in tutto il cerchio
    chiuso \([-1, 1]\). La serie delle derivate è
    \[
        \sum_{n=1}^\infty x^{n-1}{n}
    \]
    che converge in \([-1, 1)\). La derivata seconda
    \[
        \sum_{n=2}^\infty \frac{n-1}{n}x^{n-2}
    \]
    che converge solamente in \((-1, 1)\), e così via.
\end{snippetcorollary}

Abbiamo la formula \[
    f^{(k)}(x) = \sum_{n=k}^\infty n(n-1)\cdots(n-k+1)a_nx^{n-k}
\]
In \(x=0\) otteniamo \(f^{(k)}(0) = k! a_k\), cioè \(a_k = \frac{1}{k!}f^{(k)}(0)\).
Quindi conoscendo le derivate della serie di potenze in un intorno arbitrariamente piccolo,
conosciamo interamente la serie di funzioni su tutto il suo cerchio di convergenza.
È quindi impossibili avere fenomeni di biforcazione.
Non tutte le funzioni \(\mathcal C^\infty\) sono quindi serie di potenze.

\begin{snippetexample}{}
    La funzione
    \[
        f(x) = \begin{cases}
            e^{-\frac{1}{x^2}} & x \neq 0 \\
            0 & x = 0
        \end{cases}
    \]
    non è analitica, in quanto tutte le derivate nell'origine sono nulle
    ma non posso ricostruire interamente la funzione.
    La sua serie di Maclaurin converge alla funzione identicamente nulla.
\end{snippetexample}

In generale posso sempre associare ad una funzione \(f(x)\) una serie di potenze,
cioè la serie di Taylor.

\begin{snippetproposition}{}
    Una successionoe reale \(\{b_n\}\)
    si dimostra che \(\exists f \in \mathcal C^\infty((a,b))\)
    tale che se \(x_0 \in (a,b)\), allora \(f^{(k)} = b_n\).
\end{snippetproposition}

Quindi per esempio scegliendo \(b_n = \{(n!)^2\}\), la funzione \(f\)
che ha \(\{b_n\}\) come coefficienti ha serie di Taylor che converge solo in \(x_0\),
quindi \(R=0\).

\begin{snippetdefinition}{Funzioni analitiche}
    Le funzioni \(f\in\mathcal C^\infty((a,b))\)
    che almeno localmente \(\forall x_0 \in (a,b)\)
    coincide sono somma della loro serie di Taylor costituiscono una sottoclasse di
    \(\mathcal C^\infty((a,b))\) detta \emph{classe delle funzioni analitiche}
    in \((a,b)\). Tale classe viene indicata come \(\mathcal C^\omega((a,b))\).
\end{snippetdefinition}

In una funzione analitica la convergenza locale implica quella globale.
Sono la generalizzazione più immediata dei polinomi, in quanto
pure loro sono una classe di tali funzioni.

Nota: per ogni punto bisogna trovare almeno un intorno dove converge, non è detto
che la serie di Taylor in \(x_0\) in quel punto converge su tutto l'intervallo.

\begin{snippetexample}{}
    Consideriamo la funzione \(f(x) = \sum_{n=0}^\infty x^n\)
    che possiamo scrivere come serie geometrica, che converge nell'intervallo \((-1, 1)\).
    Tuttavia, la funzione \(f\) è analitica in tutto l'intervallo
    \(f\in \mathcal C^\omega((-\infty, 1))\).
    Il raggio \(R=1\) è il massimo raggio possibile per la serie centrata nell'origine,
    in quanto se oltrepassassi l'asintoto.
    Se fosse centrata in \(x=\frac13\), il raggio può essere al massimo \(R=\frac23\)
    e infatti si dimostra che lo è.
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo la funzione \(f(x) = \frac{1}{1 + x^2} = \sum_{n=0}^\infty {(-1)} x^{2n}\)
    che possiamo scrivere come serie di potenza con raggio di convergenza \(R=1\).
    Tuttavia, la funzione originale è definita su tutto \(\realnumbers\).
    Se consideriamo la funzione sul piano complesso, abbiamo due pali a \(\pm i\),
    e quindi il raggio di convergenza non può superare quei punti.
    Se invece la centrassimo in \(x=2\), il raggio di convergenza sarebbe \(\sqrt{5}\)
    cioè la distanza fra \(2\) ed \(i\).
\end{snippetexample}

\begin{snippettheorem}{}
    Sia \(f \in \mathcal C^\infty((a,b))\).
    Se \(\exists M, L \in \realnumbers\)
    tale che \(\forall x \in (a,b)\) e \(\forall n\) vale
    \(|f^{(k)}(x)| < M \cdot L^n\), allora
    \[
        f(x) = \sum_{n=0}^\infty \frac{f^{(n)}(x)}{n!}(x-x_0)^n
    \]
    per tutte le \(x_0, x \in (a,b)\).
\end{snippettheorem}

\begin{snippetproof}{}
    Sia \(x_0 \in (a,b)\) fissato come centro della serie e scriviamo
    il polinomio di Taylor di ordine \(k\) con il resto di Lagrange.
    Esiste quindi \(c(x, x_0, k)\) tra \(x\) e \(x_0\) tale che
    \[
        f(x) = \sum_{n=0}^k \frac{f^{(k)}(x_0)}{n!}(x-x_0)^n
        + \frac{f^{(k+1)}(c)}{(k+1)!}(x-x_0)^{k+1}
    \]
    vogliamo mostrare che il resto in forma di Lagrange tende a zero. Stimiamone quindi il modulo
    \begin{align*}
        \left|
            \frac{f^{(k+1)}(c)}{(k+1)!}(x-x_0)^{k+1}
        \right| \leq \frac{M \cdot (L |x-x_0|)^{k+1}}{(k+1)!} \to 0
    \end{align*}
    per \(n \to \infty\).
\end{snippetproof}

Questa condizione è solo sufficiente e implica la convergenza in tutto l'intervallo \((a,b)\),
e può essere usata localmente.

\begin{snippetexample}{}
    Consideriamo la funzione \(f(x) = 1 / (1-x)\)
    che è analitica su \((-\infty, 1)\).
    Non esistono \(M\) ed \(L\) tale che
    la condizione del teorema % TODO CITE
    sia soddisatta su tutto \((-\infty, 1)\).
\end{snippetexample}

\begin{snippetproposition}{}
    Sia \(f \in \mathcal C^\infty\) tale che la sa successione delle derivate sia equilimitate,
    allora è analitica.
\end{snippetproposition}

\begin{snippetproof}{}
    Se le derivate sono equilimitate con costante \(K\), basta scegliere \(L=\) e \(M=K\)

Quindi \(\sin(x)\) e \(\cos(x)\) sono analitiche in quanto le derivate sono equilimitate.
La funzione \(e^x\) non è limitata ma possiamo applicare il criterio negli intervalli
\((-\infty, a)\), e quindi è una funzione intera.

La serie di Taylor ci permette di estendere la funzione esponenziale ai complessi.

Le serie di potenza possono essere integrate termine a termine.
\[
    \integral[0][x][
        \sum_{n=0}^\infty a_nt^n
    ][t]
    = \sum_{n=0}^\infty \frac{a_nx^{n+1}}{n+1}
\]

\begin{snippetexample}{}
    Integrando la serie geometrica, che ha raggio di convergenza \(R=1\), otteniamo
    \[
        \ln(1-x) = - \sum_{n=0}^\infty \frac{x^{n+1}}{n+1}=
        - \sum_{n=1}^\infty \frac{x^n}{n}
    \]
    Ne segue che \[
        \ln(1 + x) = \sum_{n=1}^\infty {(-1)}^{n} \frac{x^n}{n}
    \]
    che è quindi analitica
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo \(f(x) = 1/(1 + x^2)\)
    che possiamo scrivere come \(\sum_{n=0}^\infty {(-1)}^n x^{2n}\), allora integrando otteniamo
    \[
        \arctan(x) = \sum_{n=0}^\infty {(-1)}^{2n + 1} \frac{x^{2n+1}}{2n+1}
    \]
    che è quindi analitica.
\end{snippetexample}

\begin{snippetexample}{}
    Mostriamo che \(f(x) = (1 + x)^\alpha\) con \(\alpha \in \realnumbers\).
    La sua serie di Taylor è data da
    \[
        g(x) = \sum_{n=0}^\infty \binom{\alpha}{n}x^n, \quad x\in(-1,1)
    \]
    Ci aspettiamo che il raggio di convergenza sia
    \(R=1\). Usiamo il criterio del rapporto
    \[
        \left|
            \frac{\alpha(\alpha-1)\cdots(\alpha - n + 1)}{n!}
        \right| \cdot \frac{(n+1!)}{|\alpha(\alpha-1)\cdots(\alpha-n)|}
        = \frac{n+1}{|\alpha- n|} \to 1
    \]
    Quindi il raggio è \(R=1\).
    Notiamo la relazione \((1 + x)g'(x) = \alpha g(x)\).
    Inoltre, sappiamo che \(g(0) = 1\).
    Otteniamo quindi il problema di Cauchy
    \[
        \begin{cases}
            g'(x) = \frac{\alpha}{1 + x}g(x) \\
            g(0) = 1
        \end{cases}
    \]
    La soluzione nulla non rispetta il punto fissato, altrimenti l'equazione è separabile e integrando
    \[
        \frac{g'(x)}{g(x)} = \frac{\alpha}{1 + x}
    \]
    otteniamo
    \[
        \ln|g(x)| = \alpha \ln |1 + x| =  \alpha \ln (1 + x)
    \]
    siccome \(x > -1\). Possiamo pure togliere il modulo di \(g(x)\)
    siccome la nostra soluzione non può mai diventare negativa, in quanto non può
    intersecare la soluzione costante dell'equazione differenziabile. Abbiamo quindi dimostrato che
    \[
        g(x) = {(1 + x)}^\alpha
    \]
    ed è una funzione analitica.
\end{snippetexample}

Sappiamo già dal teorema di Stone-Weierstrass che tutte le funzioni continue
possono essere approssimate da polinomi.
La differenza è che per migliore una approssimazione basta aggiungere termini
al polinomi di Taylor, mentre per il teorema di Stone-Weierstrass
dobbiamo rifare tutto da zero.

\begin{snippetexercise}{}
    Studiare la convergenza della serie
    \[
        S(x) = \sum_{n=0}^\infty 2^n \sin\left(\frac{x}{3^n}\right)
    \]
    e calcolare
    \[
        \lim_{x \to 0} \frac{S(x)}{x}
    \]
\end{snippetexercise}

\begin{snippetsolution}{}
    Possiamo cominciare stimando
    \[
        |f_n(x)| \leq 2^n\frac{|x|}{3^n} = \left(\frac23\right)^n |x| \to 0
    \]
    quindi la serie converge puntualmente.
    Notiamo che su un insieme limitato, \(|x| < R\),
    \(|f_n(x)| \leq R \left(\frac23\right)^n\)
    e converge totalmente, e la convergenza è localmente uniforme.
    La convergenza è uniforme dappertutto?
    Per il limite notiamo che
    \[
        \lim_{x \to x_0} \frac{S(x) - S(0)}{x-0} = S'(0)
    \]
    quindi dobbiamo studiare la derivabilità.
    La convergenza è uniforme sui reali se 
    \[
        \sup_{x\in \realnumbers} \left|
            \sum_{k=n}^\infty f_n(x)
        \right| \to 0
    \]
    per \(n \to \infty\).
    Proviamo a prendere un punto di massimo delle \(f_n\),
    ossia \(\hat x_n = 3^n\frac{n}{2}\) che è il primo massimo di \(f_n\),
    e quindi \(f(\hat x_n) = 2^n\).
    Per \(k > n\) possiamo dire che
    \(f_k\) è dilatata, quindi il primo massimo si trova più a destra del primo massimo con
    \(f_n\). Quindi, in \([0, \hat x_n]\) la \(f_k\) è crescente,
    e siccome parte da 0 è quindi positiva.
    Abbiamo quindi che la coda della serie calcolata nel punto \(\hat x_n\)
    \[
        \sup_{x\in \realnumbers} \left|
            \sum_{k=n}^\infty f_n(x)
        \right| \leq \sum_{k=n}^\infty f_k(\hat x_n) \geq f(\hat x_n) = 2^n \to \infty
    \]
    Quindi il sup non tende a zero e non abbiamo convergenza uniforme.
    Per il limite notiamo che
    \[
        \lim_{x \to x_0} \frac{S(x) - S(0)}{x-0} = S'(0)
    \]
    quindi dobbiamo studiare la derivabilità.
    Nell'origine tutti gli addendi sono \(0\),
    quindi abbiamo la convergenza uniforme.
    Ci serve la convergenza uniforme locale, in quanto la derivata è una operazione locale.
    Abbiamo
    \[
        f_n'(x) = \frac{2^n}{3^n} \cos \frac{x}{3^n}
    \]
    Dobbiamo qunidi verificare se
    \(\sum_{n=0}^\infty f_n'\) converge. Ma chiaramente converge totalmente e quindi uniformemente in quanto
    \[
        \sum_{n=0}^\infty |f_n'| \leq \sum_{n=0}^\infty \left(\frac{2}{3}\right)^n
    \]
    Abbiamo quindi
    \[
        S'(0) = \sum_{n=0}^\infty f_n'(0) = \sum_{n=0}^\infty \left(\frac{2}{3}\right)^n = 3
    \]
\end{snippetsolution}

\section{Equazioni differenziali}

\begin{snippetdefinition}{Curva integrale}
    Una curva integrale è una curva nel piano che in ogni punto è
    dotata di retta tangente e che sia localmente
    il grafico di qualche soluzione
    di una equazione differenziale.
\end{snippetdefinition}

Siamo ammettendo anche il caso in cui la tangente sia verticale in qualche punto.

\begin{snippetexample}{}
    La funzione \(y = \sqrt[3]{x}\)
    è soluzione dell'equazione \(y' = 3x^{-2/3}\).
    Abbiamo una pendenza verticale in \(x=0\), e l'equazione amette
    due soluzioni separate per le due parti separate.
    Tuttavia, l'intera funzione è una curva integrale in quanto ammette una tangente,
    anche verticale, ed è localmente il grafico di una funzione.
\end{snippetexample}

In generale, se una famiglia di soluzioni
di un'equazione differenziale scalare del primo ordine ammette
un inviluppo, allora anch'esso è soluzione.

\begin{snippetexample}{}
    \begin{wrapfigure}{r}{6cm}
        \begin{center}
            \begin{tikzpicture}[scale=0.75]
                \begin{axis}[
                    axis lines = middle,

                    ylabel = \(y\),

                    samples=100,
                    restrict y to domain=-10:15,

                \end{axis}
            \end{tikzpicture}
        \end{center}

    Consideriamo \(y_k = (x-k)^3\) che sono delle cubiche traslate orizzontalmente,
    con inviluppo la retta \(y=0\).
    Questa famiglia è soluzione, per esempio, dell'equazione
    \[
        y' = 3y^{\frac23}
    \]
    di cui sembrerebbe l'integrale generale, mentre \(y=0\) l'integrale singolare.
    Ma ciò è sbagliato sbagliato in quanto ci sono altre soluzioni,
    cioè le funzioni che sono costanti a sinistra e poi si coniugano in una delle cubiche,
    o nell'altra direzione, o che percorrono un pezzo di cubica, poi diventano costanti e poi seguono una cubica successiva.
    Cioè tutte le possibili saldature.
    L'unicità dove due soluzioni si intersecano
    con la medesima tangente. Se invece considero una soluzione sull'asse delle \(x\),
    non ho unicità nemmeno localmente, mentre ce l'abbiamo sulle soluzioni cubiche.
    \wrapfill
\end{snippetexample}

\begin{snippetdefinition}{Forma differenziali esatte}
    È un'equazione differenziale della forma
    \[
        y' = \frac{a(x,y)}{b(x,y)}
    \]
    tali che \(a(x,y)\,\text{d}x - b(x,y)\,\text{d}y\) è una forma differenziale esatta.
\end{snippetdefinition}

Sia \(V(x,y)\) un potenziale della forma differenziale. Ogni funzione definita implicitamente da
\(V(x,y) = 0\) è soluzione dell'equazione differenziale.
Le equazioni separabili sono un caso particolare delle equazioni esatte.

\pagebreak

\begin{snippetexample}{}
    \begin{wrapfigure}{r}{6cm}

    Consideriamo l'equazione differenziale \[
        y' = \sin(y^2)
    \]
    Le soluzioni costante sono \(y = \pm \sqrt{k\pi}\) per \(k \in \integernumbers\).
    Separando l'equazione otteniamo \(y' / \sin(y^2) = 1\), e quindi
    \[
        x = \integral[1][y][\frac{1}{\sin(t^2)}][t]
    \]
    Non riusciamo a ricavare \(y\) in funzione di \(x\),
    ma abbiamo la funzione inversa, e quindi possiamo ribaldare il grafico di quest'ultima.
    Consideriamo il punto iniziale \(y(0) = 1\).
    La funzione passante per questo punto non può intersecare le due soluzioni costanti
    per il teorema di unicità locale (in quanto \(f(x,y) = \sin(y^2)\) che è in \(\mathcal C^\infty(\realnumbers^2)\)).
    La sua derivata è positiva, e quindi crescente e allora
    ha un asintoto un asintoto orizzontale.
    Questo asintoto può essere \(y = \sqrt{\pi}\)
    oppure uno più piccolo.
    Sia \(y=L\) tale asintoto, quindi \(1 < L \leq \sqrt{\pi}\).
    Per trovare tale quota facciamo quindi il limite. Per \(x \to \infty\) abbiamo
    \begin{align*}
        \integral[1][y(x)][\frac{1}{\sin(t^2)}][t] \to L
    \end{align*}
    L'unico modo purché questo integrale sia improprio
    è quando \(t\) tende a \(\sqrt{\pi}\).
    In tal caso per \(t \to \sqrt{\pi}\), \(\sin(t^2) = - \sin(t^2 - \pi)\),
    quindi è asintotico a \(-(t^2 - \pi) = -(t-\sqrt{\pi})(t+\sqrt{\pi})\),
    che a sua volta è asintotico a \(-2\sqrt{\pi}(t-\sqrt{\pi})\).
    Quindi non è integrabile in quanto
    \[
        \frac{1}{\sin(t^2)}
        \sim
        \frac{1}{
            2\sqrt{\pi}(t-\sqrt\pi)
        }
    \]
    Per cui l'asintoto è \(\sqrt{\pi}\).
    Se così non fosse stato, la curva avrebbe raggiunto l'asintoto in tempo finito,
    e quindi avremmo avuto una saldatura, ma non siamo in regime di biforcazioni.
    Invece, per \(x \to - \infty\), abbiamo sempre un asintoto orizzontale
    di quota \(0 \leq l < 1\). Non può essere un numero maggiore di zero,
    in quanto l'integrale non sarebbe improprio e sarebbe finito.
    Quindi, per forza vale \(l=0\).
    Infatti,
    \[
        \integral[0][1][\frac{1}{\sin(t^2)}][t]
    \]
    diverge.
    Quindi, nella prima striscia sono crescenti, mentre nella seconda
    sono decrescenti alternandosi.
    Visto che abbiamo due asintoti abbiamo
    un fleso. Per trovarlo basta derivare
    \[
        \frac{d}{dx} \frac{1}{\sin(y^2)} = - \frac{2y \cos(y^2)}{\sin^2(y^2)}
    \]
    che si annulla quando \(y^2 = \frac{\pi}{2}\),
    e quindi il flesso si ha per \(y = \frac{\sqrt{\pi}}{2}\).
    \wrapfill
\end{snippetexample}

Consideriamo \(y^{(k)} = f(x,y,y', \cdots, y^{(k-1)})\)
con \(f \in \mathcal C^(\Omega, \realnumbers^n)\)
e \[
    \Omega \subseteq \realnumbers \cartesianprod
    \underbrace{\realnumbers^n \cartesianprod \cdots \cartesianprod \realnumbers^n}_{n \text{ times}}
\]
Per fissare un'unica soluzione è necessario imporre molteplici condizioni.
Possiamo fissare \(k\) punti di \(\Omega\), cioè
\[
    \begin{cases}
        y^{(k)} = f(x,y,y', \cdots, y^{(k-1)}) \\
        y(x_1) = y_1 \\
        y(x_2) = y_2 \\
        \cdots \\
        y(x_n) = y_n \\
    \end{cases}
\]
che viene detto problema ai limiti.
Alternativamente possiamo imporre il problema di Cauchy di ordine \(k\)
\[
    \begin{cases}
        y^{(k)} = f(x,y,y', \cdots, y^{(k-1)}) \\
        y(x_0) = y_1 \\
        y'(x_0) = y_2 \\
        \cdots \\
        y^{(n-1)}(x_0) = y_{n-1} \\
    \end{cases}
\]
Oppure ancora un problema misto fra i due.
Se vogliamo studiare il comportamento locale di una soluzione,
l'unico sensato è quello di Cauchy.
Gli altri due possono essere studiati solo dal punto di vista globale.

Dato un problema
\[
    \begin{cases}
        y' = f(x,y) \\
        y(x_0) = y_0
    \end{cases}
\]
con \(f \in \mathcal C(\Omega)\) con \(\Omega \subseteq \realnumbers^{n+1}\) aperto e
\((x_0, y_0) \in \Omega\).
L'intervallo più grande su cui la soluzione può essere definita è la proiezione di
\(\Omega\) sull'asse delle \(x\). Una soluzione può terminare prima se sbatte
prima sul bordo di \(\Omega\).

Un problema è ben posto se:
\begin{enumerate}
    \item la soluzione esiste;
    \item la soluzione è unica;
    \item dipendenza continua del dato iniziale, cioè cambiando di poco
    il dato iniziale la soluzione cambia di poco.
    Serve una dipendenza continua della soluzione anche da \(f\).
    Se salta anche solo una delle tre richieste, diremo che il problema è mal posto.
\end{enumerate}

\begin{snippetdefinition}{Problema di Cauchy di ordine \(k\) scalare}
    Problema di Cauchy di ordine \(k\) scalare
    \[
        \begin{cases}
            y^{(k)} = f(t,y,y',\cdots, y^{(k-1)}) \\
            y(t_0) = y_0 \\
            y'(t_0) = y_1 \\
            \vdots \\
            y^{(k-1)}(t_0) = y_{(k-1)}
        \end{cases}
    \]
\end{snippetdefinition}

Un problema di Cauchy di ordine \(k\) è equivalente ad un problema di Cauchy del primo ordine vettoriale.
Possiamo infatti porre un problema di Cauchy vettoriale analogo ponendo
\[
    x = \begin{pmatrix}
        x_1 = y \\
        x_2 = y' \\
        \vdots \\
        x_k = y^{(k-1)}
    \end{pmatrix}
\]
Derivando otteniamo
\[
    \begin{pmatrix}
        x_1' = x_2 \\
        x_2' = x_3 \\
        \vdots \\
        x_{k-1'} = x_k
        x_k' = f(t,x_1, x_2, \cdots, x_k)
    \end{pmatrix}
\]
quindi abbiamo il sistema del primo ordine in forma vettoriale
\[
    x' = F(t,x)
\]
Quindi i due problemi sono equivalenti.
Se \(\varphi \in \mathcal C^k(I)\) soddisfa le equazioni di ordine \(k\),
tramite \(\varphi\) (e le sue derivate) riesco a
costruire un vettore \(\Phi = (\varphi_1, \cdots, \varphi_k)\) con \(\varphi_k = \varphi_{k-1}'\) tale che
\(\varphi \in \mathcal C'(I, \realnumbers^k)\) e soddisfa l'equazione data.
Viceversa, se il vettore \(\Phi\) soddisfa le equazioni,
la sua prima componente \(\varphi_1\), è automaticamente una funzione di classe \(\mathcal C^k(I)\) che soddisfa l'equazione
di ordine \(k\).

Il medesimo procedimento vale anche per equazioni vettoriali di ordine \(k\),
basta aumentare la dimensione.
Per questo motivo si sviluppa la teoria delle equazioni differenziali considerando solo
problemi di Cauchy vettoriali al primo ordine.

Consideriamo \(y' = f(x,y)\) con \(|f(x,y)| \leq M\),
cioè \(|y'| \leq M\).
Per il teorema dell'incremento finito, il grafico di \(y\) deve essere contenuto in un cono
centrato nel punto iniziale \((x_0, y_0)\) e di semiampiezza \(\arctan M\).
La condizione di limitatezza implica che se la soluzione esiste allora soddisfa \[
    |y(x) - y(x_0)| \leq M|x-x_0|
\]
Allora il grafico di \(y(x)\) sta dentro il cono \(|y-y_0| \leq M|x-x_0|\).
Supponiamo che l'ipotesi di limitatezza sia soddisfatta su tutti i \((a,b) \cartesianprod \realnumbers\).
In questo caso la soluzione è globale: esiste su tutto \(I=(a,b)\).
Non può terminare prima perché dovrebbe avere un asintoto verticale,
ma questo violerebbe la proprietà del cono.

Supponiamo invece che l'ipotesi sia soddisfatta su \(I \cartesianprod J\) con \(I, J\) intervalli.
Se l'ampiezza del cono \(;\) è troppo grande, il cono interseca il rettangolo sui lati orizzontali.
In questo caso non possiamo più garantire che la soluzione sia definita su tutto l'intervallo \(I\).

\begin{snippettheorem}{di esistenza e unicità locale di Cauchy-Lipschitz}
    Sia \(f\colon \Omega \fromto \realnumbers^n\) con \(\Omega \subseteq \realnumbers^{n+1}\) aperto
    tale che \(f \in \mathcal C(\Omega, \realnumbers^n)\)
    e consideriamo il problema di Cauchy
    \[
        \begin{cases}
            y' = f(x,y) \\
            y(x_0) = y_0
        \end{cases}
    \]
    Supponiamo che \(\exists I \subseteq \realnumbers\) intervallo chiuso e limitato con \(x_0 \in I^\circ\) e
    \(\exists J \subseteq \realnumbers^n\) con \(J = \overline{B_r(y_0)}\) per qualche \(r>0\) tale che
    \(I \cartesianprod J \subseteq \Omega\). Supponiamo che \(f\) sia lipschitziano
    in \(J\) uniformemente rispetto ad \(x \in I\), cioè supponiamo che \(\exists L > 0\) tale che
    \[
        ||f(x,y_1) - f(x, y_2)|| \leq L||y_1-y_2||, \quad \forall y_1, y_2 \in J, \forall x \in I
    \]
    Allora \(\exists \tilde I \subseteq I\) intervallo con \(x_0 \in \tilde I^\circ\) su cui il problema di Cauchy
    ha una e una sola soluzione. In particolare, se \(I=[a,b]\), con \(a < x_0 < b\), \(\tilde I = [\alpha, \beta]\),
    con \(\alpha < x_0 < \beta\) dove
    \[
        \beta-x_0 = \min\left\{b-x_0, \frac{r}{M}\right\}, \quad
        x_0-\alpha = \min\left\{x_0-a, \frac{r}{M}\right\}, \quad
        M = \max_{(x,y) \in I \cartesianprod J} ||f(x,y)||
    \]
\end{snippettheorem}

L'ipotesi di lipschitzianità non è sufficiente a garantire la continuità
di \(f\), perché si tratta di una lipschitzianità solo nella \(y\).

Per esempio \(f(x,y) = \chi_{\rationalnumbers}(x)\) è discontinua in ogni punto, ma è lipschitziana rispetto ad \(y\) con \(L=0\).
Infatti, \(f\) è costante sulle rette verticali.

Il teorema di Peano assume solo la continuità, non la lipschitzianità, e garantisce solo l'esistenza di una soluzione, ma non l'unicità.

\begin{snippettheorem}{di esistenza e unicità globale}
    Sia \(f\colon \Omega \fromto \realnumbers^n\) con \(\Omega \subseteq \realnumbers^{n+1}\) aperto
    tale che \(f \in \mathcal C(\Omega, \realnumbers^n)\)
    e consideriamo il problema di Cauchy
    \[
        \begin{cases}
            y' = f(x,y) \\
            y(x_0) = y_0
        \end{cases}
    \]
    Supponiamo che \(\exists I \subseteq \realnumbers\) intervallo
    chiuso e limitato con \(x_0 \in I^\circ\) tale che \(I \cartesianprod \realnumbers^n \subseteq \Omega\) e che \(f\)
    sia lipschitziana rispetto a \(y\) uniformemente a \(x\in I\). Allora esiste una unica soluzione al problema in \(I\).
\end{snippettheorem}

La dimostrazione sfrutta il fatto che \(I\) sia compatto, ma la tesi è vera anche per intervalli non copatti.
Infatit, ogni intervallo è unione di intervalli compatti.
Su ognuno di questi la soluzione esiste ed è unica.
Se \([a_1, b_1] \subseteq [a_2, b_2]\), la soluzione su \([a_2, b_2]\) deve essere un prolungamento
di quella definita su \([a_1, b_1]\), altrimenti avremmo due soluzioni.
In questo modo si può definire la soluzione in tutti i punti
dell'intervallo anche se questo non è compatto.

\begin{snippettheorem}{di esistenza e unicità globale}
    Sia \(f \in \mathcal C(\Omega; \realnumbers^n)\) con \(\Omega \subseteq \realnumbers^{n+1}\) aperto.
    Se \(\exists I \subseteq \realnumbers\) intervallo tale che
    \(I \cartesianprod \realnumbers^n \subseteq \Omega\), se \(f\) è lipschitziana rispetto a \(y \in \realnumbers^n\) uniformemente
    rispetto a \(x\in I\), allora \(\forall (x_0, y_0) \in I \cartesianprod \realnumbers^n\),
    esiste un'unica soluzione al problema di Cauchy
    \[
                \begin{cases}
            y' = f(x,y) \\
            y(x_0) = y_0
        \end{cases}
    \]
    su \(I\).
\end{snippettheorem}

Notiamo che \(f \in \mathcal C(\Omega; \realnumbers^n)\) non è sufficiente a garantire le ipotesi del teorema in grande.

\begin{snippetexample}{}
    Consideriamo \(y' = xy^3 = f(x,y)\), quindi \(f \in \mathcal C^1(\realnumbers^2)\),
    ma \(f\) non è lipschitziana rispetto alla \(y\).
    Abbiamo \(y=0\) che è la soluzione costante. Se \(y \neq 0\),
    possiamo separare le variabili
    \begin{align*}
        \integral[y_0][y][
            \frac{1}{t^3}
        ][t]
        &= \frac12(x^2 - x_0^2) \\
        - \frac{1}{y^2} + \frac{1}{y_0^2} &= x^2 - x_0^2 \\
        y &= \operatorname{sign}(y_0) \frac{1}{\sqrt{
            \frac{1}{\sqrt{y_0^2}} + x_0^2 - x^2
        }}
    \end{align*}
    Qui \(I = \realnumbers\) però le soluzioni hanno degli asintoti verticali e sono definite solo nell'intervallo
    \((-\infty, \alpha)\) dove 
    \[
        \alpha = \sqrt{
            \frac{1}{y_0^2} + x_0^2
        }
    \]
    Queste succede perché \(y^3\) cresce troppo velocemente.
\end{snippetexample}

\begin{snippetproof}{Teorema di esistenza e unicità locale}
    L'ipotesi di lipschitzianità di \(f\) è soddisfatta
    necessariamente solo sul cilindro \(I \cartesianprod J\)
    dove \(J = \overline B_r(y_0)\)
    e \(I = [x_0, x_0 + a]\).
    Nel caso del teorema in grande, l'operatore di Volterra operava su \(\mathcal C^0(I, \realnumbers^n)\).
    Invece, adesso, dobbiamo considerare \(\mathcal C^0(I, \overline B_r(y_0))\).
    Questo nuovo spazio non è tuttavia uno spazio vettoriale,
    quindi non è di Banach.
    Inoltre, per poter applicare il teorema delle contrazioni ci serve un endo operatore:
    se usiamo l'operatore di Volterra su una funzione continua nella bolla, il risultato potrebbe fuoriuscire
    dalla bolla.
    Per fortuna ci basta che lo spazio sia uno spazio metrico completo, non per forza di Banach.
    Studiamo quindi il nostro spazio \(\mathcal C^0(I, \overline B_r(y_0))\).
    Fra i suoi elementi abbiamo la funzione
    \(\phi \equiv y_0\) costante.
    Dalla definizione di \(\mathcal C^0(I, \overline B_r(y_0))\),
    notiamo che relativamente alla norma uniforme, i suoi elementi distano
    al più \(r\) dalla funzione \(\phi\).
    Quindi, il nostro spazio è precisamente una bolla centrata su \(\phi\)
    \[
        \mathcal C^0(I, \overline B_r(y_0))
        = \overline B_r^{\mathcal C^0(I, \realnumbers^n)}(\phi)
    \]
    Questa bolla è chiusa in \(\mathcal C^0(I, \realnumbers^n)\),
    e un chiuso in uno spazio compatto, quindi è uno spazio metrico completo.
    Abbiamo quindi risolto il primo problema.
    Per quanto concerne l'operatore di Volterra,
    dobbiamo verificare se porta questo spazio metrico completto in sè stesso.
    Se \(||f(x_0, y|| \leq M)\), possiamo garantire solo che il grafico della soluzione
    sta dentro il cono di 
    ampiezza \(\arctan M\).
    Se il cono esce dal cilindro lungo i bordi orizzontali, dobbiamo restringere l'intervallo \(I\).
    Applicheremo il teorema
    a un intervallo \(\tilde I \subseteq I\) da determinare, tale che l'operatore di volterra porti
    \(\mathcal C^0(\tilde I, \overline B_r(y_0))\) in sè stesso.
    Sia \(\varphi \in \mathcal C^0(\tilde I, \overline B_r(y_0))\), e applichiamo l'operatore di Volterra
    \[
        T(\varphi)(t) = y_0 + \integral[x_0][t][f(u, \varphi(u))][u]
    \]
    Imponiamo che questa funzione assuma valori in \(\overline B_r(y_0)\).
    Calcoliamone quindi la norma e imponiamo che stia nella bolla
    \begin{align*}
        ||(T(\varphi))(t) - y_0|| &= \left|\left|\,
            \integral[x_0][t][f(u, \varphi(u))][u]
        \right|\right| \leq \integral[x_0][t][||f(u, \varphi(u))||][u]
    \end{align*}
    Sia allora \[
        M = \max_{I \cartesianprod J} ||f(u, \varphi(u))||
    \]
    Allora abbiamo che
    \begin{align*}
        ||(T(\varphi))(t) - y_0|| &= M\integral[x_0][t][][u]
        \leq M(t-x_0) \leq r \\
        t-x_0 &\leq \frac rM
    \end{align*}
    Quindi basta scegliere
    \[
        \tilde I = \left[x_0, x_0 + \frac rM\right]
    \]
    La soluzione potrebbe esistere oltre questo intervallo, ma noi ci restringiamo a qui.
    Con questa scelta, l'operatore di \(\tilde I\)
    \[
        T \colon \mathcal C^0(\tilde I, \overline B_r(y_0)) \fromto \mathcal C^0(\tilde I, \overline B_r(y_0))
    \]
    è un endo operatore e quindi per il teorema delle contrazioni ammette un unico punto fisso.
\end{snippetproof}

Il teorema in grande si basa sul teorema delle contrazioni.
Per quest'ultimo, in particolare, dice che se \(x_0 \in X\)
è un punto arbitrario, allora l'orbita di \(x_0\), \(T^n(x_0)\),
converge all'unico punto fisso.
In particolare, se prendiamo una qualunque \(\varphi \colon I \fromto \realnumbers^n\) continua,
la sua orbita \(T^n(\varphi)\) converge all'unica soluzione del problma di Cauchy,
e la convergenza è uniforme.

\begin{snippetexample}{Operatore di Volterra}
    Consideriamo per esempio
    \[
        \begin{cases}
            y' = y \\
            y(0) = 1
        \end{cases}
    \]
    la cui unica soluzione è \(\varphi(x) = e^x\).
    Abbiamo \(f(x,y) = y\), e
    prendendo \(\varphi_1 \equiv 1\) possiamo iterare l'operatore di Volterra
 *       \begin{align*}
            \varphi_2 &= T\varphi_1 = 1 + \integral[0][x][][t] = 1 + x \\
            \varphi_3 &= T\varphi_2 = 1 + \integral[0][x][1 + t][t] = 1 + x + \frac{x^2}{2}
        \end{align*}
    E in generale questa successione tende allo sviluppo di \(e^x\).
    Se avessimo scelto una \(\varphi_0\) sarebbe comunque uscita una successione convergente ad \(e^x\),
    ma non la serie di Maclaurin.
\end{snippetexample}

Spesso applichiamo il teorema in grande su intervalli non compatti, tipicamente aperti.
Il caso classico è \(y' = a(x) + b(x) = f(x,y)\) con \(a,b \in \mathcal C(I)\)
e di solito \(I\) aperto.
Per applicare il teorema in grande su \(I \cartesianprod \realnumbers^n\),
dobbiamo applicare agli intervalli compatti contenuti in \(I\).

Per esempio per \(y' = \frac yx + \frac 1{x-3}\)
abbiamo tre intervalli. Se il dato iniziale è \(y(\pi) = 27\), devo considerare l'intervallo
\(I = (3, +\infty)\), che possiamo scomporre in intervalli compatti della forma \(I_{n \geq n_0} = \left[3 + \frac 1n, \pi + n\right]\),
che sono annidati. Su tutti questi
la soluzione esiste su tutti questi e coincidono e quindi la estendiamo a quello più grande.

Se consideriamo una funzione \(\varphi \colon I \fromto \realnumbers\) e la successione delle iterate
di \(\varphi\), questa converge uniformemente alla soluzione solo localmente (non su tutto \(I\)).

In generale la condizione di lipschitzianità su tutta la striscia nel teorema in grande,
è molto forte. In molti casi in cui questa non è soddisfatta, le cose vanno bene comunque.

\begin{snippetexample}{}
    Consideriamo
    \(y' = f(x,y)\) con \(f\) continua e limitata, \(\Omega = \realnumbers^2\) ed
    \(f \in \mathcal C^1(\realnumbers^2)\).
    Vige esistenza e unicità locale perché \(f\)
    è \(\mathcal C^1\) e quindi è localmente lipschitziana.
    Sicuramente non abbiamo fenomeni di biforcazione.
    Tuttavia, in un caso del genere la soluzione non può che non essere definita su tutto \(\realnumbers\),
    cioè se avesse un asintoto verticale dovrebbe uscire dal cono, ma non è possibile (e inoltre una soluzione non può sparire nel nulla).
    Tuttavia, ci manca l'ipotesi di lipschitzianità su tutto \(\realnumbers^2\),
    e quindi non possiamo applicare il teorema di unicità ed esistenza globale.
\end{snippetexample}

\begin{snippettheorem}{}
    Sia \(f \colon \Omega \fromto \realnumbers\) continua,
    \((x_0, y_0) \in \Omega \supseteq I \cartesianprod \realnumbers\) e consideriamo il probelma di Cauchy
    \[
        \begin{cases}
            y' = f(x,y) \\
            y(x_0) = y_0
        \end{cases}
    \]
    tale che \(\exists a,b \in \mathcal C(I, \realnumbers)\)
    per cui \(|f(x,y)| \leq a(x)|y| + b(x)\)
    per \((x,y) \in I \cartesianprod \realnumbers\) (cioè è sublineare).
    Allora, esiste almeno una soluzione del problema di Cauchy in \(I\).
\end{snippettheorem}

In \(\realnumbers^n\) l'ipotesi diventa
\(\exists a,b \in \mathcal(I, \realnumbers^n)\)
tale che
\[
    ||f(x,y)|| \leq a(x)||y|| + b(x)
\]

Questa ipotesi è una sorta di lipschitzianità in grande, anche se non implica quest'ultima.
Ci dice che quando \(y\) è grande, la \(f\) non può crescere troppo.

Questa versione del teorema non parla di unicità.
Se vogliamo anche l'unicità, possiamo aggiungere la condizione
\(f \in \mathcal C^1(\Omega)\).
Quindi questa condizione più la sublinearità implica esistenza e unicità in grande.

\begin{snippetexample}{}
    Consideriamo \(y' = \sin(y^2) = f(x,y)\).
    Chiaramente \(f\) non è di Lipschitz rispetto a \(y\).
    Infatti, \((\sin(y^2))' = 2y\cos(y^2)\).
    Tuttavia, \(|f| \leq 1\), e quindi possiamo applicare la sublinearità
    con \(a(x) = 0\) e \(b(x) = 1\).
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo l'equazione del pennello di Peano
    \(y' = \sqrt{|y|}\).
    Chiaramente \(\sqrt{|y|}\) è uniformemente continua, ma non è lipschitziana
    in un intorno di \(y=0\).
    Togloiendo un intorno di \(y=0\), abbiamo esistenza e unicità locale.
    Tuttavia, \(\sqrt{|y|} \leq |y| + 1\), quindi è sublineare e abbiamo esistenza in grande.
\end{snippetexample}

\begin{snippetexercise}{Studio qualitativo equazione differenziale}
    \begin{wrapfigure}{r}{6cm}
        \begin{center}

                \node[below left] at (0,0) {$O$};

    Consideriamo \(y' = x\sqrt{|\log y|}\) e studiamo qualitativamente l'andamento delle curve
    integrali, con particolare attenzione all'origine.
    Chiaramente consideriamo solo il primo e secondo quadrante.
    L'unica soluzione costante è \(y=1\).
    L'asse delle \(y\) è il luogo dei punti stazionari delle soluzioni non costanti.
    Notiamo che \(f(-x, y) = -f(x,y)\), quindi le linee integrali
    sono simmetriche rispetto all'asse delle \(y\).
    Chiaramente la funzione non è \(\mathcal C^1\), quindi non possiamo applicare il teorema
    di unicità globale. Potrebbe succedere che abbiamo delel saldature lungo la soluzione costante.
    Consideriamo il dato iniziale  \((x_0, 1)\), oltre alla soluzione costante abbiamo la soluzione
    \begin{align*}
        \integral[1][y][\frac{1}{\sqrt{|\log t|}}][t] = \frac{x^2}{2} - \frac{x_0^2}{2}
    \end{align*}
    Possiamo anche trovare la soluzione esplicita (versione positiva ma c'è anche quella negativa)
    \[
        x = \sqrt{
            x_0^2 + 2\integral[1][y][\frac{1}{\sqrt{|\log t|}}][t]
        }
    \]
    La soluzione tocca la retta \(y=1\) siccome l'integrale
    è improprio e converge grazie alla radice, abbiamo quindi delle saldature.
    Se non avessimo avuto la radice, l'integrale sarebb divergente e non toccheremmo quindi la retta.
    Consideriamo ora il dato iniziale \((0,1)\). Oltre alla soluzione costante
    abbiamo anche
    \[
        x = \sqrt{
            2\integral[1][y][\frac{1}{\sqrt{|\log t|}}][t]
        }
    \]
    Se invece prendiamo il dato iniziale sopra la soluzione costante, \(x_0 > 1\) allora
    abbiamo unicità globale.
    Dobbiamo tuttavia verificare che la soluzione che sta sopra, non abbia un asintoto verticale.
    Questo lo si può fare studiando l'integrale improprio manualmente
    \[
        \integral[1][y][\frac{1}{\sqrt{|\log t|}}][t] = \frac{x^2}{2} - \frac{x_0^2}{2}
    \]
    e mostrando che diverge.
    Per verificarlo, non si può usare il criterio della sublinearità, nonostante la funzione lo sia,
    in quanto \(\Omega\) non contiene un intervallo cartesiano della forma \(I\cartesianprod \realnumbers = \realnumbers^2\).
    Infatti non tutte le soluzioni sono globali, quelle sotto terminano prima.
    Verifichiamo la concavità delle soluzioni che stanno sopra \(y>1\). Abbiamo
    \begin{align*}
        y'' &= \sqrt{|\log y|} + x \frac{
            1
        }{
            2\sqrt{\log y}
        } \frac{y'}{y} \\
        &= \sqrt{\log x} + \frac{x^2}{2y} > 0
    \end{align*}
    Quindi le soluzioni per \(y>1\) sono convesse.
    Studiamo ora il caso \(0 < y < 1\).
    Consideriamo la soluzione che si attesta nell'origine
    per \(x \to 0\).
    Essa esiste ed è
    \[
        \integral[0][y][\frac{1}{\sqrt{|\log t|}}][t] = \frac{x^2}{2}
    \]
    che è integrabile nell'origine.
    Vogliamo capire come si comporta questa soluzione, che farà da discriminante.
    Studiamo la soluzione calcolandone la derivata dell'inversa
    \begin{align*}
        x'(y) &= \frac{
            \cancel2 \cdot \frac{1}{\sqrt{|\log y|}}
        }{
            \cancel2 \sqrt{
                2\integral[0][y][\frac{1}{\sqrt{|\log t|}}][t]
            }
        } = \frac{
            \frac{1}{\sqrt{|\log y|}}
        }{
            \sqrt{
                2\integral[0][y][\frac{1}{\sqrt{|\log t|}}][t]
            }
        } = \sqrt{
            \frac{
                \frac{1}{|\log y|}
            }{
                2 \integral[0][y][\frac{1}{\sqrt{|\log t|}}][t]
            }
        }
    \end{align*}
    Calcoliamo il limite
    \begin{align*}
        \lim_{y\to 0}
        \frac{
            \frac{1}{-\log y}
        }{
            2 \integral[0][y][\frac{1}{\sqrt{|\log t|}}][t]
        }
        \overset{H}{=} \lim_{y\to 0}
        \frac{
            \frac{1}{\log^2(y)} \cdot \frac{1}{y}
        }{
            2 \frac{1}{\sqrt{- \log y}}
        } = +\infty
    \end{align*}
    Quindi la derivata ha soluzione infinita, e l'inversa ha quindi derivata nulla.
    Facendo la stessa per gli altri casi (l'inversa ha derivata zero) troviamo che le soluzioni rimanenti
    si intersecano all'asse \(x\) in maniera perpendicolare, tagliandosi.
    Il valore del limite della derivata per \(y\to 0\)
    si potrebbe calcolare direttamente dall'equazione differenziale originale, ma non per la soluzione
    che si attesta sull'origine (in questo caso ha una forma di indecisione \(\infty \cdot 0\)).

\end{snippetexercise}

Se in un'equazione differenziale non compare la \(x\),
che si dice autonoma, le soluzioni sono tutte una traslata dell'altra.

% Parziale finisce qua 

Andiamo ora a mostrare che una soluzione di una equazione differenziale non può terminare nel nulla.
Sia \(y' = f(x,y)\) con \(f \colon \Omega \fromto \realnumbers^n\)
con \(\Omega \subseteq \realnumbers^{n+1}\) aperto e \(f\in \mathcal C(\Omega, \realnumbers^n)\).
Data una soluzione massimale, allora il grafico di questa deve andare a sbattere contro
la frontiera di \(\Omega\).
Non è necessariamente vero che se \(\varphi \colon (a,b) \fromto \realnumbers^n\)
è una soluzione massimale, allora \(\varphi\) tende ad un punto del bordo per \(x \to b\).
Per esempio \(y' = \frac{d}{dx} \sin(1/x)\) va a sbattere contro il bordo \(x=0\),
ma non con il limite. Quindi l'idea non va bene.
L'idea giusta è quella di uscire dai compatti.
Il bordo di \(\Omega\) ha distanza positiva da tutti i compatti \(K\).
Altrimenti, riuscirei  atrovare una successione di punti di \(K\) che converge a \(0\).
Avrei che un punto di \(K\) appartiene al bordo, ma ciò non è possibile in quanto \(\Omega\)
è aperto e non contiene il bordo.

\begin{snippettheorem}{Teorema di buon fine delle soluzioni}
    Sia \(\Omega \subseteq \realnumbers^{n+1}\) aperto,
    \(f \in \mathcal C(\Omega, \realnumbers^n)\).
    Sia \(\varphi \colon (a,b) \fromto \realnumbers^n\)
    una soluzione massimale del problema \(y'=f(x,y)\),
    \(a,b\in \realnumbers\).
    Allora, \(\forall K \subseteq \Omega\) compatto, \(\exists \delta\)
    di forma \[  0 < \delta < \frac{b-a}{2} \]
    tale che se \(x\in (a,b) \difference (a + \delta, b - \delta)\),
    allora \((x, \varphi(x)) \notin K\).
\end{snippettheorem}

Nota: \(\varphi\) deve essere definita su un aperto. Se il punto \(b\)
facesse parte dell'intervallo, potremmo usare il probelma di Cauchy centrato in \(b\)
per prolungarla, contro l'ipotesi che è massimale.

Supponiamo per esempio di avere \(\Omega = I \cartesianprod \realnumbers^n\)
con \(I = (a,b)\) intervallo aperto.
Supponiamo di avere una soluzione \(\varphi \colon (a,c) \fromto \realnumbers^n\)
con \(c < b\), \(\varphi\) limitata. Allora, non può essere una soluzione massimale,
in quanto non uscirebbe da tutti i compatti: per esempio un compatto centrato in \(c\).

Caso particolare: nelle ipotesi del teorema di buon fine, se
\(\varphi \colon (a,b) \fromto \realnumbers^n\)
è una soluzione massimale ed
\[
    \exists \lim_{x \to b^-} \varphi(x) = \alpha \in \realnumbers^n
\]
allora \((b, \alpha) \notin \Omega\) (cioè un iperrettangolo).
Quindi una soluzione non può terminare nel vuoto.
Se avessi \((b, \alpha) \in \Omega\), cioè avrei un pozzo che inghiotte la soluzione, potrei costruirci un compatto attorno, ma da questo
la soluzione non uscirebbe, il che è impossibile.

\begin{snippettheorem}{Teorema di non esistenza di pozzi}
    Sia \(\Omega \subseteq \realnumbers^{n+1}\) aperto,
    \(f \in \mathcal C(\Omega, \realnumbers^n)\).
    Sia \(\varphi \colon (a,b) \fromto \realnumbers^n\)
    una soluzione massimale del problema \(y'=f(x,y)\),
    \(a,b\in \realnumbers\), tale che \(\varphi(x) \to \alpha\) per \(x \to b\)
    e \((b, \alpha) \in \Omega\).
    Allora \(\varphi\) è prolungabile oltre \(b\).
\end{snippettheorem}

\begin{snippetproof}{Teorema di non esistenza di pozzi}
    Consideriamo il problema di Cauchy
    \[
        \begin{cases}
            y' = f(x,y) \\
            y(b) = \alpha
        \end{cases}
    \]
    Il punto \(\alpha\) appartiene all'insieme e quindi il problema ha senso.
    Inoltre, il problema ammette soluzione per il teorema di Peano.
    Sia \(\Psi\) una soluzione dello stesso problema definita su \((b-\varepsilon, b+\varepsilon)\)
    per qualche \(\varepsilon > 0\). Costruiamo allora la soluzione incollata
    \[
        g(x) = \begin{cases}
            \varphi(x) & x < b \\
            \Psi(x) & b \leq x \leq b + \varepsilon
        \end{cases}
    \]
    Vogliamo mostrare che \(g \colon (a,b+\varepsilon) \fromto \realnumbers^n\)
    risolve \(y' = f(x,y)\) nell'intervallo \((a,b+\varepsilon)\).
    Il problema è il raccordo.
    Sicuramente \(g\) è continua in \(b\), infatti per definizione
    \(g(b) = \Psi(b) = \alpha\) e \[
        \lim_{x \to b^-} g(x) = \lim_{x \to b^-} \varphi(x) = \alpha
    \]
    per ipotesi (la continuità da destra è banale).
    Mostriamo che \(\exists g'(b)\).
    Ciò segue dalla continuità della \(f\).
    Per il teorema del liimte della derivata, mi basta
    calcolare il limite solo in quel punto
    \[
        \lim_{x\to b^-} g'(x) = \lim_{x\to b^-} \Phi'(x)
        = \lim_{x\to b^-} f(x,\varphi(x)) = f(b,\alpha)
    \]
    per continuità di \(f\).
    Questo coincide con il limite da destra, che è \(f(b,\alpha) = \Psi'(b)\)
    (visto che \(\Psi\) risolve l'equazione).
\end{snippetproof}

\begin{snippetproof}{Teorema di buon fine}
    Sia \(K \subseteq \Omega\) compatto, \(\varphi \colon I \fromto \realnumbers^n\)
    tale che \((x, \varphi(x)) \in K\), \(\forall x \in I\).
    Vogliamo dimostrare che \(\varphi\) è prolungabile oltre \(I\).
    Sia \(I = (a,b)\). Se \(I\) fosse chiuso basta prolungarlo con il problema di Cauchy centrato in \(b\),
    come fatto nella dimostrazione della non esistenza di pozzi.
    Consideriamo una qualunque successione \(\{x_n\} \to b^-\)
    e consideriamo \(\{\varphi(x_n)\}\), quindi \((x_n, \varphi(x_n)) \in K\).
    La successione \(\{(x_n, \varphi(x_n))\}\) è contenuta in un compatto, e quindi
    a meno di sottosuccessioni converge ad un punto di \(K\).
    Senza perdita di generalità supponiamo che \(\{(x_n, \varphi(x_n))\} \to (b, \alpha) \in K\)
    per qualche \(\alpha\). L'idea è quella di applicare il teorema di non esistenza di pozzi.
    Tuttavia, ci serve che \(\varphi(x)\) abbia limite ad \(\alpha\) per \(x \to b^-\).
    Tuttavia, sappiamo solo che esiste una sottosuccessione per la quale abbiamo il limite,
    non sappiamo se pure per \(\varphi\).
    La funzione \(\varphi\) è lipschitz continua in un intorno di \(b\),
    in quanto la sua derivata \(f\) è limitata in quanto è continua in un compatto \(K\).
    Allora, è anche uniformemente continua, quindi ammette limite.
\end{snippetproof}

\begin{snippetproposition}{}
    Ogni funzione uniformemente continua ammette limite.
\end{snippetproposition}

\subsection{Esistenza di soluzioni massimali}

Supponiamo che relatiamente ad ogni punto di \(\Omega\)
siano soddisfatte le ipotesi del teorema di esistenza e unicità locali.
Allora ogni soluzione di \(y' = f(x,y)\),
cioè ogni coppia \((\varphi, I)\) ammette un unico prolungamento massimale.
L'idea quella dimostrazione è quella di considerare la famiglia dei prolungamenti
di \((\varphi, I)\), \(\{(\varphi_\gamma, I_\gamma)\}\)
tale che se \(\restr{\varphi_\gamma}{I} = \varphi\) allora \(I_\gamma \supseteq I\).
Sia \(I_\gamma = (a_\gamma, b_\gamma)\).
Si spera di poter definire un prolungamento sull'intervallo massimale \((a,b)\)
dove \(a = \inf_\gamma a_\gamma\) e \(b = \sup_\gamma b_\gamma\).
Dobbiamo capire come definire questo \(\tilde \varphi\), il prolungamento massimale nell'intervallo \((a,b)\).
L'idea è \(\tilde \varphi(x) = \varphi_\gamma(x)\)
se \(x \in I_\gamma\). Infatti,
\[
    (a,b) = \bigcup_\gamma I_\gamma
\]
Tuttavia, afifnché questa costruzione abbia senso, dobbiamo assicurarci che non vi siano biforcazioni.
Se due prolungamenti sono definite usllo stesso intervallo \(I_\gamma\), allora devono coincidere
in tutti i punti di \(I_\gamma \difference I\).

\begin{snippetlemma}{}
    Se due soluzioni definite sullo stesso intervallo coincidono in un qualche punto,
    allora coincidono in tutti i punti dell'intervallo di definizione.
\end{snippetlemma}

\begin{snippetproof}{}
    Siano \(\varphi_1\) e \(\varphi_2\) due soluzioni definite sullo stesso intervallo che coincidono in qualche
    punto \(x_0\).
    Abbiamo due casi: o esiste un ultimo punto in cui coincidono, oppure esiste un primo punto in cui non coincidono.
    Il caso in cui ha un ultimo punto in cui coincidono è il caso in cui c'è un punto di biforcazione.
    Non può succedere che ci sia un primo punto in cui non coincidono.
    L'insieme in cui \(\varphi_1 = \varphi_2\)
    è la fibra \((\varphi_1 - \varphi_2)^{-1}(\{0\})\), ma \(\varphi_1 - \varphi_2\)
    è continua, e siccome \(\{0\}\) è chiuso allora anche tale insieme è chiuso.
    Quindi, l'insieme dove le due funzioni non coincidono è un aperto.
    Siccome è un aperto, non può esserci un primo punto in cui non coincidono.
    Infatti, ogni punto in cui non coincidono ha un intorno sinistro di punti dove non coincidono.
    Siccome siamo in condizione di esistenza e unicità locale, non può esistere nemmeno l'ultimo punto, che sarebbe
    di biforcazione. Dobbiamo quindi avere soluzione unica.
\end{snippetproof}

Se siamo in regime di esistenza locale ma non unicità, allora le soluzioni massimali esistono ancora
ma non sono necessariamente uniche.

Dato l'equazione lineare vettoriale\[
    y' = A(x)y + b(x)
\]

Supponiamo di avere una matrice fondamentale \(\Phi(x) = [w_1(x), \cdots, w_n(x)]\).
Tutte le soluzioni dell'omogenea associata si scrivono nella forma \(\Phi(x) \cdot c\)
al variare di \(c \in \realnumbers^n\).
L'idea di Lagrange è stata quella di fare variare \(c\) come funzione,
possiamo trovare un \(c\) opportuno per una soluzione particolare?
Ogni funzione arbitraria di classe \(\mathcal C^1\) può essere scritta come
\(\Phi(x) c(x)\) per qualche \(c \in \mathcal C'(I, \realnumbers^n)\).
Data \(\varphi \in \mathcal C'(I, \realnumbers^n)\)
esiste \(c \in \mathcal C'(\realnumbers^n)\) tale che \(\varphi = \Phi(x) c\)?
Certo, basta scegliere \(c(x) = \Phi(x)^{-1}\varphi(x)\)
che è invertibile in quanto il determinante è non nullo.
Questo si chiama metodo di variazione delle costanti arbitrarie di Lagrange.
Vediamo quali condizioni deve soddisfare \(c(x)\)
in maniera tale che \(\Phi(x) c(x)\) risolva l'equazione non omogenea
\[
    y' = A(x)y + b(x)
\]
Chiaramente la soluzione costante è dell'omogenea associata, quindi non va bene.
Sostituiamo quindi \(y=\Phi(x) c(x)\)
\begin{DispWithArrows*}
    \Phi' c + \Phi c' &= (\Phi c)' = A\Phi c + b \Arrow{\(\Phi\) contiene \\ le soluzioni} \\
    \cancel{A\Phi c} + \Phi c' &= \cancel{A\Phi c} + b \\
    \Phi c' &= b \\
    c' &= \Phi^{-1} b
\end{DispWithArrows*}
Quindi \(c\) deve soddisfare \(\Phi c' = b\) cioè \(c' = \Phi^{-1} b\)
e poi come soluzione particolare
\[
    c = \integral[x_0][x][\Phi(t)^{-1}b(t)][t]
\]
e la soluzione della non è omogenea è
\[
    \tilde \varphi(x) = \Phi(x) \integral[x_0][x][\Phi(t)^{-1}b(t)][t]
\]
Nel caso \(n=1\) ritroviamo la formula per risolvere l'equazione del primo ordine lineare
\[
    y(x) = \underbrace{e^{\integral[x_0][x][a(t)][t]}}_{\Phi} \left[\,

\begin{snippettheorem}{di dipendenza continua dai dati}
    Sia \(\Omega \subseteq \realnumbers^{n+1}\) aperto e siano
    \(f_n, f \colon \Omega \fromto \realnumbers^n\)
    funzioni continue tali che
    \begin{enumerate}
        \item \(\{f_n\}\) converge uniformemente ad \(f\) sui compatti di \(\Omega\);
        \item \(\forall K \subseteq \Omega\) compatto esiste una costante \(L_K\)
        tale che \[
            ||f_n(x, y_1) - f_n(x, y_2)||
            \leq ||y_1-y_2||, \quad \forall (x,y_1), (x,y_2) \in \Omega, n \in \naturalnumbers
        \]
        (e quindi la stessa disuguaglianza vale anche per \(f\)).
    \end{enumerate}
    Sia \((x_0, y_0) \in \Omega\) e sia \(\varphi \colon (\alpha, \beta) \fromto \realnumbers^n\)
    la soluzione (massimale) del problema di Cauchy
    \[
        \begin{cases}
            y' = f(x,y) \\
            y(x_0) = y_0
        \end{cases}
    \]
    Sia \([a, b] \subseteq (\alpha, \beta)\) e sia \(\{y_n\}\)
    convergente a \(y_0\) in \(\realnumbers^n\).
    Allora \(\forall n\) sufficientemente grandi il problema di Cauchy
        \[
        \begin{cases}
            y' = f_n(x,y) \\
            y(x_0) = y_n
        \end{cases}
    \]
    ammette soluzione unica \(\varphi_n\) definita su \([a,b]\)
    e la successione \(\{\varphi_n\}\) convere uniformemente a \(\varphi\)
    sull'intervallo \([a,b]\).
\end{snippettheorem}

\section{Teoria della misura}

Lo spazio \((\mathcal C^0(0,1), ||\cdot||_1)\)
dove \[ ||f||_1 = \int\limits_0^1 |f| \]
non è uno spazio completo. 
Passando all'integrale di Lebesgue lo spazio diventa completo.

Sia \(X\) un insieme generico e consideriamo \(\Phi \subseteq \powerset(X)\)
e vogliamo definire una misura per \(\Phi\), cioè \(\mu \colon \Phi \fromto Y\).
Nel caso più generale ci basta richiedere che \(Y\) sia uno spazio vettoriale.
Nel contesto classico dell'analisi reale e complessa ci si limita a considerare
\(Y = \complexnumbers, \realnumbers\).
La versione più genuina è quella sui complessi, noi ci limiteremo ai reali non-negativi,
quindi \(\mu \colon \Phi \fromto [0,+\infty]\).
Tuttavia, questo non è più uno spazio vettoriale.
In particolare ci sarebbero problemi con forme \(0 \cdot \infty\).
Per come definiremo la misura un caso del genere viene sempre valutato con \(0\).
Se vogliamo una misura che assuma pure valori negativi, allora per forza
devo escludere gli infiniti, in quanto non riuscirei a gestire casi del tipo \(+\infty - \infty\),
e quindi tutti gli spazi avrebbero misure finite.
Non faremo nessuna restrizione su \(X\), mentre dovremo restringere \(\Phi\).
In generare non è possibile definire la misura su tutti i sottoinsieme delle parti.
\(\Phi\) deve essere una \(\sigma\)-algebra.

La parola \(\sigma\) rappresenta l'unione numerabile.

\begin{snippetdefinition}{\(\sigma\)-algebra}
    Sia \(\Phi \subseteq \powerset(X)\). Diciamo che \(\Phi\) è una \emph{\(\sigma\)-algebra}
    se \begin{enumerate}
        \item \emph{chiusura rispetto ai complementi:} se \(A \in \Phi\), allora \(X \difference A \in \Phi\);
        \item \emph{chiusura rispetto alle unioni numerabili:} se \(\{A_n\}_{n \in \naturalnumbers} \subseteq \Phi\), allora
        \[
            \bigcup_{n=1}^\infty A_n \in \Phi
        \]
        \item \(\emptyset \in \Phi\).
    \end{enumerate}
\end{snippetdefinition}

Le prime due proprietà ci danno la chiusura anche rispetto alle intersezioni numerabili.
La terza è analoga a \(X \in \Phi\).
Abbiamo anche la chiusura rispetto alla differenza insiemistica,
ciop se \(A,B \in \Phi\), allora \(B \difference A \in \Phi\),
e pure rispetto alla differenza simmetrica, cioè se \(A, B \in \Phi\),
allora \(A \Delta B \in \Phi\).

Se avessimo l'unione arbitraria allora tutti i sottoinsiemi sarebbero misurabili.

\begin{snippetexample}{}
    L'insieme delle parti \(\powerset(X)\) è una \(\sigma\)-algebra,
    e pure \(\{\emptyset, X\}\) è una \(\sigma\)-algebra.
\end{snippetexample}

\begin{snippetdefinition}{\(\sigma\)-algebra generate}
    Sia \(S \subseteq \powerset(X)\).
    La \emph{\(\sigma\)-algebra generata da \(S\)}
    è la più piccola \(\sigma\)-algebra che contiene \(S\).
\end{snippetdefinition}

Di sicuro \(\powerset(X)\) è una \(\sigma\)-algebra
che contiene \(S\). Si dimostra che l'intersezione di tutte le \(\sigma\)-algebre
che contengono \(S\) è una \(\sigma\)-algebra, quindi questa è la più piccola.

\begin{snippetexample}{}
    Consideriamo \(X=\realnumbers\). È una \(\sigma\)-algebra
    la famiglia di tutti i sottoinsiemi di \(X\)
    al più numerabili o con complementare al più numerabile.
    Questa è generata da tutti gli insiemi finiti.
    Non è una \(\sigma\)-algebra la famiglia
    degli insiemi che contengono un dato punto di \(\realnumbers\)
    (non è chiusa rispetto al complementare).
\end{snippetexample}

\begin{snippetdefinition}{\(\sigma\)-algebra di Borel}
    Se \(X\) è uno spazio topologico, allora
    la \emph{\(\sigma\)-algebra di Borel} è la \(\sigma\)-algebra generata dalla topologia. 
\end{snippetdefinition}

Questa contiene sicuramente tutti gli aperti, tutti i chiusi, e possibilmente
molti aperti che non sono nè aperti nè chiusi.

\begin{snippetdefinition}{}
    \(G_\delta\) è la famiglia degli insiemi che possono essere rappresentati
    come intersezione numerabile di aperti di una topologia.
\end{snippetdefinition}

\begin{snippetdefinition}{}
    \(F_\sigma\) è la famiglia degli insiemi che possono essere rappresentati
    come unione arbitraria di chiusi di una topologia.
\end{snippetdefinition}

\(G_\delta\) ed \(F_\sigma\) stanno nella \(\sigma\)-algebra di Borel.

\begin{snippetexample}{Alcuni sottoinsieme di \(\realnumbers\) rispetto alla topologia euclidea e \(\sigma\)-algebra di Borel}
    Gli insiemi \([a,b), [a,b], (a,b], (a,b)\)
    sono tutti contemporaneamente insiemi \(G_\delta\) e \(F_\sigma\)
    rispetto a \((\realnumbers, \tau_e)\).
    Abbiamo per esempio
    \[
        [a,b) = \bigcap_n \left(
            a-\frac1n, b
        \right)
        = \bigcup_n \left[a,b-\frac1n\right]
    \]
\end{snippetexample}

In \(\realnumbers\), ogni insieme numerabile è un \(F_\sigma\) (basta che sia \(T_1\)).
Per esempio, \(\mathbb{Q}\) è un \(F_\sigma\), in quanto unione numerabile di punti.

\begin{snippettheorem}{Teorema di Baire}
    Se uno spazio metrico completo \((X, d)\)
    è un'unione numerabile di chiusi,
    \[
        X = \bigcup_{i=1}^\infty X_i
    \]
    Allora, almeno uno di questi chiusi ha interno non vuoto.
\end{snippettheorem}

Possiamo considerare anche le intersezioni numerabili di \(F_\sigma\),
cioè \({F_{\sigma}}_\delta\),
e le unioni numerabili di \(G_\delta\), cioè \({G_{\delta}}_{\sigma}\),
e ottene nuovi insiemi di elementi Borealiani.
Possiamo continuare con \(G_{\delta\sigma\delta}\) etc.
Si dimostrare che esistono elementi boreliani che non sono contenuti in
una di queste concatenazioni.

\begin{snippetproposition}{}
    Se \(S\) ha la cardinalità del continuo, allora anche \(\sigma(S)\)
    ha la cardinalità del continuo.
\end{snippetproposition}

\begin{snippetproof}{}
    Si dimostra usando i numeri ordinali e il prinicpio di induzione transfinita.
\end{snippetproof}

Se \(S = \{(a,b) \suchthat a<b\}\), allora \(\sigma(S)\)
è la \(\sigma\)-algebra di Borel.
Quindi, sia \(S\) che \(\sigma(S)\) hanno la cardinalità del continuo.
Quindi, non riusciamo a coprire tutto l'insieme delle parti, ed esistono quindi
sottoinsiemi che non sono Borel-misurabili.

\begin{snippetexercise}{}
    Dimostrafre che non esistono \(\sigma\)-algebre esattamente numerabili.
    Una \(\sigma\)-algebra può essere finita oppure almeno continua.
\end{snippetexercise}

La \(\sigma\)-algebra di Borel non ci basta per quello che vogliamo fare,
la \(\sigma\)-algebra di Lebesgue \(\mathcal L(\realnumbers)\) sarà più grande.

\begin{snippetdefinition}{Spazio misurabile}
    Sia \(X\) un insieme e \(\mathcal M\) una \(\sigma\)-algebra su \(X\).
    La coppia \((X, \mathcal M)\) è uno \emph{spazio misurabile}.
\end{snippetdefinition}

Non potremo integrare tutte le funzioni, solo quelle misurabili.

\begin{snippetdefinition}{Funzione misurabile}
    Sia \((X, \mathcal M)\) uno spazio misurabile e \((Y, \tau)\)
    uno spazio topologico e sia \(f \colon X \fromto Y\).
    Diciamo che \(f\) è misurabile se \(f^{-1}(V) \in \mathcal M, \forall V \in \tau\).
\end{snippetdefinition}

Nella nostra teoria dell'integrazione vogliamo includere la teoria dell'integrazione classica.
Vogliamo quindi che, in particolare, tutte le funzioni continue siano integrabili.

\begin{snippetexercise}{}
    Dimostrare che \(f\) è misurabile se \(f^{-1}(V) \in \mathcal M\) per ogni \(V \in \sigma(\tau)\).  \\
    \emph{Hint:} usare la proposizione sottostante.
\end{snippetexercise}

\begin{snippetproposition}{}
    Sia \(X\) uno spazio misurabile e \(Y\) un insieme e sia \(f \colon X \fromto Y\).
    Allora, la famiglia dei sottoinsiemi di \(Y\) la cui controimmagine tramite \(f\) è misurabile,
    forma una \(\sigma\)-algebra. 
\end{snippetproposition}

Consideriamo il caso particolare di funzioni \(Y=\overline \realnumbers\).
C'è un criterio per stabilire la misurabilità di \(f\).

\begin{snippettheorem}{}
    Sia \(Y = \overline \realnumbers\) con la topologia euclidea. Allora,
    \(f \colon X \fromto Y\) è misurabile se e solo se
    \[
        f^{-1}((a,+\infty]) \in \mathcal M, \quad \forall a \in \realnumbers
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    \begin{iffproof}
        \item banale;
        \item Se \(f^{-1}((a,+\infty])\) è misurabile, allora anche
        \(f^{-1}([b,+\infty))\) è misurabile in quanto una \(\sigma\)-algebra è chiusa per i complementi
        e la preimmagine di un complemento è il complemento della preimmagine.
        Quindi, facendo intersezioni, \(f^{-1}((a,b))\) è misurabile.
    \end{iffproof}
\end{snippetproof}

La condizione del teorema vale anche con l'estremo finito incluso o con il complementare
dell'insieme, quindi 4 totali condizioni equivalenti.

\begin{snippetcorollary}{}
    Se \(f\) è misurabile, allora \(f^{-1}(\{a\}) \in \mathcal M, \forall a \in \realnumbers\).
\end{snippetcorollary}

\begin{snippetproposition}{}
    Abbiamo che
    \[
        \{a\} = [-\infty, a] \intersection [a, +\infty]
    \]
\end{snippetproposition}

\begin{snippetexample}{Non vale il viceversa}
    Consideriamo \(f \colon \realnumbers \fromto \realnumbers\)
    e sia \(\mathcal M\) una \(\sigma\)-algebra
    su \(\realnumbers\) per cui \(\exists A \in \realnumbers\)
    tale che \(A \notin \mathcal M\) (altrimenti è impossibile).
    Sia \(\mathcal M\) la \(\sigma\)-algebra di Borel.
    Consideriamo una funzione iniettiva e positiva, come per esempio \(e^x\), e definiamo
    \[
        f(x) = \begin{cases}
            e^x & x \in A \\
            -e^x & x \in A^c
        \end{cases}
    \]
    Abbiamo che \(f^{-1}((0, +\infty)) = A \notin \mathcal M\).
    Tuttavia, \(f^{-1}(\{a\})\) è un singoletto (misurabile) in quanto \(f\) è iniettiva.
\end{snippetexample}

\begin{snippetproposition}{}
    Sia \(A \subseteq X\). Allora \(A\) è misurabile se e solo se
    \(\chi_A\) è misurabile.
\end{snippetproposition}

\begin{snippetproposition}{}
    Siano \(X\) uno spazio misurabile,
    \(Y,Z\) spazi topologici,
    \(g \colon X \fromto Y\) e \(h \colon Y \fromto Z\).
    Sia \(f = h \circ g\). Se \(h\) è continua e \(g\) è misurabile, allora \(f\) è misurabile.
\end{snippetproposition}

In generale la composizione di due funzioni misurabili non è misurabile.

\begin{snippetproposition}{}
    Somma e prodotto di funzioni misurabili sono misurabili.
\end{snippetproposition}

\begin{snippetproposition}{}
    Sia \(f_n \colon X \fromto \overline \realnumbers\) misurabile.
    Allora, \(\limsup f_n\) e \(\liminf f_n\) (punto per punto) sono misurabili.
\end{snippetproposition}

\begin{snippetproof}{}
    Osserviamo che \(\sup_n f_n\) (cioè la funzione \(x \to \sup_n f_n(x)\))
    è misurabile (e analogamente con l'infimum).
    Per verificarlo usiamo il criterio e mostriamo che
    \begin{align*}
        (\sup_n f_n)^{-1}((a, +\infty])
        = \bigcup_n f^{-1}_n ((a, +\infty])
    \end{align*}
    e quindi è misurabile in quanto unione di insiemi misurabili.
    Vale l'analogo con l'intersezione per l'infimum.
    Mostriamo la doppia inclusione:
    \begin{doubleinclusion}
        \item Sia \(x \in (\sup_n f_n)^{-1}((a, +\infty])\) e denotiamo \(g = \sup_n f_n\).
        Abbiamo allora che \(g(x) \in (a,+\infty]\), ossia \(g(x) > a\).
        Quindi, \(\sup f_n(x) > a\).
        Siccome il supremum è maggiore di \(a\), allora esiste almeno un elemento
        \(\overline n\) tale che \(f_{\overline n}(x) > a\).
        Quindi \(x \in f_{\overline n}^{-1}((a, +\infty])\) cioè \[
            x \in \bigcup_n f^{-1}_n ((a, +\infty])
        \]
        \item Sia \[
            x \in \bigcup_n f^{-1}_n ((a, +\infty])
        \]
        Allora \(\exists \overline n\) tale che \(x \in f^{-1}_{\overline n}((a, +\infty])\),
        cioè \(f_{\overline n}(x) > a\).
        Ma \(\sup_n f_n(x)\) è un maggiorante, cioè \(\sup_n f_n(x)>a\) e quindi
        \(x \in (\sup_n f_n)^{-1}((a, +\infty])\).
    \end{doubleinclusion}
    Dalla definizione di limsup,
    \[
        \limsup f_n = \lim_n \sup_{k \geq n}
        f_k = \inf_n \sup_{k \geq n} f_k
    \]
    Siccome il supremum e l'infimum di funzioni misurabili sono misurabili, allora
    vale anche che il limsup (e liminf) rimangono funzioni misurabili.
\end{snippetproof}

In particolare, il massimo e il minimo fra due funzioni misurabili rimane misurabile.

\begin{snippetproposition}{}
    Sia \(f \colon X \fromto \overline \realnumbers\)
    misurabili. Allora \(|f|\) è misurabile.
\end{snippetproposition}

\begin{snippetproof}{}
    Siano \(f^+ = \max\{f,0\}\) e \(f^- = -\min\{f,0\}\)
    che sono funzioni misurabili e rispettivamente positive e negative.
    Chiaramente \(|f| = f^+ + f^-\) e quindi è misurabile.
\end{snippetproof}

\begin{snippetexample}{Non vale il converso}
    Sia \(A \notin \mathcal M\) non misurabile. Scegliamo
    \(f = \chi_A - \chi_{A^c}\). Chiaramente \(|f| = 1\) è misurabile
    mentre \(f\) non lo è.
\end{snippetexample}

\begin{snippetproposition}{}
    Sia \(f \colon X \fromto \complexnumbers\).
    Allora, \(f\) è misurabile se e solo se \(\Re f\) e \(\Im f\) sono misurabili.
\end{snippetproposition}

\begin{snippetdefinition}{Funzione semplice}
    Sia \(X\) un'insieme ed \(f \colon X \fromto [0,+\infty)\).
    Diciamo che \(f\) è \emph{semplice} se \(f\) ha immagine finita, cioè
    \(f(X)\) è un'insieme finito.
\end{snippetdefinition}

Ci interessa il caso di \(X\) spazio misurabile e considereremo funzioni semplici misurabili.

\begin{snippetproposition}{}
    Una funzione semplice è misurabile se e solo se
    tutte le sue fibre sono misurabili.
\end{snippetproposition}

In generale vale solo che le fibre sono misurabili, non il converso.
Invece con le funzioni semplici vale l'equivalenza.

\begin{snippetproof}{}
    \begin{iffproof}
        \item banale;
        \item Sappiamo che \(f(X)\) è finito.
        La controimmagine di un intervallo è unione di un numero finito di fibre.
    \end{iffproof}
\end{snippetproof}

Se \(f\) è una funzione semplice, allora \(\exists \{\alpha_i\}_{i=1}^n \subseteq \realnumbers^+\)
tale che
\[
    f = \sum_{i=1}^n \alpha_i \chi_{A_i}
\]
dove \(A_i = f^{-1}(\{\alpha_i\})\).

\begin{snippettheorem}{}
    Sia \(X\) uno spazio misurabile, \(f \colon X \fromto \realnumbers^{\geq 0} \union \{+\infty\}\).
    Se \(f\) è misurabile, allora \(\exists \{s_n\}\)
    successione di funzioni semplici misurabili tali che
    \(s_n(x) \uparrow f(x), \forall x \in X\).
\end{snippettheorem}

Una misura è una funzione numerabilmente additiva di insiemi.

\begin{snippetdefinition}{Misura}
    Sia \((X, \mathcal M)\) uno spazio misurabile. Una \emph{misura}
    è una funzione \(\mu \colon \mathcal M \fromto [0,+\infty]\)
    che sia numerabilmente additiva, cioè se \(\{A_i\}_{i=1}^\infty \subseteq \mathcal M\)
    e \(A_i \intersection A_j = \emptyset\) per \(i \neq j\), allora
    \[
        \mu\left(
            \bigcup_{i=1}^\infty A_i
        \right)
        = \sum_{i=1}^\infty \mu(A_i)
    \]
\end{snippetdefinition}

Notiamo che fra questi addendi ci potrebbe essere infinito.
Non può mai capire un caso \(\infty - \infty\) perché stiamo considerando misure positive.
In questi casi abbiamo \[
    a + \infty = \infty, \quad
    a \cdot \infty = \begin{cases}
        \infty & a>0 \\
        0 & a = 0
    \end{cases}
\]
Inoltre, l'unione insiemistica nella definizione non dipende
dall'ordine degli insiemi, mentre una serie in generale sì.
Tuttavia, la serie è incondizionata in quanto i termini sono positivi,
quindi il valore è indipendente dall'ordine della somma.
Se la misura fosse complessa, non potremmo fare la stessa cosa:
dobbiamo supporre che l'intero spazio abbia misura finita per
far sì che la serie sia incondizionata.

I casi banali sono quello in cui tutti gli insiemi hanno misura nulla o infinita.
Consideriamo quindi i casi dove esiste almeno un insieme di misura non nulla
o un insieme di misura non infinita.

\begin{snippettheorem}{}
    \begin{enumerate}
        \item \(\mu(\emptyset) = 0\);
        \item \emph{finita-addività:} \[
            \mu\left(
                \bigcup_{i=1}^n A_i
            \right)
            = \sum_{i=1}^n \mu(A_i)
        \]
        \item \emph{monotonia:} Se \(A \subseteq B\), con \(A,B \in M\) allora
        \[
            \mu(A) \leq \mu(B)
        \]
        \item se \(A_1 \subseteq A_2 \subseteq \cdots\) è una successione di insiemi misurabili annidati, allora
        \[
            \mu\left(
                \bigcup_{i=1}^\infty A_i
            \right)
            = \lim_{n} \mu(A_n) = \sup_n \mu(A_n)
        \]
        \item se \(A_1 \supseteq A_2 \supseteq \cdots\) è una successione di insiemi misurabili annidati,
        e \(\exists \overline n\) tale che \(\mu(A_n) < +\infty\),
        allora
        \[
            \mu\left(
                \bigcap_{i=1}^\infty A_i
            \right)
            = \lim_{n} \mu(A_n) = \inf_n \mu(A_n)
        \]
    \end{enumerate}
\end{snippettheorem}

    In \(\realnumbers\),
    consideriamo \(A_n = (n ,+\infty)\) la cui intersezione è vuota, la misura 
    dell'intersezione è quindi nulla, ma l'infimum delle misure è infinito.
\end{snippetexample}

\begin{snippetdefinition}{Spazio di misura}
    Sia \((X, \mathcal M)\) uno spazio misurabile e sia \(\mu\)
    una misura di \(\mathcal M\).
    Allora, \((X, \mathcal M, \mu)\) è deto \emph{spazio di misura}. \\
    \emph{Syntax:} scriviamo \((X, \mu)\) se non vogliamo specificare la \(\sigma\)-algebra.
\end{snippetdefinition}

Nel caso in cui \(\mu(X) < +\infty\), si può normalizzare la misura
\(\sfrac{\mu}{\mu(X)}\) in maniera tale che \(\mu(X) = 1\).
In questo caso si parla di spazi di probabilità e gli elementi si chiamano eventi.

\begin{snippetdefinition}{Misura del conteggio}
    Sia \(X\) un insieme con la \(\sigma\)-algebra \(\powerset(X)\).
    La \emph{misura del conteggio} su questo spazio è definita come
    \[
        \mu(E) = \begin{cases}
            |E| & |E| < +\infty \\
            +\infty \text{ otherwise}
        \end{cases}
    \]
\end{snippetdefinition}

Con questa misura tutti gli insiemi e tutte le funzioni sono misurabili.

\begin{snippetdefinition}{Misura di condensazione o di concentrazione di massa o di Dirac}
    Sia \(X\) un insieme e \(x_0 \in X\). La \emph{misura di Dirac} è definita come
    \[
        \delta_{x_0}(E) = \begin{cases}
            1 & x_0 \in E \\
            0 & x_0 \notin E
        \end{cases}
    \]
\end{snippetdefinition}

\subsection{Completezza delle misure}

Esistono insiemi non vuoti la cui misura è nulla.

Consideriamo per esempio \(E \in \mathcal M\)
tale che \(E \neq 0\) e \(\mu(E) = 0\).
Può capire che esista \(B \subseteq E\) tale che \(B \notin \mathcal M\).
Ciò non è desiderabile in quanto
noi vogliamo costruire degli integrali usando la misura. Dal punto di vista integrale,
se due funzioni differiscono su un insieme di misura nulla, devono coincidere.
Consideriamo le funzioni
\(\chi_X, \chi_{E^c}\).
Queste due funzioni differiscono 
solamente in un insieme di misura nulla (\(E\)).
Ma allora le funzioni \(\chi_X\) e \(\chi_B\)
differiscono ancora meno in quanto \(B \subseteq E\).
Tuttavia, \(B\) non è misurabile.
Definiamo quindi la misura completa.

\begin{snippetdefinition}{Misura completa}
    Sia \((X, \mathcal M, \mu)\) uno spazio di misura.
    Diciamo che \(\mu\) è completa
    se \(\forall E \in \mathcal M\) tale che \(\mu(E) = 0\), se \(B \subseteq E\) allora \(B \in \mathcal M\)
\end{snippetdefinition}

E quindi pure \(\mu(B) = 0\) per la monotonia.
Purtroppo esistono misure non complete, ma possiamo completarle ad uno spazio completo.
Basta aggiungere ad \(\mathcal M\) tutti i sottoinsiemi di insiemi misurabili con misura nulla e associargli una misura nulla.
Se facciamo questo otteniamo una \(\sigma\)-algebra più grande.
Annunciamo la costruzione formale.

\begin{snippetdefinition}{Completamento di una misura}
    Sia \((X, \mathcal M, \mu)\) uno spazio di misura.
    Consideriamo tutti gli insiemi \(E\)
    tale che \(A \subseteq E \subseteq B\)
    con \(A,B \in \mathcal M\) e \(\mu(B \difference A) = 0\).
    La nuova \(\sigma\)-algebra ottenuta aggiungendo tali insiemi \(E\)
    ad \(\mathcal M\) viene denotata \(\mathcal M^*\).
    Definiamo la misura associata per il nuovo spazio \(\mu^*\)
    come segue: \(\mu^*(A) = \mu(A)\) se \(A\in \mathcal M\),
    mentre \(\mu^*(E) = \mu(A) = \mu(B)\) se \(E\) è come sopra. 
\end{snippetdefinition}

Dire che \(\mu(B \difference A = 0)\) implica che \(\mu(A) = \mu(B)\)
in quanto \(\mu(B) = \mu(A) + \mu(B \difference A)\).
Il procedimento descritto corrisponde ad aggiungere ad \(\mathcal M\)

Dobbiamo verificare che se lo stesso insieme è
contenuto fra molteplici coppie, la misura coincide.

\begin{snippetproposition}{TODO: lettere sbagliate}
    La definizione di completamento di una misura \(\mu\) è ben posta, cioè
    se \(A_1 \subseteq E_1 \subseteq B_1\) e \(A_2 \subseteq E_2 \subseteq B_2\) con \(A_1, B_1, A_2, B_2 \in M\),
    allora se \(E_1 = E_2\) vale che \(\mu^*(E_1) = \mu^*(E_2)\).
    Inoltre, \(\mathcal M^*\) è una \(\sigma\)-algebra
    ed è la più piccola \(\sigma\)-algebra che contiene \(\mathcal M\)
    tale che la misura sia completa.
\end{snippetproposition}

\begin{snippetproof}{TODO: lettere sbagliate}
    Abbiamo che \(A_1 \difference A_2 \subseteq B_2 \difference A_2\)

\end{snippetproof}

Per questo motivo \(\mathcal B(\realnumbers) \subseteq \mathcal L(\realnumbers)\).

\subsection{Misura di Lebesgue}

Vogliamo costruire
una misura che rispecchi il nostro senso comune circa la misura degli insiemi.
Vogliamo che
\begin{enumerate}
    \item \(\mu([a,b]) = b-a\);
    \item \emph{invarianza per traslazione:} \(\mu(E + t) = \mu(E)\)
\end{enumerate}
Vorremmo anche che tutti i sottoinsiemi siano misurabili, ma come vedremo non possiamo
soddisfare pure questa condizione.

La misura di Peano-Jordan è solo finitamente additiva, non numerabilmente.
Quindi non è una misura nel senso moderno.
Peano-Jordan può essere definita considerando i ricoprimento dell'insieme \(E\)
con un numero finito di iper-rettangolo limitati e prendendo l'infimum delle loro misure.

Nel contesto della misura di Lebesgue, ricoriamo \(E\)
con una collezione numerabile di intervalli limitati.
Consideriamo quindi \(I_i = (a_i, b_i)\) e la serie
\[
    \sum_{i=1}^\infty l(I_i)
\]
Definiamo quindi la misura esterna di \(E\)
come \[
    \mu^*(E) = \inf_{\substack{E \subseteq U_{i=1}^\infty I_i \\ I_i = (a_i, b_i)}} 
    \sum_{i=1}^\infty l(I_i)
\]

\begin{snippetexample}{Misura di \(\rationalnumbers \intersection [0,1]\)}
    Consideriamo l'insieme \(\rationalnumbers \intersection [0,1]\).
    Secondo Peano-Jordan, la sua misura interna è 0, mentre quella esterna è 1,
    che non coincidono e quindi non è misurabile secondo Peano-Jordan.
    Con la misura di Lebesgue possiamo estenderci ad un infinito numerabile
    di intervalli. Sia
    \(\{q_n\}\) una successione che enumera elementi di \(\rationalnumbers \intersection [0,1]\).
    Copriamo \(q_1\) con un intervallo \(I_1\) di lunghezza
    \(l(I_1) = \sfrac{\varepsilon}{2^1}\).
    Induttivamente ricopriamo \(q_n\) con un intervallo \(I_n\)
    di lunghezza \(l(I_n) = \frac{\varepsilon}{2^n}\).
    Questi itervalli si sovrappongono sicuramente, ma ciò non importa
    perché loro lunghezza complessiva è
    \[
        \sum_{i=1}^\infty \frac{\varepsilon}{2} = \varepsilon
    \]
    Quindi, siccome \(\varepsilon\) è arbitrario, prendendo l'inf otteniamo \(0\).
    Quindi la misura è \(\mu(\rationalnumbers \intersection [0,1]) \leq 0\),
    cioè \(0\).
    Lo stesso vale per ogni singolo insieme numerabile.
\end{snippetexample}

Con Peano-Jordan non conta solamente quanti punti abbiamo, ma anche come sono disposti.
Se per esempio abbiamo una successione di punti con limite, allora possiamo ricoprire il punto di limite
con un intervallo di lunghezza \(\varepsilon\), e i termini che escono da tale intervallo
sono finiti \(N\), quindi basta ricoprirli con \(N\) intervalli di raggio \(\varepsilon\).
Allora, abbiamo una misura di \(\varepsilon + N\varepsilon \to 0\), e quindi è misurabile.
Tuttavia, se la disposizione è diversa, non è più misurabile.

\begin{snippetproposition}{}
    Si dimostra che la misura esterna di un intervallo coincide con la lunghezza.
    Inoltre, è invariante per traslazione.
\end{snippetproposition}

Abbiamo tuttavia un problema. Questa misura esterna non è numerabilmente additiva
(nemmeno finitamente additiva).

\begin{snippetproposition}{La misura esterna è numerabilmente subadditiva}
    La misura esterna è subadditiva \[
        \mu^*\left(
            \bigcup_{n=1}^\infty S_n
        \right) \leq \sum_{n=1}^\infty \mu^*(S_n)
    \]
\end{snippetproposition}

\begin{snippetproof}{}
    Sia \(\varepsilon > 0\).
    Ricopriamo ciascuno degli \(S_n\) con una collezione numerabile di intervalli
    \[
        S_n \subseteq \bigcup_{i=1}^\infty I_i^n
    \]
    Per definizione di misura esterna, possiamo
    ricoprire \(S_n\) in maniera tale che
    \[
        \mu^*(S_n) > \sum_{i=1}^\infty l(I^n_i) - \frac{\varepsilon}{2}
    \]
    Facendo variare sia \(i\) che \(n\), la famiglia \(\{I_i^n\}\)
    ricopre \(\bigcup_{i=1}^\infty S_n\).
    Questa famiglia rimane numerabile per diagonalizzazione di Cantor.
    Questo è quindi un ricoprimento dell'insieme che vogliamo misurare, e per definizione
    di misura esterna dell'unione con l'infimum
    \[
        \mu^*\left(
            \bigcup_{n=1}^\infty S_n
        \right) \leq \sum_{n=1}^\infty l(I_i^n)
    \]
    Questa somma non dipende dall'ordine perché è incondizionata.
    Mettendo assieme le due espressioni troviamo la somma delle misure esterne
    \begin{align*}
        \varepsilon + \sum_n \mu^*(S_n) &> \sum_{n,i} l(I_i^n)
        \geq \mu^* \left(
            \bigcup_{n=1}^\infty S_n
        \right)
    \end{align*}
    Quindi per arbitrarietà di \(\varepsilon\) vale la tesi.
\end{snippetproof}

Abbiamo quindi un problema in quanto ci serve una misura numerabilmente additiva.
L'idea di Caratheodory è quella di considerare una sottofamiglia
di insiemi dove la misura è numerabilmente additiva.
Anzi, è sufficiente considerare una sottofamiglia dove la misura è finitamente additiva.
Questa famiglia è una \(\sigma\)-algebra e la misura
esterna ristretta a questa famiglia è numerabilmente additiva.
Inoltre, la misura risultante è completa. Questa viene detta \(\sigma\)-algebra di Lebesgue.

Imponiamo quindi che un insieme è misurabile
se ogni altro insieme reale lo spacca in maniera additiva.
Questo corrisponde a dire che la misura sia finitamente additiva.

\begin{snippetdefinition}{Condizione di Carathéodory}
    Un insieme \(E \subseteq \realnumbers\) è \emph{misurabile secondo Lebesgue}
    se \(\forall A \subseteq \realnumbers\),
    \[
        \mu^*(A) = \mu^*(A \cap E) + \mu^*(A \cap E^c)
    \]
\end{snippetdefinition}

Alternativamente possiamo dire \[
    \mu^*(A) \geq \mu^*(A \cap E) + \mu^*(A \cap E^c)
\]
Il segno \(\leq\) è sempre vero per la subadditività.

\begin{snippetproposition}{}
    La famiglia di sottoinsiemi reali \(\mathcal L\) che soddisfano la condizione di Carathéodory
    è una \(\sigma\)-algebra, e \(\restr{\mu^*}{\mathcal L}\) è una misura.
\end{snippetproposition}

\begin{snippetproposition}{}
    \(\mathcal L\) contiene ogni insieme di misura esterna nulla,
    cioè \(\mu^*(E) = 0 \implies E \in \mathcal L\).
\end{snippetproposition}

\begin{snippetproof}{}
    Abbiamo che \(\mu^*(A \intersection E) = 0\)
    in quanto \(A \intersection E \subseteq E\) siccome \(\mu^*\)
    ha una monotonia, e quindi
    \[
        \mu^*(A \intersection E^c) \leq \mu^*(A)
    \]
    perché \(A \intersection E^c \subseteq A\).
\end{snippetproof}

Ogni sottoinsieme di misura esterna nulla ha misura esterna nulla per la monotonia della misura esterna,
e quindi sta in \(\mathcal L\) cioè \(\restr{\mu^*}{\mathcal L}\) è completa.

\begin{snippetdefinition}{Misura di Lebesgue}
    Definiamo la misura di Lebesgue
    \[
        \mu \triangleq \restr{\mu^*}{\mathcal L}
    \]
\end{snippetdefinition}

\begin{snippettheorem}{}
    La \(\sigma\)-algebra di Lebesgue ristretta
    alla \(\sigma\)-algebra di Borel \(\restr{\mu^*}{\mathcal B}\) non è completa.
\end{snippettheorem}

Gli insiemi \((a,+\infty)\)
sono misurabili e quindi anche tutti gli intervalli.

\begin{snippetproof}{}
    Il fatto si misura deriva dal fatto che se restringo ad una misura, rimane
    comunque additiva per la \(\sigma\)-algebra.
    \(\restr{\mu}{\mathcal B}\) non è completo: esistono insiemi di misura nulla con
    cardinalità del continuo, borealini (Ternario di Cantor, vedremo in futuro).
    Anche \(\mathcal B\) ha la cardinalità del continuo, quindi l'insieme delle parti
    dell'insieme ternario di Cantor non può essere tutto contenuto in \(\mathcal B\).
\end{snippetproof}

Mostriamo ora che il completamento è precisamente \(\restr{\mu^*}{\mathcal L}\).

\begin{snippetproposition}{}
    Sia \(E \in \mathcal L\). Allora \(\exists G \in G_\delta\) tale che
    \(G \supseteq E\) e \(\mu(G \difference E) = 0\), ed
    \(\exists F \in F_\sigma\) tale che \(F \subseteq E\) e \(\mu(E \difference F) = 0\).
    Se \(\mu(E) < +\infty\), allora \(\forall \varepsilon > 0, \exists I, \cdots, I_n\)
    intervalli limitati tale che \(\mu(E \Delta \bigcup I_j) < \varepsilon\).
\end{snippetproposition}

Cioè dal punto di vista della misura posso approssimare arbitrariamente
l'insieme con un \(G_\delta\) e un \(F_\sigma\).
Possiamo ulteriormente approssimarlo con una successione finita di intervalli.
Abbiamo proprio la condizione del completamento con l'insieme che sta fra due insiemi.
Infatti devo prendere tutti gli insiemi contenenti in un boreliano e contenuti in
un borealiano tale che la misura della differenza sia nulla. 

La proposizione può essere invertita mettendo \(\mu^*\)
al posto di \(\mu\). Cioè, se \(\exists G \in G_\delta\)
tale che \(G \supseteq E\) e \(\mu^*(G \difference E) = 0\),
allora \(E \in \mathcal L\).

\begin{snippetproof}{Sketch}
    Cominciamo misurando che se \(E \in \mathcal L\) vale il resto.
    Mostriamo che \(\forall \varepsilon > 0, \exists A\) aperto
    tale che \(A \supseteq E\) e \(\mu(A \difference E) < \varepsilon\).
    Lo dimostriamo nel caso in cui \(\mu(E) < \infty\).
    Per definizione di misura esterna si può ricoprire \(E\)
    con una collezione numerabile di intervalli \(I_1, \cdots\) tale che
    \[
        E \subseteq \bigcup_{i=1}^\infty I_n, \quad \sum_{n=1}^\infty l(I_n) < \mu(E) + \varepsilon
    \]
    L'unione degli intervalli è un aperto, quindi poniamo
    \[
        A = \bigcup_{i=1}^\infty I_i
    \]
    Abbiamo costruito un aperto tale che \(\mu(A \difference E) < \varepsilon\).
    Infatti, 
    \begin{align*}
        \mu(A) &\leq \sum_{n=1}^\infty l(I_n) < \mu(E) + \varepsilon \\
        \mu(A) &< \mu(E) + \varepsilon, \quad A \supseteq E
    \end{align*}
    Abbiamo che
    \[
        A = (A \difference E) \union E
        \implies \mu(A) \leq \mu(E) + \mu(A \difference E)
    \]
    per la subadditività.
    Tuttavia, per dedurre che \(\mu(A \difference E) < \varepsilon\)
    mi serve l'uguaglianza. Qui usiamo il fatto che \(E\) è misurabile e quindi
    usiamo la condizione di Caratheodory.
    Mi serve un \(G_\delta\). Scegliamo \(\varepsilon = \frac{1}{n}\).
    Abbiamo costruito una famiglia di aperti \(\{A_n\}\)
    tali che \(A_n \supseteq E\) e \(\mu(A_n \difference E) < \frac{1}{n}\).
    Basta porre \(G = \bigcap_{n=1}^\infty A_n\).
    Sicuramente \(G \supseteq E\), e inoltre
    \[
        \mu(G \difference E) \leq \mu(A_n \leq E) = \frac{1}{n}
    \]
    per monotonia, e sicocme vale per ogni \(n\) allora \(\mu(G \difference E) = 0\).
    Se \(\mu(E)\) fosse infinite non potremmo spostare \(\mu(E)\) a sinistra in \(\mu(A) < \mu(E) + \varepsilon\)
    per esempio.
    Nel caso di un insieme di misura infinite, spezziamo retta reale in tanti intervalli
    \([n,n+1]\), e consideriamo \(E_n = E \intersection [n,n+1]\).
    Quindi stiamo spezzettando \(E\) numerabilmente in insiemi limitati. Ma allora per ciascuno di questi,
    possiamo applicare il ragionamento precedente e scegliere un \(G_n \in G_\delta\)
    per ogni pezzo. Adesso tuttavia non possiamo fare l'unione
    numerabili di questi \(G_n\), perché non è in generale un \(G_\delta\),
    sarebbe un \(G_{\delta\sigma}\). Per risolvere questo problema,
    al posto di definire \(G\) come intersezione di aperti e poi unirli tutti,
    cominciamo unendo gli aperti (che rimane un aperto), e alla fine facciamo l'intersezione di tutti.
    Ripetiamo quindi la costruzione su ogni singolo \(E_k\),
    sostituendo \(\varepsilon\) con \(\sfrac{\varepsilon}{2^k}\).
    Unendo gli \(A_k\) abbiamo ancora un aperto e \[
        \mu\left((\bigcup A_k) \difference E\right) < \varepsilon
    \]
    perché sommo su \(k\).
    Adesso prendiamo come prima \(\varepsilon = \sfrac1n\),
    definiamo i \(G_\delta\) come prima, come intersezione di questi aperti.
    Invece, per costruire gli \(F_\sigma\) passiamo al complementare,
    quindi riapplichiamo il primo punto ad \(E^c\).
    Poniamo quindi \(F = G^c\) e \(F\) è un \(F_\sigma \subseteq E^c\).
    In generale vale sempre che \(E^C \difference F = G \difference E\),
    e quindi \(\mu(E^c \difference F) = 0\). \\
    \textbf{Ultimo punto:} (Esercizio).
\end{snippetproof}

L'ultim punto con gli intervalli finiti ci dice che la patologia della misura di Lebesgue
(che è tanta e varia) può essere confinata in un insieme di misura piccola: a meno di
insiemi di misura \(\varepsilon\), ogni insieme di misura finita
è un'unione finita di intervalli. La patologia sta nell'\(\varepsilon\).

\begin{snippetexample}{}
    Se considero in \(\realnumbers^2\) una area sottesa alla curva gaussiana,
    è illimitata ma ha misura finita. Posso ricoprirla con un numero finito di rettangoli
    a meno di un'area con misura arbitrariamente piccola.
    Invece, se l'area ha misura infinita come una striscia del piano, la parte che non riesco a coprire
    con gli intervalli ha sempre misura infinita, non riesco ad approssimarla a meno di \(\varepsilon\).
    Tagliando le code, che hanno misura piccola,
    quello che resta è un limitato.
\end{snippetexample}

Abbiamo visto che \(\forall \varepsilon > 0\)
esiste \(F \subseteq E\) chiuso tale che \(\mu(E \difference F) < \varepsilon\).
Questa affermazione non è così intuitiva.
Consideriamo \(E = [0,1] \intersection \rationalnumbers\).
Per la proprietà, esiste un compatto contenuto in \(E\)
con misura magiore \(1-\varepsilon\).
Possiamo costruire tale compatto in maniera esplicita
nella maniera seguente:
sia \(\{q_n\}\) una successione che enumera i razionali
e consideriamo gli intervalli
\[
    \left(
        q_n - \frac{\varepsilon}{2^{n+1}},
        q_n + \frac{\varepsilon}{2^{n+1}}
    \right)
\]
La loro unione è un aperto che contiene tutti i razionali e con misura minore di \(\varepsilon\).
Quindi, il suo complementare intersecato con \([0,1]\) è un compatto
contenuto in \(E\) con misura maggiore di \(1-\varepsilon\).

\subsection{Costruzione dell'integrale}

Cominciamo da una funzione semplice, che si può scrivere con un numero finito di valori
\[
    s = \sum_{i=1}^n \alpha_i \chi_{s^{-1}(\alpha_i)}
\]
Se \(\alpha_i \geq 0\), allora \(s\) è non-negativa.
Se \(s^{-1}(\alpha_i)\) sono misurabili, allora \(s\) è misurabile.

\begin{snippetdefinition}{Integrale di funzione semplice non-negativa}
    Sia \((X, \mathcal M, \mu)\) uno spazio di misura e sia
    \(s\) una funzione semplice misurabile non-negativa. Sia \(\Omega \subseteq X\).
    Definiamo l'\emph{integrale di \(s\)} nello spazio come
    \[
        \int_\Omega s\,d\mu \triangleq
        \sum_{i=1}^n \alpha_i \mu(s^{-1}(\alpha_i) \intersection \Omega)
    \]
\end{snippetdefinition}

Nell'integrale di Riemann sommiamo su intervalli del dominio,
mentre qua sommiamo sulle fibre del codominio.
In questo contesto se un coefficiente è nullo ma la misura
è infinita, il prodotto fa zero. Non può succedere \(\infty-\infty\).

Nei reali con la misura di Lebesgue, i plurirettangolo di Riemann
sono funzioni semplici misurabili e il loro integrare coincide con quello di Riemann,
quindi stiamo facendo una generalizzazione che include la teoria classica.

\begin{snippetdefinition}{Integrale di funzione non-negativa}
    Sia \(f \colon X \fromto [0, +\infty]\)
    una funzione numerabile
    su uno spazio di misura \((X, \mathcal M, \mu)\). Sia \(\Omega \subseteq X\).
    Allora definiamo
    \[
        \int_\Omega s\,d\mu \triangleq
        \sup_{s \leq f} \int_\Omega s\,d\mu
    \]
\end{snippetdefinition}

In realtà possiamo definire gli integrali solo sull'insieme totale,
senza intersecare. Se facessimo ciò, ogni qualvolta avessimo un sottoinsieme,
la restrizione della misura e dell'algebra a questo sottoinsieme rimane
uno spazio di misura \((E, \restr{\mathcal M}{E}, \restr{\mu}{E})\),
dove \(\restr{\mathcal M}{E}\) è la \(\sigma\)-algebra ottenuta dalle intersezioni di \(E\)
con gli insiemi di \(\mathcal M\).

Vogliamo assicurarci che se \(f\) è semplice allora il nuovo integrale coincide
con quello vecchio. Ma ciò è banale in quanto la funzione semplice stessa è la funzione
semplice più grande che approssima sè stessa.

\begin{snippetproposition}{}
    \begin{enumerate}
        \item se \(f \leq g\) allora
        \[
            \int_E f\,d\mu \leq \int_E g\,d\mu
        \]
        \item se \(E \supseteq F\)
        \[
            \int_E f\,d\mu \geq \int_f f\,d\mu
        \]
        perché \(f\) è positiva.
        \item sia \(\alpha \in \realnumbers^+\)
        \[
            \int_E \alpha f\,d\mu
            = \alpha \int_E f\,d\mu
        \]
        \item se \(f=0\) su \(E\), allora
        \[
            \int_E f\,d\mu = 0
        \]
        anche se \(\mu(E) = \infty\).
        \item se \(\mu(E)\)
        \[
            \int_E f\,d\mu = 0
        \]
        anche se \(f = \infty\).
        \item \[
            \int_E f\,d\mu
            = \int_X f \chi_E \,d\mu
        \]
    \end{enumerate}
\end{snippetproposition}

Anche la somma di integrali è l'integrale della somma, ma la dimostrazione non è banale.
Ci serono altri teoremi e dobbiamo definire una nuova misura.

\begin{snippetproposition}{}
    Sia \((X, \mathcal M, \mu)\)
    uno spazio di imsura,
    \(s\) una funzione semplice misurabile non-negativa.
    Sia, per \(E \in \mathcal M\)
    \[
        \varphi(E) = \int_E s\,d\mu
    \]
    \(\varphi\) è una nuova misura sulla \(\sigma\)-algebra \(\mathcal M\).
    Stiamo quindi dando una densità \(s\) alla misura \(\mu\).
    Abbamo quindi un nuovo spazio di misura \((X, \mathcal M, \varphi)\).
    Usando la finita addività di questa misura si dimostra che
    \[
        \int_E (s+t)\,d\mu = \int_E s\,d\mu
        + \int_E t\,d\mu
    \]
    dove \(t\) è un'altra funzione semplice misurabile non-negativa.
\end{snippetproposition}

Non dimostriamo che questa è una nuova misura. Bisognerebbe dimostrare che è numerabilmente additiva.

Usando la finita addività di questa misura si dimostra che
\[
    \int_E (s+t)\,d\mu = \int_E s\,d\mu
    + \int_E t\,d\mu
\]
dove \(t\) è un'altra funzione semplice misurabile non-negativa.
Una volta fatto ciò possiamo approssimare le funzioni cno funzioni semplici
e portare il risultato a funzioni non-semplici.
Dimostriamo questa additività.

\begin{snippetproof}{Da completare i dettagli}
    Scriviamo
    \[
        s=\sum_{i=1}^n \alpha_i \chi_{A_i}, \quad
        s=\sum_{i=1}^m \beta_i \chi_{B_i}
    \]
    Poniamo \(E_{i,j} = A_i \intersection B_j\).
    Si ha che
    \[
        X = \bigsqcup_{i,j} E_{i,j}
    \]
    Quindi sicocme sono disgiunti \begin{align*}
        \int_{E_{i,j}} (s+t)\,d\mu
        &= (\alpha_i + \beta_j)\mu(E_{i,j})
        =\alpha_i \mu(E_{i,j}) + \beta_j\mu(E_{i,j}) \\
        &= \int_{E_{i,j}} s\,d\mu + \int_{E_{i,j}} t\,d\mu
    \end{align*}
    Per la finita additività della misura \(\varphi\) si ha che
    \[
        \int_X (s+t)\,d\mu
        =\int_X s\,d\mu + \int_X t\,d\mu
    \]
\end{snippetproof}

Per dimostrare l'additività dell'integrale
ci serve il passaggio al limite.

\begin{snippettheorem}{Convergenza monotona}
    Sia \((X, \mathcal M, \mu)\)
    uno spazio di misura e siano \(f_n \colon X \fromto [0, +\infty]\)
    misurabili e supponiamo che \(f_n \uparrow f\), coè \(f_n(t) \leq f_{n+1}(t)\)
    per ogni \(t \in X\). Allora, \[
        \int_X f_n\,d\mu \uparrow \int_X f\,d\mu
    \]
\end{snippettheorem}

Il fatto che la successione converga è gratis dal fatto
che la successione sia monotona, alla peggio converge in alcuni punti ad infinito.

\begin{snippetproof}{}
    Dimostriamo che esiste il limite degli integrali.
    Per la monotonia delle funzioni e per la monotonia dell'integrale, abbiamo una successione
    monotona crescente in \([0,+\infty]\)
    \[
        \int_X f_n\,d\mu \leq \int_X f_{n+1}\,d\mu
    \]
    che quindi converge ed
    \[
        \exists \lim \int_X f_n \,d\mu = \alpha \in [0,+\infty]
    \]
    Dobbiamo mostrare che \[
        \alpha \geq \int_X f\,d\mu
    \]
    Per farlo mostriamo che \[
        \alpha \geq \int_X s\,d\mu
    \]
    dove \(s\) è una funzione semplice tale che \(0 \leq s \leq f\).
    Se dimostriamo ciò, passando al sup troviamo la tesi.
    Per dimostrarlo, dimostriamo qualcosa di apparentemente più forte ma più semplice da dimostrare.
    Consideriamo \(0 \leq c \leq 1\), allora se \(s \leq f\), \(cs < f\)
    con disuguaglianza stretta. Mostriamo allora che
    \[
        \alpha \geq c \int_X s\,d\mu
    \]
    per ogni \(0 < c < 1\) come prima. Se dimostriamo questa proposizione
    allora vale \[
        \alpha \geq \int_X s\,\mu
    \]
    Fissati \(c,s\) consideriamo l'insieme \[
        E_n = \{
            x \in X \suchthat f_n(x) \geq cs(x)    
        \}
    \]
    Siccome \(\{f_n\}\) è una successione monotona crescente, questi insiemi sono tutti annidati
    \(E \subseteq E_{n+1}\). Inoltre,
    siccome \(f_n(x) \to f(x)\) per ogni \(x \in X\), ogni \(X\)
    cade definitivamente in un \(E_n\), quindi
    \[
        \bigcup_{n=1}^\infty E_n = X
    \]
    Infatti, se \(f(x)=0\) allora \(x \in E_n\) per ogni invece.
    Invece, se \(f(x) \neq 0\), \(cs(x) < f(x)\) e quindi \(\exists N\)
    tale che \(cs(x) \leq f_N(x) < f\).
    Per la monotonia sul domiino,
    \begin{align*}
        \int_X f_n \,d\mu &\geq \int_{E_n}f_n\,d\mu \geq \int_{E_n} cs(x)\,d\mu = c \int_{E_n} s\,d\mu
    \end{align*}
    La funzione \(s\) è fissata e
    \[
        \int_{E_n} s\,d\mu = \varphi(E_n) \to \varphi(X) = \int_X s \,d\mu
    \]
    per la proprietà delle misure di insiemi scatolate che sono il limite.
    Ma \[
        \int_X f_n\,d\mu \to \alpha
    \]
    quindi
    \[
        \alpha \geq c \int_X s\,d\mu
    \]
    che è quello che volevamo.
\end{snippetproof}

\begin{snippetexample}{L'ipotesi di monotonia è strettamente necessaria}
    Consideriamo \((\realnumbers, \mathcal L, \mu)\)
    e consideriamo la funzione 
    \[
        f_n = n \chi_{\left[0, \frac1n\right]}
    \]
    che converge puntualmente a zero. Tuttavia,
    \[
        \int_X f_n \,d\mu = 1 \neq \int_X f \,d\mu = 0
    \]
    Infatti, la successione non è monotona.
\end{snippetexample}

\begin{snippettheorem}{Teorema di integrazione per serie}
    Sia \(f_n \colon X \fromto [0,+\infty]\)
    misurabili. Allora
    \[
        \int_X \sum_{n=1}^\infty f_n \,d\mu
        = \sum_{n=1} \int_X f_n \,d\mu
    \]
\end{snippettheorem}

Qua dimostriamo anche l'additività dell'integrale (magari mettilo in una proposizione separata).

\begin{snippetproof}{}
    Dimostriamo che l'integrale è additivo per due funzioni.
    Una volta dimostrato, per induzione si arriva ad un numero finito di addendo, cioè alla successione delle somme parziali.
    A questo punto la successione delle somme parziali è una successione monotona crescente,
    e applicando il teorema di convergenza monotona vale la tesi.
    Mostriamo quindi che
    \[
        \int_X (f_1 + f_2)\,d\mu
        = \int_X f_1\,d\mu + \int_X f_2\,d\mu
    \]
    Per farlo sfruttiamo le funzioni semplici convergenti ad \(f_1\) ed \(f_2\)
    monotonamente, che si pososno costruire. Abbiamo quindi \(s_n^1 \uparrow f_1\)
    e \(s_n^2 \uparrow f_2\). Sappiamo che l'integrale è additivo per le funzioni semplici.
    Inoltre, \(s_n^1 + s_n^2 \uparrow f_1 + f_2\).
    Ora basta usare il teorema della convergenza monotona.
\end{snippetproof}

Il seguente è il teorema più generale.

\begin{snippetlemma}{Lemma di Fatou}
    Siano \(f_n \colon X \fromto [0,+\infty]\) funzioni misurabili. Allora,
    \[
        \liminf \int_X f_n \,d\mu
        \leq \int_X \liminf f_n \,d\mu
    \]
\end{snippetlemma}

Se avessi funzioni non-positive varrebbe lo stesso con il minore uguale e con il \(\limsup\).

\begin{snippetproof}{}
    Per definizione di limite inferiore
    \begin{align*}
        \liminf f_n = \lim_k \inf_{n \geq k} f_n
    \end{align*}
    Poniamo \[
        g_k \triangleq \inf_{n \geq k} f_n
    \]
    Abbimao che \(g_k\) è una successione monotona crescente che converge a \(\liminf f_n\).
    Le \(g_k\) sono misurabili in quanto l'infimum di funzioni misurabili è misurabile.
    Siccome \(\{g_k\}\) p monotona, \(\exists \lim g_k\) e quindi \[
        \exists \lim \int_X g_k \,d\mu
    \]
    Inoltre \(g_k \uparrow \liminf f_n\), quindi
    \[
        \int g_k\,d\mu \uparrow \int \liminf f_n\,d\mu
    \]
    per il teorema della convergenza monotona.
    Ma \(g_n \leq f_n\) in quanto \(g_n = \inf\{f_n, f_{n+1}, \cdots\}\).
    Di conseguenza per monotonia dell'integrale
    \[
        \int_X g_n\,d\mu \leq \int_X f_n\,\mu
    \] 
    Prendendo il limite inferiore di ambo i membri,
    \[
        \lim \int_X g_n \,d\mu \leq \liminf \int_X f_n\,d\mu
    \]
    in quando il membro a sinistra converge, quindi il limite inferiore coincide con il limite.
    Ma allora abbiamo la tesi.
\end{snippetproof}

Sia \((X, \mathcal M, \nu)\)
e \(f \colon X \fromto [0,+\infty]\) misurabile. Possiamo definire un nuovo spazio di misura
\((X, \mathcal M, \varphi_f)\) dove \[
    \varphi_f(E) \triangleq \int_E f\,d\mu
\]
Questa è una conseguenza del teorema della convergenza monotona.
È una misura in quanto è numerabilmente addivita:
siano \(E_i \in \mathcal M\) a due a due disgiunti. Allora,
\[
    \chi_{\bigcup_{i=1}^\infty E_i} = \sum_{i=1}^\infty \chi_{E_i}
\]
E di conseguenza
\[
    \varphi_f\left(
        \bigcup_{i=1}^\infty E_i
    \right)
    = \int_{\bigcup_{i=1}^\infty E_i} f\,d\mu
    = \int_X f \chi_{\bigcup_{i=1}^\infty E_i}\,d\mu =
    \int_X f \sum_{i=1}^\infty \chi_{E_i} \,d\mu
    = \sum_{i=1}^\infty \varphi_f(E_i)
\]
Se \(g\) è una funzione misurabile, come integriamo rispetto a questa misura?
Abbiamo
\[
    \int_X g\,d\varphi_f = \int_X fg\,d\mu
\]
Lo si dimostra prima per le caratteristiche, poi per le combinazioni linearei di caratteristiche,
cioè le funzioni semplici, e infine per le funzioni qualsiasi con la convergenza monotona.
Possiamo anche scrivere \(d\varphi = f\,d\mu\). Possiamo anche scrivere
\[
    f = \frac{d\varphi}{d\mu}
\]
questa \(f\) si chiama derivata di Radon-Nikodym.
Data una funzione \(f\) posso quindi definire la nuova misura
che è legata alle \(\mu\) da \(d\varphi = fd\mu\).

Vale pure il converso? Date due misure \(\varphi, \mu\) sulal stessa \(\sigma\)-algebra,
esiste \(f\) tale che \(d\varphi = f\,d\mu\)?
Sicuramente se \(\exists f\) tale ch \[
    \varphi(E) = \int_E f\,d\mu
\]
allora \(\mu(E) = 0 \implies \varphi(E) = 0\).

\begin{snippetdefinition}{Assolutamente continua}
    \(\varphi\) è \emph{assolutamente continua} rispetto a \(\mu\) (\(\varphi \ll \mu\))
    se vale \(\mu(E) = 0 \implies \varphi(E) = 0\)
\end{snippetdefinition}

\subsection{Integrale per funzioni di segni qualunque}

Sia \(f \colon X \fromto \realnumbers\)
dove \(X\) è uno spazio di misura. Vogliamo definire
\[
    \int f \,d\mu
    = \int f^+ \,d\mu
    - \int f^- \,d\mu
\]
Abbiamo il problema del caso \(\infty - \infty\).
Questa volta non possiamo prendere valori nei reali estesi.
Stiamo imponendo che ambo gli integrali devono essere finite,
quindi \(f^+, f^- \leq |f|\). Se \(f\) è misurabile
allora anche \(|f|\) è misurabile.

\begin{snippetdefinition}{Funzione integrabile}
    Sia \(f \colon X \fromto \realnumbers\)
    o \(f \colon X \fromto \complexnumbers\)
    dove \(X\) è uno spazio di misura.
    Diciamo che \(f\) è \emph{integrabile}
    se \[
        \int_X |f|\,d\mu < +\infty
    \]
\end{snippetdefinition}

\begin{snippetdefinition}{Integrlae per funzioni di segni qualunque}
    Sia \(f \colon X \fromto \realnumbers\)
    o \(f \colon X \fromto \complexnumbers\)
    una funzione integrabile. Definiamo allora
    \[
        \int f \,d\mu
        \triangleq \int f^+ \,d\mu
        - \int f^- \,d\mu
    \]
\end{snippetdefinition}

Nel caso di una funzione complessa, \(f = \Re f + i \Im f\).
In tal caso spezziamo la funzione
in \(\Re f^+, \Re f^-, \Im f^+, \Im f^-\).

\begin{snippetdefinition}{Insieme delle funzioni integrabili}
    \[
        \mathcal L^1((X, \mathcal M, \mu))
        \triangleq \{
            f \colon X \fromto \complexnumbers
            \suchthat f \text { misurabile e integrabile }    
        \}
    \]
\end{snippetdefinition}

\begin{snippettheorem}{}
    \(\mathcal L^1(\mu)\) è uno spazio vettoriale e l'integrale è un operatore
    funzionale lineare di questo spazio vettoriale.
\end{snippettheorem}

L'integrale è un elemento del duale algebrico di \(\mathcal L^1(\mu)\).

\begin{snippettheorem}{}
    Sia \(f \in \mathcal L^1(\mu)\). Allora,
    \[
        \left|\int_E f\,d\mu \right|
        \leq \int_E |f|\,d\mu
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    Ovvio per i reali, si può scrivere
    con un clever trick per quelle complesse.
\end{snippetproof}

Se \(f\) è misurabile e \(|f| \leq g \in \mathcal L^1(\mu)\),
allora \(f \in \mathcal L^1(\mu)\)
(per la monotonia).

\begin{snippettheorem}{Teorema della convergenza dominata di Lebesgue}
    Sia \((X, \mathcal M, \mu)\) uno spazio di misura, \(\{f_n\}\) successione
    di funzioni misurabili con \(f_n \colon X \fromto \complexnumbers\)
    e supponiamo che \(f_(x) \to f(x), \forall x \in X\).
    Inoltre, \(\exists g \in \mathcal L^1(\mu)\) tale che
    \(|f_n| \leq g\) definitivamente.
    Allora,
    \(f_{n \geq N}, f \in \mathcal L^1(\mu)\)
    e \[
        \int_X |f_m - f|\,d\mu \to 0
    \]
    e quindi \[
        \int_X f_n \,d\mu \to \int_X f \,d\mu
    \]
\end{snippettheorem}

\begin{snippetproof}{}
    L'ultima convergenza viene implicata dalla pnultima in quanto
    \[
        \int_X |f_n - f|\,d\mu \to 0
        \implies \int_X f_n\,d\mu
    \]
    in quanto \[
        \left|
            \int_X f_n\,d\mu
        \right| \leq \int_X |f_n - f|\,d\mu
    \]
    Torniamo ora all'inizio.
    Le funzioni \(f_n\) sono integrabili in quanto \(|f_n| \leq g\) che è integrabile, quindi sarà finito anche
    l'integrale del modulo.
    La funzione \(f\) è misurabile in quanto limite di funzioni misurabile, quindi misurabile,
    ed \(f\) è integrabile perché ciò vale anche per il limite.
    Abbiamo quindi \(f_n, f \in \mathcal L^1(\mu)\). Consideriamo ora
    \[
        |f_n - f| \leq |f_n| + |f| \leq 2g
    \]
    che a sua volta \(2g \in \mathcal L^1(\mu)\)
    in quanto è uno spazio vettoriale.
    Chiaramente \(2g - |f_n-f| \geq 0\), e quindi possiamo applicare il lemma di Fatou,
    notando che \(2g - |f_n - f| \to 2g\) puntualmente. Abbiamo quindi
    \begin{align*}
        \int_X \liminf (2g - |f-f_n|)\,d\mu &\leq \liminf
        \int_X (2g - |f-f_n|)\,d\mu
    \end{align*}
    Possiamo portare fuori il \(2g\) spezzando l'integrale negli integrali della somma
    e portarlo fuori dal limite inferiore
    \begin{align*}
        \int_X 2g\,d\mu
        + \liminf \left(
            -\int_X |f_n-f|\,d\mu
        \right)
        &= \int_X 2g\,d\mu - \limsup \int_X |f_n-f|\,d\mu \\
        \cancel{\int_X 2g\,d\mu} \leq \cancel{\int_X 2g\,d\mu} - \limsup \int_X |f_n-f|\,d\mu
    \end{align*}
    Se non fossero integrabili avremmo \(\infty-\infty\),
    quindi possiamo fare la semplificazione data in quanto sono finiti.
    Ci resta quindi che
    \begin{align*}
        - \limsup \int_X |f_n - f|\,d\mu &\geq 0 \\
        \limsup \int_X |f_n - f|\,d\mu &\leq 0
    \end{align*}
    Siccome stiamo integrando una successione positiva,
    e il suo limite superiore è minore o uguale a zero, deve necessariamente
    coincidere con zero.
    Quindi, ammette limite
    \[
        \lim \int_X |f_n - f|\,d\mu = 0
    \]
\end{snippetproof}

Negli esercizi dove le successioni non si comportavano bene rispetto all'integrale,
mancava la dominazione.

\begin{snippetexample}{Può succedere che l'integrale del limite convderge all'integrale senza che
    l'integrale del modulo della differenza tenda a zero.}
    Consideriamo \(f(x) = \sin(x) \chi_{(n\pi, (n+2)\pi)}(x)\).
    Abbiamo quindi un periodo di seno che si sposta verso destra sempre di più.
    Tale funzione converge puntualmente a \(f(x) = 0\).
    Abbiamo che
    \[
        \int_\realnumbers f_n\,d\mu = 0
    \]
    per ogni \(n\), e pure \[
        \int_\realnumbers fd\mu = 0
    \]
    Tuttavia,
    \[
        \int_\realnumbers |f_n-f|\,d\mu = 4
    \]
\end{snippetexample}

\begin{snippetexample}{Può succedere che l'integrale della differenza in modulo tenda a zero senza avere la dominante \(\mathcal L^1(\mu)\)}
    Consideriamo \(f_n = \sfrac{1}{n} \chi_{(n,n+1)}\).
    Abbiamo quindi un un segmento, di altezza sempre minore, che si sposta verso a destra (disgiunti).
    Tale funzione tende puntualmente (e pure uniformemente) a \(f(x) = 0\) e \[
        \int_X |f_n - 0|\,d\mu = \frac1n \to 0
    \]
    L'unico modo per dominare tutte le \(f_n\) è quello di dominare tutta la scaletta.
    La funzione più piccola che fa ciò è \[
        g = \sum_{n=1}^\infty \frac1n \chi_{(n,n+1)}
    \]
    che tuttavia non è integrabile in quanto la serie armonica non converge.
\end{snippetexample}

Se voglio maggiorare delle funzioni con supporti disgiunti, la più piccola tale funzione
è la somma delle funzioni.

\begin{snippetexample}{}
    Sia \(X = \{1, \cdots, n\}\)
    con la misura del conteggio. La \(\sigma\)-algebra è quella delle parti,
    e quindi tutti gli insiemi e tutte le funzioni sono misurabili.
    Inoltre, tutte le funzioni sono integrabili,
    quindi
    \[
        \mathcal L^1(\mu) = \realnumbers^n
    \]
    oppure \(\complexnumbers^n\).
    Le funzioni sono i vettori di \(\realnumbers^n\),
    e i loro integrali sono la somma delle componenti, cioè la norma \(1\).
    Scriviamo infatti \(v = \sum \alpha_i e_i\)
    dove \(e_i = \chi_{\{i\}}\) che ha misura \(1\)
    \[
        \int_X v\,d\mu = \sum_{i=1}^n \int_X \alpha_i \chi_{\{i\}}\,d\mu
    \]
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo \(X = \naturalnumbers\)
    con la misura del conteggio. La \(\sigma\)-algebra quella delle parti.
    Non abbiamo problemi di misurabiltà.
    Le funzioni sono le successioni a valori reali (o complessi).
    Il modulo della successione è la successione dei moduli.
    Quindi, una tale funzione è integrabile se la serie associata
    alla successione converge assolutamente. In tal caso
    \[
        \int_\realnumbers f\,d\mu = \sum_{n=1}^\infty a_n
    \]
    con \(f = \{a_n\}_{n \in \naturalnumbers}\),
    che esiste in quanto la convergenza assoluta implica la convergenza.
\end{snippetexample}

\begin{snippetexample}{}
    Consideriamo \(X = \realnumbers\) con la misura di Lebesgue.
    In questo caso abbiamo il problema della misurabilità
    e non possiamos cegliere l'insieme delle parti come \(\sigma\)-algebra.
\end{snippetexample}

Un criterio di misurabilità di \(f\) è che
se l'insieme dei suoi punti di discontinuità ha misura nulla,
in quanto ogni funzione continua è integrabile.

\subsection{Confronto con integrale di Riemann}

Sia \(X = (a,b) \subseteq \realnumbers\) con la misura di Lebesgue.
Studiamo la relazione fra le funzioni integrabili rispetto a Riemann
\(R(a,b)\) e \(\mathcal L^1((a,b))\).

\begin{snippettheorem}{Immersione dell'integrale di Riemann in quello di Lebesgue}
    Se \(f\) è integrabile secondo Lebesgue,
    allora lo è pure secondo Riemann e gli integrali coincidono.
\end{snippettheorem}

Sia \(f\) integrabile secondo Riemann e mostriamo che è misurabile.
\text{Ragionamento sbagliato:} \(f\) si può approssimare
con il plurirettangolo
della somme inferiori, che è una funzione semplice misurabile minore o uguale di \(f\)
(scegliamo per semplicità \(f\) positiva). Verrebbe da dire che
queste funzioni costituiscono, raffinando la partizione, una successione
di funzioni semplici misurabili che converge puntualmente ad \(f\).
Questo è falso. Scegliamo per esempio \(\tilde x \in (a,b)\)
e sia \(f = \chi_{\{\tilde x\}}\).
In questo caso tutte le funzioni semplici sono nulle (del plurirettangolo inferiore).
Quindi, in nessun caso convergono a \(1\) in quel punto.
Potremmo avere esempi patologici con tanti punti del genere, anche con la cardinalità del continuo
ma misura nulla, facendo rimanere la funzione Riemann integrabile.

\begin{snippetproof}{}
    Una funzione \(f\) è Riemann integrabile su \((a,b)\)
    se e solo se è limitata e l'insieme dei punti di discontinuità ha misura nulla.
\end{snippetproof}

Questo teorema è stato dimostrato molto prima dell'invenzione della misura di Lebesgue,
questo è un argomento a favore che Lebesgue sia il modo naturale di integrare.
Questo criterio è un criterio di Riemann-integrabilità:
se una funzione è meno discontinua di una funzione Riemann-integrabile, allora lo è pure lei.
Il criterio sta in realtà usando la completezza della misura di Lebesgue.
Infatti, usa il fatto che se \(D_f \subseteq D_g\)
e \(\mu(D_g) = 0\) allora \(\mu(D_f) = 0\),
dove \(D\) sono gli insiemi di discontinuità.


Esistono funzioni Riemann-integrabili che sono discontinue su un insieme con la cardinalità
del continuo 
In generale dato un insieme, non esiste una funzione
il cui insieme dei punti di discontinuità è tale insieme.
Per esempio non esistono funzioni discontinue esattamente su \(\realnumbers \difference \rationalnumbers\).
\begin{snippetexercise}{}
    Costruire una funzione che è discontinua solo su tutti i razionali.
    Devono essere degli \(F_\sigma\) come \(\rationalnumbers\),
    invece \(\realnumbers\) è un \(G_\delta\)
    come abbiamo già visto con il teorema di Baire.
\end{snippetexercise}
La funzione di Dirichlet è discontinua ovunque e non va bene.
\begin{snippetexercise}{}
    Dimostra che se \(f \colon \realnumbers \to \realnumbers\),
    \(D_f \in F_\sigma\) (ovvero \(C_f \in G_\delta\))
    vale che \(\forall D \in F_\sigma, \exists f\) tale che \(D = D_f\).
\end{snippetexercise}
Cioè possiamo sempre trovare una funzione il cui insieme di discontinuità
dato se questo è un \(F_\sigma\).

\begin{snippetdefinition}{Misura di Borel}
    Diciamo che una misura è di Borel se contiene la topologia
    \(\mathcal M \supseteq \tau\).
\end{snippetdefinition}
Questa non deve essere necessariamente la \(\sigma\)-algebra di Borel.

Consideriamo ora \((X, \tau)\) spazio topologico
che è anche uno spazio di misura \((X, \mathcal M, \mu)\)
e supponiamo che la misura sia di Borel (come per esempio la misura di Lebesgue in quanto misura i borealiani).
Sia \(f \colon X \fromto \mathbb F\).
Vogliamo vedere se \(\mu(D_f) = 0\) implica che \(f\) sia misurabile.

\begin{snippetsolution}{Esercizio 1}
    Prendiamo \(x \neq 0\) e \(y \neq -\frac{x}{2}\).
    L'equazione è esatta ed è equivalente a
    \begin{align*}
        (x^2 + 2xy)\,\text{d}y + (2xy + y^2 + 1)\,\text{d}x = 0
    \end{align*}
    Il potenziale è \(F(x,y) = x^2y + xy^2 + x + C\).
    Allora, esplicitando per \(y\) in \(F(x,y) = 0\) troviamo

\begin{snippetexercise}{}
    Consider the sequence of functions \(\{f_n\}_{n\in\naturalnumbers}\) definita da
    \[
        f_n(x) \triangleq n^2 \left(
            e^{\sfrac{x^2}{n^2}} - 1
        \right)
    \]
    \begin{enumerate}
        \item Determine the set \(D \subseteq \realnumbers\) where the sequence converges pointwise
        and the limit function \(f\).
        \item Show that \(\{f_n\}_{n \in \naturalnumbers}\) does not converge uniformly on \(D\).
        \item Determine whether the sequence of derivatives \(\{f_n'\}_{n \in \naturalnumbers}\)
        converges uniformly to \(f'\) on \(D\).
        \item Determine on which subsets of \(D\) the sequence \(\{f_n\}_{n \in \naturalnumbers}\) converges uniformly.
    \end{enumerate}
\end{snippetexercise}

\begin{snippetsolution}{}
    \begin{enumerate}
        \item Abbiamo il limite di convergenza puntuale calcolato con gli asintotici
        \[
            n^2 \left(
                e^{\sfrac{x^2}{n^2}}-1
            \right) \sim n^2 \cdot \frac{x^2}{n^2} = x^2
        \]
        Quindi la funzione limite è \(f = x^2\) e l'insieme di convergenza puntuale
        è tutto \(D = \realnumbers\).
        \item Per la convergenza uniforme abbiamo
        \begin{align*}
            \sup_{x\in \realnumbers} \left|
                n^2 \left(
                    e^{\sfrac{x^2}{n^2}}-1
                \right)
            \right|
        \end{align*}
        Ma \(f(n) = n^2(e-2)\), quindi il supremum è maggiore o uguale di \(n^2(e-2)\)
        che per \(n \to \infty\) non tende a zero. Quindi non abbiamo convergenza uniforme su \(D\).
        \item La derivata del termine è
        \[
            f_n'(x) = 2x e^{\sfrac{x^2}{n^2}}
        \]
        che converge puntualmente a
        \[
            \lim_n f_n'(x) = 2x = \frac{d}{dx}(x^2)
        \]
        Quindi \(\{f'\}\) converge puntualmente alla derivata \(f'\)
        su \(D\). Per la convergenza uniforme abbiamo
        \begin{align*}
            \sup_{x\in\realnumbers} \left|2x e^{\sfrac{x^2}{n^2}} - 2x\right|
            &= \sup_{x\in\realnumbers} 2|x| \left|
                e^{\sfrac{x^2}{n^2}} - 1
            \right| = \lim_n 2|x| \left|
                e^{\sfrac{x^2}{n^2}} - 1
            \right| = +\infty
        \end{align*}
        Quindi non abbiamo convergenza uniforme su \(D\).
        \item Mostriamo che \(f_n\) converge uniformemente su ogni \(A \subseteq D\) limitato.
        Senza perdita di generalità sia \(A = (a,b)\) con \(a,b \realnumbers\) e \(a<b\).
        Abbiamo quindi \begin{align*}
            \sup_{x\in A} \left|
                2xe^{\sfrac{x^2}{n^2}} - 2x
            \right|
            &= \sup_{x\in A} 2|x| \left|e^{\sfrac{x^2}{n^2}} - 1\right| \\
            &\leq \max\{
                2C\left(
                    e^{\sfrac{C^2}{n^2}} - 1
                \right), 2C
            \}, \quad C = \sup\{|a| \suchthat a \in A\}
        \end{align*}
    \end{enumerate}
\end{snippetsolution}

\begin{snippetexercise}{}
    Consider the following series of functions
    \[
        \sum_{n=0}^{+\infty} \frac{\cos(2nx)}{n!}
    \]
    \begin{enumerate}
        \item Show that the series converges uniformly on \(\realnumbers\), denoting with \(f\) its sum.
        \item Show that the su \(f \colon \realnumbers \fromto \realnumbers\) is infinitely differentiable.
        \item Compute \(f\).
    \end{enumerate}
\end{snippetexercise}

\begin{snippetproof}{}
    \begin{enumerate}
        \item Abbiamo la maggioranza con la convergenza totale
        \[
            \sum_{n=0}^{+\infty} \frac{\cos(2nx)}{n!}
            \leq \sum_{n=0}^{+\infty} \frac{|\cos(2nx)|}{n!}
            \leq \sum_{n=0}^\infty \frac{1}{n!} < \infty
        \]
        Quindi abbiamo convergenza totale e quindi pure uniforme.
        \item Scriviamo le somme parziali
        \[
            S_N = \sum_{n=0}^N \frac{\cos(2nx)}{n!}
        \]
        e studiamo le somme parziali delle derivate \(k\)-esima
        \[
            |S_N^{(k)}| = \sum_{n=0}^N \frac{(2n)^k |g_k(x)|}{n!},
            \quad g_k(x) = \begin{cases}
                \sin(2nx) & k = 2l + 1, l \in \naturalnumbers \\
                \sin(2nx) & k = 2k, l \in \naturalnumbers
            \end{cases}
        \]
        E come prima possiamo maggiorare la sommatoria
        \[
            |S_N^{(k)}| \leq \sum_{n=0}^N \frac{(2n)^k}{n!} \leq \sum_{n=0}^\infty \frac{(2n)^k}{n!} \leq
            L_k < +\infty
        \]
        Quindi la serie delle derivata \(k\)-esima converge totalmente, quindi uniformemente,
        e quindi \(f\) è \(\mathcal C^k\) per ogni \(k\).
        Dunque \(f\) è liscia.
        \item We have \begin{DispWithArrows*}
            \sum_{n=0}^{+\infty} \frac{\cos(2nx)}{n!}
            &= \sum_{n=0}^{+\infty} \Re \frac{e^{2inx}}{n!}
            = \Re \sum_{n=0}^{+\infty} \frac{{\left(e^{2ix}\right)}^n}{n!} \Arrow{Taylor series \\ of \(e^x\)} \\
            &= \Re e^{e^{2ix}} = \Re \left(
                e^{\cos(2x)}\cdot e^{i\sin(2x)}
            \right) \\
            &= \Re \left(
                e^{\cos(2x)} \left[
                    \cos(\sin(2x)) + i\sin(\sin(2x))
                \right]
            \right) \\
            &= e^{\cos(2x)}\cos(\sin(2x))
        \end{DispWithArrows*}
        Possiamo estrapolare la parte reale fuori dalla serie perché converge uniformemente.
    \end{enumerate}
\end{snippetproof}

\begin{snippetexercise}{}
    Consider the power series
    \[
        \sum_{n=0}^\infty n^2 x^n
    \]
    \begin{enumerate}
        \item Determine the set \(S\) of pointwise convergence and determine whether the convergence is uniform on \(S\).
        \item Find the sum of the series.
    \end{enumerate}
\end{snippetexercise}

\begin{snippetsolution}{}
    Sia \(a_n = n^2\).
    \begin{enumerate}
        \item Usando il criterio del rapporto, troviamo il raggio di convergenza
        \[
            \frac1r = \lim_n \left|\frac{a_{n+1}}{a_{n}}\right|
            = \lim_n \left|
                \frac{(n+1)^2}{n^2}
            \right| = 1
        \]
        Quindi \(R=1\).
        Analizziamo ora la frontiera.
        Se \(x = \pm 1\), il termine generale \(a_nx^n = (-1)^n n^2\)
        che non tendono a \(0\), quindi la serie non converge.
        Dunque, l'insieme di convergenza è \(S = (0,1)\).
        \item \begin{align*}
            \sum_{n=0}^\infty n^2x^n &=
            \left(
                \sum_{n=0}^\infty (n+1)(n+2)x^n
            \right)
            - \left(
                \sum_{n=0}^\infty (3n+2)x^n
            \right) \\
            &= \left(
                \sum_{n=0}^\infty \frac{d^2}{dx^2} x^{n+2}
            \right)
            - 3\left(
                \sum_{n=0}^\infty (n+1)x^n
            \right)
            + \sum_{n=0}^\infty x^n \\
            &= \frac{d^2}{dx^2}\left(
                \sum_{n=0}^\infty x^{n+2}
            \right)
            - 3 \frac{d}{dx} \left(
                \sum_{n=0}^\infty x^{n+1}
            \right) + \sum_{n=0}^\infty x^n \\
            &= \frac{d^2}{dx^2} \left(
                \frac{1}{1-x} - 1 - x
            \right)
            - 3 \frac{d}{dx} \left(\frac{1}{1-x} - 1\right)
            + \frac{1}{1-x} \\
            &= \frac{2-3+3x + 1 - 2x + x^2}{(1-x)^3} = \frac{x(x+1)}{(1-x)^3}
        \end{align*}
    \end{enumerate}
\end{snippetsolution}
